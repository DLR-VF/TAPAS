73  FileInputStream fis = new FileInputStream("/proc/stat");
74  InputStreamReader inS = new InputStreamReader(fis);

76 
77  //first line is the overall cpu performance
78  String line = in.readLine();
79  in.close();
80  inS.close();
81  fis.close();
82  String tmp;
83  if (!"".equals(line)) {
84  StringTokenizer st = new StringTokenizer(line);
85  //values are: cpu <user-time> <nice-time> <kernel-time> <idle-time> ...
86  if (st.countTokens() > 4) {
87  st.nextToken(); // cpu id (ignore)
88  tmp = st.nextToken(); // user-time
89  procTime += Double.parseDouble(tmp);
90  tmp = st.nextToken(); // nice-time
91  procTime += Double.parseDouble(tmp);
92  tmp = st.nextToken(); // kernel-time
93  procTime += Double.parseDouble(tmp);
94  tmp = st.nextToken(); // idle time
95  totalTime = procTime + Double.parseDouble(tmp);
96  if (this.firstRun) {
97  cpu = 0;
98  this.firstRun = false;
99  } else {
100  cpu = (procTime - this.lastProcTime) / (totalTime - this.lastTotalTime);
101  }
102  this.lastTotalTime = totalTime;
103  this.lastProcTime = procTime;
104  }
105  }
106 
107  return cpu;
108  }
