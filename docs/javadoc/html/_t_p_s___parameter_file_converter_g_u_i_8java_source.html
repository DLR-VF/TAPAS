446  private boolean isSimulationStartedAlready(Connection con) throws SQLException {

449  this.parameterClass.getString(ParamString.DB_TABLE_SIMULATIONS) + " WHERE sim_key = '" + mSimKey + "'");
450  while (resultSet.next()) {
451  // iteriere über alle Simulationen mit diesem simKey. Im Regelfall
452  // ist dies nur eine Zeile.
453  boolean simStarted = resultSet.getBoolean("sim_started");
454  boolean simFinished = resultSet.getBoolean("sim_finished");
455  // Sobald die Simulation bereits gestartet oder beendet ist, wird
456  // true zurückgegeben
457  if (simStarted || simFinished) {
