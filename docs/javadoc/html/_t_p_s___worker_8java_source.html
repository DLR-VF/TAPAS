75  public static List<Integer[]> getAllLists(List<Integer> elements, int lengthOfVector) {
76  List<Integer[]> allList = new LinkedList<>();
77  if (lengthOfVector < 0) { //no permutations
78  System.err.println(
79  "Du willst eine Permutation mit einer LÃ¤nge von " + lengthOfVector + "! Das geht nicht!");
80  return allList;
81  }
82  if (lengthOfVector == 0) { //no permutations
83  return allList;
84  } else if (lengthOfVector == 1) { //lists of length 1 are just the list of elements
85  for (Integer i : elements) {
86  Integer[] val = new Integer[lengthOfVector];
87  val[0] = i;
88  allList.add(val);
89  }
90  } else {
91  //the recursion--get all lists of length 3, length 2, all the way up to 1
92  List<Integer[]> allSublists = getAllLists(elements, lengthOfVector - 1);
93  //append the sublists to each element
94  for (Integer i : elements) {
95  //add the newly appended combination to the list
96  for (Integer[] j : allSublists) {
97  Integer[] val = new Integer[lengthOfVector];
98  val[0] = i;
99  //copy the remainder elements
100  System.arraycopy(j, 0, val, 1, lengthOfVector - 1);
101  allList.add(val);
102  }
103  }
104  }
105  return allList;
106  }
