242  private boolean checkMatrixMap() {
243  boolean result = true, changed = true;
244  double tmpD;
245  String tmpS;
246  //sort entries: use bubble sort
247  for (int j = 0; j < this.matrixEntries.length && changed; ++j) {
248  changed = false;
249  for (int i = this.matrixEntries.length - 1; i > j; --i) {
250  if (this.distribution[i] < this.distribution[i - 1]) {
251  //swap the entries in distribution and matrixEntries
252  changed = true;
253  tmpD = this.distribution[i];
254  this.distribution[i] = this.distribution[i - 1];
255  this.distribution[i - 1] = tmpD;
256  tmpS = this.matrixEntries[i];
257  this.matrixEntries[i] = this.matrixEntries[i - 1];
258  this.matrixEntries[i - 1] = tmpS;
259  } else if (this.distribution[i] == this.distribution[i - 1]) { // perform check for same time entries
260  result = false;
261  }
262  }
263  }
264  return result;
265  }

601  private void loadMatrixMapEntries() {

606  "SELECT * FROM core." + this.dbName + " WHERE \"matrixMap_name\" = '" + this.entryName + "'";

608  if (rs.next()) {

622  this.distribution[j] = Double.parseDouble(dist[j].toString());
623  }

626  }
