114  private String capStupidAmendments(String number) {
115  String returnString = "";
116  if (number != null && number.length() > 0) {
117  number = number.replaceAll(" ", ""); //remove spaces
118  boolean advanceToNextPos = true;
119  int pos = 0;
120  while (advanceToNextPos) {
121  switch (number.charAt(pos)) {
122  case '0':
123  case '1':
124  case '2':
125  case '3':
126  case '4':
127  case '5':
128  case '6':
129  case '7':
130  case '8':
131  case '9':
132  case '-':
133  //a number! yeh!
134  if (pos + 1 == number.length()) advanceToNextPos = false;
135  else pos++;
136  break;
137  default:
138  advanceToNextPos = false;
139  break;
140  }
141  }
142  if (pos + 1 == number.length() || pos == number.length()) returnString = number;
143  else returnString = number.substring(0, pos).trim();
144  }
145  return returnString;
146  }

149  private boolean checkCharacterAfterTag(String input, String tag) {
150  if (input.startsWith(tag)) {
151  if (input.length() == tag.length()) // nothing behind the tag
152  return true;
153  char c = input.charAt(tag.length());
154  //legal character?
155  return c == ' ' || c == '1' || c == '2' || c == '3' || c == '4' || c == '5' || c == '6' || c == '7' ||
156  c == '8' || c == '9' || c == '0';
157  } else {
158  return false;
159  }
