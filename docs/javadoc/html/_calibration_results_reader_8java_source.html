205  @SuppressWarnings("rawtypes")
206  public HashMap<Mode, Double> getModalSplit(Enum... categories) {

208  validate(categories);

210  HashMap<Mode, Double> result = new HashMap<>();
211 
212  String wc = buildWhereClause(categories);
213 
214  // build full query
215  String q = "SELECT trip_mode,SUM(cnt_trips) AS sum" + " FROM calibration_results WHERE sim_key = '" + simkey +
216  "' " + wc + " GROUP BY trip_mode" + " ORDER BY trip_mode";
217 
218  try {
219  ResultSet rs = dbCon.getConnection(this).createStatement().executeQuery(q);
220  double sum = 0;
221  while (rs.next()) {
222  Mode mo = Mode.getById(rs.getInt("trip_mode"));
223  Double v = (double) rs.getLong("sum");
224  result.put(mo, v);
225  sum += v;
226  }
227  rs.close();
228 
229  if (sum == 0) return result;
230 
231  for (Entry<Mode, Double> e : result.entrySet()) {
232  result.put(e.getKey(), e.getValue() / sum);
233  }
234  } catch (SQLException e) {
235  e.printStackTrace();
236  return null;
237  }
