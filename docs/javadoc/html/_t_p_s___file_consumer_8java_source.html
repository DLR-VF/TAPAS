113  private void consumeFile(boolean hasHeader) throws Exception {
114  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
115  TPS_Logger.log(SeverenceLogLevel.INFO, "Opened file: " + this.getFileName());
116  }
117 
118  if (hasHeader) this.init(this.reader.getHeaders());
119  else this.init(null);
120 
121  String[] record;
122  int lineCount = 0;
123  SortedMap<Integer, Exception> exMap = new TreeMap<>();
124 
125  while ((record = reader.getRecord()) != null) {
126  if (record.length == 0 || record[0].length() == 0) {
127  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {
128  TPS_Logger.log(SeverenceLogLevel.WARN,
129  "Skipped empty line in file " + this.getShortFileName() + " in line " + (lineCount + 1) +
130  ": " + Arrays.toString(record));
131  }
132  } else {
133  try {
134  if (this.getRecordLength() > 0) {
135  if (record.length < this.getRecordLength()) {
136  throw new IOException(
137  "Record in file " + this.getShortFileName() + " in line " + (lineCount + 1) + " " +
138  Arrays.toString(record) + " is too short: " + this.getRecordLength() +
139  "(requested) > " + record.length);
140  } else if (record.length > this.getRecordLength()) {
141  if (TPS_Logger.isLogging(SeverenceLogLevel.FINE)) {
142  TPS_Logger.log(SeverenceLogLevel.FINE,
143  "Record in file " + this.getShortFileName() + " in line " + (lineCount + 1) +
144  " " + Arrays.toString(record) + " is too long: " +
145  this.getRecordLength() + "(requested) < " + record.length);
146  }
147  }
148  }
149  this.consume(record, lineCount);
150  } catch (Exception e) {
151  // An exception is thrown when the whole file was read
152  exMap.put(lineCount, e);
153  if (exMap.size() > 10) {
154  throw new IOException("At least one exception occured while reading file: " +
155  new File(this.getFileName()).getName() + "\n" + createException(exMap));
156  }
157  }
158  lineCount++;
159  }
160  }
161 
162  if (exMap.size() > 0) {
163  throw new IOException(
164  "At least one exception occured while reading file: " + new File(this.getFileName()).getName() +
165  "\n" + createException(exMap));
166  }
167 
168  this.finish(lineCount);
169  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
170  TPS_Logger.log(SeverenceLogLevel.INFO,
171  "Read " + lineCount + " entries of file: " + new File(this.getFileName()).getName());
172  }
173  }

181  private String createException(SortedMap<Integer, Exception> exMap) {
182  StringWriter sw = new StringWriter();
183  PrintWriter pw = new PrintWriter(sw);
184  for (Integer key : exMap.keySet()) {
185  pw.write("Exception happened in line " + key + ":\n");
186  Exception ex = exMap.get(key);
187  ex.printStackTrace(pw);
188  }
189  pw.flush();
190  pw.close();
191  return sw.getBuffer().toString();
192  }
