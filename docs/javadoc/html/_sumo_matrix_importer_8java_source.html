84  public void readSumoValues(String schema, String table) {

102

106  distance = new Matrix(numOfTAZ, numOfTAZ);
107 
108  //get the values
109  query = "SELECT taz_id_start, taz_id_end, travel_time_sec, distance_real from " + tableName;
110 
111  int from, to;
112  rs = dbCon.executeQuery(query, this);
113  while (rs.next()) {
114  from = this.tazMap.get(rs.getInt("taz_id_start"));
115  to = this.tazMap.get(rs.getInt("taz_id_end"));
116  tt = rs.getDouble("travel_time_sec");
117  dist = rs.getDouble("distance_real");
118 
119  travelTime.setValue(from, to, Math.round(tt));
120  distance.setValue(from, to, Math.round(dist));
121  }
122 
123  double[][] valuesTT = new double[numOfTAZ][numOfTAZ];
124  double[][] valuesDist = new double[numOfTAZ][numOfTAZ];
125  int i, j;
126  for (i = 0; i < numOfTAZ; ++i) {
127  for (j = 0; j < numOfTAZ; ++j) {
128  valuesTT[i][j] = travelTime.getValue(i, j);
129  valuesDist[i][j] = distance.getValue(i, j);
130 
131  }
132  }
133 
134 
135  TPS_Geometrics.calcTop3(valuesTT);
136  TPS_Geometrics.calcTop3(valuesDist);

138  //fill in diagonal

140  for (j = 0; j < numOfTAZ; ++j) {
141  if (i == j) {
142  travelTime.setValue(i, j, valuesTT[i][j]);

157  public void writeMatrix(Matrix matrix, String matrixName) {

159  String query = "SELECT * FROM " + this.parameterClass.getString(ParamString.DB_SCHEMA_CORE) +

165  }

169  this.parameterClass.getString(ParamString.DB_TABLE_MATRICES) + " SET matrix_values = ";

171  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
172  TPS_Logger.log(SeverenceLogLevel.INFO, "Updating data for entry: " + matrixName + " in table " +
173  this.parameterClass.getString(ParamString.DB_TABLE_MATRICES) + ".");
174  }
175  } else {
176  query = "INSERT INTO " + this.parameterClass.getString(ParamString.DB_SCHEMA_CORE) +
177  this.parameterClass.getString(ParamString.DB_TABLE_MATRICES) +
178  " (matrix_name, matrix_values) VALUES ('" + matrixName + "', ";
