154  public void selectMode(TPS_Plan plan, TPS_Stay prevStay, TPS_LocatedStay locatedStay, TPS_Stay nextStay, TPS_PlanningContext pc) {
155  // log.debug("\t\t\t\t\t\t '--> In tpsSelectLocation.selectMode");
156 
157 
158  TPS_ExtMode currentArrivalMode = locatedStay.getModeArr();
159  TPS_ExtMode currentDepartureMode = locatedStay.getModeDep();
160 
161  if (currentArrivalMode == null || currentDepartureMode == null) {
162  // The episodes for activities out of home receive this call in a hierarchical order. That means that either
163  // 'comingFrom' or 'goingTo' point to an episode at home or they point to superior episodes. In the latter case
164  // the mode eventually has to be adopted.
165  TPS_LocatedStay prevStayLocated = plan.getLocatedStay(prevStay);
166  TPS_LocatedStay nextStayLocated = plan.getLocatedStay(nextStay);
167  TPS_ExtMode previousStayDepartureMode = prevStayLocated.getModeDep();
168  TPS_ExtMode nextStayArrivalMode = nextStayLocated.getModeArr();
169  TPS_Location previousStayLocation = prevStayLocated.getLocation();
170  TPS_Location currentStayLocation = locatedStay.getLocation();
171 
172  // The mode "walking" is used to get distances on the net.
173  double distanceNet = Math.max(this.parameterClass.getDoubleValue(ParamValue.MIN_DIST),
174  TPS_Mode.get(ModeType.WALK)
175  .getDistance(previousStayLocation, currentStayLocation, SimulationType.SCENARIO, null) -
176  obscureDistanceCorrectionNumber);
177  TPS_ModeChoiceContext mcc = new TPS_ModeChoiceContext();
178 
179 
180  if (nextStayArrivalMode == null) { //does the next stay allow unrestricted mode choice
181  if (previousStayDepartureMode == null) { //does the previous stay allow unrestricted mode choice
182  mcc.fromStayLocation = previousStayLocation;
183  mcc.toStayLocation = currentStayLocation;
184  mcc.toStay = locatedStay.getStay();
185  mcc.duration = mcc.toStay.getOriginalDuration();
186  mcc.startTime = mcc.toStay.getOriginalStart();
187  mcc.isBikeAvailable = pc.isBikeAvailable;
188  mcc.carForThisPlan = pc.carForThisPlan;
189  currentArrivalMode = selectMode0(plan, distanceNet, mcc);
190  } else { // adopt arrival mode according to departure mode
191  currentArrivalMode = previousStayDepartureMode;
192  }
193  if (currentArrivalMode == null) {
194  TPS_Logger.log(HierarchyLogLevel.EPISODE, SeverenceLogLevel.SEVERE,
195  "Error: no possible mode found!");
196  }
197  locatedStay.setModeArr(currentArrivalMode);
198 
199  //now we have an arrival mode
200  if (currentDepartureMode == null) {
201  if (currentArrivalMode.isFix()) { // do we have to stick to the mode
202  currentDepartureMode = currentArrivalMode;
203  } else {
204  TPS_TourPart tourpart = (TPS_TourPart) locatedStay.getStay().getSchemePart();
205  mcc.fromStayLocation = currentStayLocation;
206  mcc.toStayLocation = nextStayLocated.getLocation();
207  mcc.toStay = nextStay;
208  mcc.duration = mcc.toStay.getOriginalDuration();
209  mcc.startTime = mcc.toStay.getOriginalStart();
210  mcc.isBikeAvailable = tourpart.isBikeUsed();
211  mcc.carForThisPlan = tourpart.getCar();
212  currentDepartureMode = selectMode0(plan, distanceNet, mcc);
213  }
214  locatedStay.setModeDep(currentDepartureMode);
215  if (currentDepartureMode == null) {
216  TPS_Logger.log(HierarchyLogLevel.EPISODE, SeverenceLogLevel.SEVERE,
217  "Error: no possible mode found!");
218  }
219  }
220  } else { //adopt departure mode to the next stay arrival mode
221  currentDepartureMode = nextStayArrivalMode;
222  locatedStay.setModeDep(currentDepartureMode);
223  if (currentArrivalMode == null) { // is the arrival mode set?
224  if (previousStayDepartureMode == null) { //does the previous stay allow unrestricted mode choice
225  if (currentDepartureMode.isFix()) { // do we have to stick to the mode
226  currentArrivalMode = currentDepartureMode;
227  } else {
228  TPS_TourPart tourpart = (TPS_TourPart) locatedStay.getStay().getSchemePart();
229  mcc.fromStayLocation = previousStayLocation;
230  mcc.toStayLocation = currentStayLocation;
231  mcc.toStay = locatedStay.getStay();
232  mcc.duration = mcc.toStay.getOriginalDuration();
233  mcc.startTime = mcc.toStay.getOriginalStart();
234  mcc.isBikeAvailable = tourpart
235  .isBikeUsed(); // pc.isBikeAvailable; !!! why this - it should be same as before (pc.isBikeAvailable)
236  mcc.carForThisPlan = tourpart
237  .getCar(); // pc.carForThisPlan; !!! why this - it should be same as before (pc.carForThisPlan)
238  currentArrivalMode = selectMode0(plan, distanceNet, mcc);
239  }
240  } else {// adopt arrival mode according to departure mode
241  currentArrivalMode = previousStayDepartureMode;
242  }
243  if (currentArrivalMode == null) {
244  TPS_Logger.log(HierarchyLogLevel.EPISODE, SeverenceLogLevel.SEVERE,
245  "Error: no possible mode found!");
246  }
247  locatedStay.setModeArr(currentArrivalMode);
248  }
249  }
250  }
