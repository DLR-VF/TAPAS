30  double costsForMIV(TPS_Mode mode, TPS_Plan plan, double distanceNet, double travelTime, TPS_ModeChoiceContext mcc, SimulationType simType) {
31  double cost = 0;
32  TPS_TrafficAnalysisZone comingFromTVZ;
33  TPS_TrafficAnalysisZone goingToTVZ;
34  double tmpFactor = 1;
35  comingFromTVZ = mcc.fromStayLocation.getTrafficAnalysisZone();
36  goingToTVZ = mcc.toStayLocation.getTrafficAnalysisZone();
37  //fuel cost per km
38  if (mcc.carForThisPlan != null) {
39 
40  if (goingToTVZ.isRestricted() && mcc.carForThisPlan.isRestricted()) {
41  return Double.NaN;
42  }
43  tmpFactor = (mcc.carForThisPlan.getCostPerKilometer(simType) +
44  mcc.carForThisPlan.getVariableCostPerKilometer(simType));
45  //TODO: Bad hack to modify the costs according to the av-reduction!
46  if (mcc.carForThisPlan.getAutomation() >= mode.getParameters().getIntValue(
47  ParamValue.AUTOMATIC_VEHICLE_LEVEL) && SimulationType.SCENARIO.equals(simType)) {
48  //calculate time perception modification
49  double rampUp = mode.getParameters().getDoubleValue(ParamValue.AUTOMATIC_VEHICLE_RAMP_UP_TIME);
50  if (travelTime > rampUp) {
51  double timeMod = distanceNet > mode.getParameters().getIntValue(
52  ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_THRESHOLD) ? mode.getParameters().getDoubleValue(
53  ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_FAR) : mode.getParameters().getDoubleValue(
54  ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_NEAR);
55  tmpFactor *= timeMod; // modify the costs according to the av-reduction from the first meter!
56  }
57  }
58 
59  } else { //calc cost for a generic car
60  boolean carIsRestricted = true;
61  if (Randomizer.random() < mode.getParameters().getDoubleValue(ParamValue.PASS_PROBABILITY_HOUSEHOLD_CAR)) {
62  TPS_Car coDriver = plan.getPerson().getHousehold().getLeastRestrictedCar();
63  if (coDriver != null) { //does the household has a car???
64  carIsRestricted = plan.getPerson().getHousehold().getLeastRestrictedCar().isRestricted();
65  } else {
66  carIsRestricted = Randomizer.random() < mode.getParameters().getDoubleValue(
67  ParamValue.PASS_PROBABILITY_RESTRICTED);
68  //return Double.NaN; // Don't ride with strangers, 'Cause they're only there to do you harm
69  }
70  } else {
71  carIsRestricted = Randomizer.random() < mode.getParameters().getDoubleValue(
72  ParamValue.PASS_PROBABILITY_RESTRICTED);
73  }
74 
75  if (goingToTVZ.isRestricted() && carIsRestricted) {
76  return Double.NaN;
77  }
78 
79  tmpFactor = (mode.getCost_per_km(simType) + mode.getVariableCost_per_km(simType));
80  }
81  tmpFactor *= 0.001; //km to m
82 
83 
84  cost = distanceNet * tmpFactor;
85 
86  //should I add costs for a PT-ticket for this ride?
87  if ((goingToTVZ.isPNR() || comingFromTVZ.isPNR()) && !plan.getPerson().hasAbo()) {
88  cost += mode.getParameters().getDoubleValue(ParamValue.PNR_COST_PER_TRIP);
89  cost += mode.getParameters().getDoubleValue(ParamValue.PNR_COST_PER_HOUR) * mcc.duration *
90  2.7777777777e-4;// stay time
91  }
92 
93  // location based cost differences
94  // determine whether target zone has parking fees or toll (only relevant if entering zone)
95  // momentary situation compared to base situation
96  boolean carMustPayToll = true;
97  if (mcc.carForThisPlan != null && mcc.carForThisPlan.hasPaidToll || plan.mustPayToll) carMustPayToll = false;
98 
99  if (goingToTVZ.hasToll(simType) && carMustPayToll) {
100  // toll is relevant as entering a cordon toll zone
101  cost += goingToTVZ.getTollFee(simType);
102  }
103 
104  if (mode.getParameters().isTrue(ParamFlag.FLAG_USE_EXIT_MAUT) && !goingToTVZ.hasToll(simType) &&
105  comingFromTVZ.hasToll(simType) && carMustPayToll) {// scenario:
106  //FIXME necessary?
107  }
108  // diff parking fees
109  if (goingToTVZ.hasParkingFee(simType)
110  /*
111  &&actCode!=1&&actCode!=2&&actCode!=7 //not for work, school or university
112  */) {// scenario
113  cost += goingToTVZ.getParkingFee(simType) * mcc.duration * 2.7777777777e-4;// stay
114  }
115  return cost;
116  }

118  @Override public double getCostOfMode(TPS_Mode mode, TPS_Plan plan, double travelTime, TPS_ModeChoiceContext mcc, SimulationType simType) {
123  double cost = 0;
124  double[] parameters = this.parameterMap.get(mode);
125 
126  double expInterChanges = 0;
127  double distanceNet = mode.getDistance(mcc.fromStayLocation, mcc.toStayLocation, simType,
128  mcc.carForThisPlan); //correct the distance to the actual value!
129 
130  switch (mode.getAttribute()) {
131  case WALK:
132  cost = mode.getCost_per_km(simType) * distanceNet * 0.001;
133  break;
134  case BIKE:
135  cost = mode.getCost_per_km(simType) * distanceNet * 0.001;
136  break;
137  case MIT:
138  cost = costsForMIV(mode, plan, distanceNet, travelTime, mcc, simType);
139  break;
140  case MIT_PASS:
141  if (mode.getParameters().isTrue(ParamFlag.FLAG_CHARGE_PASSENGERS_WITH_EVERYTHING)) {
142  cost = costsForMIV(mode, plan, distanceNet, travelTime, mcc, simType);
143  } else {
144  cost = mode.getCost_per_km(simType) * distanceNet * 0.001;
145  }
146 
147  break;
148  case TAXI:
149  //Todo: HACK: No separate access /egress-times for taxi. Assumption: 5min waiting but no egress!
150  //Car has a general 4min waiting so add one minute
151  travelTime += 60; // add one more minute for TAXI
152  //TODO: make hard coded values configurable
153  /*
154  * Taxi tarif Berlin:
155  * 3€ base charge
156  * first 7 km: 1.58€
157  * after that: 1.20€
158  */
159  double baseCharge = 3.0;
160  double shortCharge = Math.min(7000, distanceNet) * 0.001 * 1.58;
161  double longCharge = Math.max(0, distanceNet - 7000) * 0.001 * 1.20;
162  cost = baseCharge + shortCharge + longCharge;
163  break;
164  case PT:
165 
166  if (!plan.getPerson().hasAbo()) {
167  if (simType == SimulationType.BASE) {
168  cost = mode.getParameters().getDoubleValue(ParamValue.PT_COST_PER_KM_BASE);
169  } else {
170  cost = mode.getParameters().getDoubleValue(ParamValue.PT_COST_PER_KM);
171  }
172  if (mode.getParameters().isFalse(ParamFlag.FLAG_FIX_PT_COSTS)) {//cost per kilometer!
173  cost *= distanceNet * 0.001;
174  }
175  }
176 
177  expInterChanges = TPS_FastMath.exp(mode.getParameters().paramMatrixMapClass
178  .getValue(ParamMatrixMap.INTERCHANGES_PT, mcc.fromStayLocation.getTAZId(),
179  mcc.toStayLocation.getTAZId(), mcc.startTime) * TPS_DB_IO.INTERCHANGE_FACTOR);
180 
181  break;
182  case TRAIN: //car sharing-faker
183  if ((mode.getParameters().isDefined(ParamFlag.FLAG_USE_CARSHARING) && mode.getParameters().isTrue(
184  ParamFlag.FLAG_USE_CARSHARING) && plan.getPerson().isCarPooler()) //cs user?
185  || (mode.getParameters().isDefined(ParamFlag.FLAG_USE_ROBOTAXI) && mode.getParameters().isTrue(
186  ParamFlag.FLAG_USE_ROBOTAXI)) // is robotaxi
187  ) { //no cs user!
188  //service area in scenario?
189  if (!mcc.toStayLocation.getTrafficAnalysisZone().isCarSharingService(SimulationType.SCENARIO) ||
190  !mcc.fromStayLocation.getTrafficAnalysisZone().isCarSharingService(
191  SimulationType.SCENARIO)) {
192  return Double.NaN;
193  }
194  // time equals MIT + 3 min more access
195  travelTime = TPS_Mode.get(ModeType.MIT).getTravelTime(mcc.fromStayLocation, mcc.toStayLocation,
196  mcc.startTime, simType, TPS_ActivityConstant.DUMMY, TPS_ActivityConstant.DUMMY,
197  plan.getPerson(), mcc.carForThisPlan);
198  travelTime += mode.getParameters().getDoubleValue(
199  ParamValue.CARSHARING_ACCESS_ADDON); //Longer Access for Carsharing
200  cost = mode.getCost_per_km(simType) * distanceNet * 0.001;
201  } else {
202  cost = Double.NaN;
203  }
204  break;
205  }
206 
207  if (Double.isNaN(cost)) return Double.NaN;
208 
209  // put together
210  boolean work = false, education = false, shopping = false, errant = false, leisure = false;
211  TPS_Stay ref = mcc.fromStay == null ? mcc.toStay : mcc.fromStay; //es muss einen geben!!!!
212  if (ref.getSchemePart().isTourPart()) { // homeparts do not have attributes listed below!
213  TPS_TourPart tourPart = (TPS_TourPart) ref.getSchemePart();
214  work = tourPart.hasWorkActivity;
215  education = tourPart.hasEducationActivity;
216  shopping = tourPart.hasShoppingActivity;
217  errant = tourPart.hasErrantActivity;
218  leisure = tourPart.hasLeisureActivity;
219  }
220 
221 
222  return parameters[0] + // mode constant
223  parameters[1] * travelTime + // beta travel time
224  parameters[2] * cost + // beta costs
225  parameters[3] * plan.getPerson().getAge() + //alter
226  parameters[4] * plan.getPerson().getAge() * plan.getPerson().getAge() + //quadratisches alter
227  parameters[5] * plan.getPerson().getHousehold().getNumberOfCars() + // anzahl autos
228  parameters[6] * expInterChanges + //umstiege (nur ÖV)
229  //ab jetzt binär-Betas, also Ja/nein
230  (plan.getPerson().mayDriveACar() ? parameters[7] : 0) + //führerschein
231  (plan.getPerson().hasAbo() ? parameters[8] : 0) + //Öffi -abo
232  (work ? parameters[9] : 0) + //tourpart mit Arbeit
233  (education ? parameters[10] : 0) + //tourpart mit Bildung
234  (shopping ? parameters[11] : 0) + //tourpart mit Einkauf
235  (errant ? parameters[12] : 0) + //tourpart mit Erledigung
236  (leisure ? parameters[13] : 0);
237  }
