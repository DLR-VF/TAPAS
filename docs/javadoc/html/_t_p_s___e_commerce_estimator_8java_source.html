320  public double getCorrectedMatrixVal(MinimalLocation start, MinimalLocation end, double[][] matrix) {
321  int startTAZIndex = start.getTAZId() - 1;
322  int endTAZIndex = end.getTAZId() - 1;
323  double result;
324  if (startTAZIndex != endTAZIndex) {
325  double beelineDistanceLoc = TPS_Geometrics.getDistance(start, end,
326  this.getParameters().getDoubleValue(ParamValue.MIN_DIST));
327  double factor = beelineDistanceLoc / beeline[startTAZIndex][endTAZIndex];
328  result = matrix[startTAZIndex][endTAZIndex] * factor;
329  } else {
330  result = matrix[startTAZIndex][endTAZIndex];
331  }
332  return result;
333  }
