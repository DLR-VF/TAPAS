119  public TPS_Car[] getAvailableCars(int start, int end) {
120  int availableCars = 0;
121  if (this.cars == null) return null;
122  for (TPS_Car car : this.cars) {
123  if (car.isAvailable(start, end)) {
124  availableCars++;
125  }
126  }
127  if (availableCars == 0) return null;
128  TPS_Car[] available = new TPS_Car[availableCars];
129  for (int i = 0, j = 0; i < this.cars.length && j < availableCars; ++i) {
130  if (this.cars[i].isAvailable(start, end)) {
131  available[j++] = this.cars[i];
132  }
133  }
134  return available;
135  }

223  public Collection<TPS_Person> getMembers(Sorting e) {

225  List<TPS_Person> agesortedPersons = new ArrayList<>(members.values());
226  //TODO why is sorted by age? It isn't, right?
227 
228  switch (e) {
229  case NONE:
230  case COST_OPTIMUM:
231  break;
232  case AGE:
233  //sort that list by age descending!
234  agesortedPersons.sort((arg0, arg1) -> {
235  return arg1.getAge() - arg0.getAge(); //age
236  });
237  break;
238 
239  case RANDOM:
240  //sort that list by random!
241  agesortedPersons.sort((arg0, arg1) -> {
242  return Math.random() < 0.5 ? -1 : 1; //random
243  });
244  break;
245  case PRIMARY_DRIVER:
246  //sort that list by age descending!
247  agesortedPersons.sort((arg0, arg1) -> arg0.primaryDriver() > arg1.primaryDriver() ? -1 : 1);
248  break;
249 
250  }
251 
252  return agesortedPersons;
