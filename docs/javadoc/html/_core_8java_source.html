82 
83  // Initial Execution of Modules

85  console.insertString(console.getLength(),
86  "Core [" + Core.currentThread().getId() + "]" + getRuntime() +
87  ": Starte it. Auswertungsmodule \n", null);
88 
89  for (CoreProcessInterface cii : modules) {
90  cii.init(outputPath, console, first);
91  }
92  }
93  // Iterative Execution of Modules
94  int countTrips = 0;
95  while (tripReader.getIterator().hasNext() && !isCancelled) {
96  TapasTrip tt = tripReader.getIterator().next();
97  for (CoreProcessInterface cii : modules) {
98  if ((++countTrips) % 250000 == 0) {
99  console.insertString(console.getLength(),
100  "Core [" + Core.currentThread().getId() + "]" + getRuntime() + ": " + (countTrips) +
101  " Trips verarbeitet\n", null);
102  }
103  cii.prepare(tripReader.getSource(), tt, this.parameterClass);
104  }
105  // System.out.println("ID: "+converter.getTrip().getIdPers()+" Job: "+converter.getTrip().getJob());
106  }
107 
108  // Final Execution of Modules
109  if (!isCancelled) {
110  console.insertString(console.getLength(),
111  "Core [" + Core.currentThread().getId() + "]" + getRuntime() +
112  ": Starte fin. Auswertungsmodule \n", null);
113  for (CoreProcessInterface cii : modules) {
114  Thread tii = new Thread(cii,
115  "Thread for ControlInputInterface implementation: " + cii.getClass().getSimpleName());
116  tii.start();
117  }
118  }
119  } else {
120  console.insertString(console.getLength(),
121  "Core [" + Core.currentThread().getId() + "]" + getRuntime() + ": Beendet \n", null);
122  String infoString = "";
123  // TODO check error message
124  if (!tripReader.getIterator().hasNext()) infoString = infoString + "Kein Tripfile angegeben.\n";
125  if (modules.size() == 0) infoString = infoString + "Keine Module angegeben.\n";
126  JOptionPane.showMessageDialog(new JPanel(), infoString, "Info", JOptionPane.INFORMATION_MESSAGE);
127  }
