43  public static Object[] checkArguments(String[] args, TPS_ArgumentType<?>... types) throws IllegalArgumentException {
44  if (types.length > args.length) {
45  throw new IllegalArgumentException(
46  "given arguments and necessary arguments have different sizes " + "-> |args|=" + args.length +
47  ", |classes|=" + types.length);
48  }
49  Object[] array = new Object[types.length];
50 
51  for (int i = 0; i < array.length; i++) {
52  if (args[i] == null || "null".equals(args[i])) {
53  if (types[i].required) throw new IllegalArgumentException(
54  "Required argument " + types[i] + " not set at position " + i);
55  array[i] = types[i].defaultObject;
56  } else if (String.class.equals(types[i].clazz)) {
57  array[i] = args[i];
58  } else if (Boolean.class.equals(types[i].clazz)) {
59  if ("true".equalsIgnoreCase(args[i]) || "false".equalsIgnoreCase(args[i])) {
60  array[i] = Boolean.parseBoolean(args[i]);
61  } else {
62  throw new IllegalArgumentException(
63  "Value '" + args[i] + "' for argument " + types[i] + " at position " + i +
64  " is no type of " + types[i].clazz.getSimpleName());
65  }
66  } else if (Integer.class.equals(types[i].clazz)) {
67  try {
68  array[i] = Integer.parseInt(args[i]);
69  } catch (Exception e) {
70  throw new IllegalArgumentException(
71  "Value '" + args[i] + "' for argument " + types[i] + " at position " + i +
72  " is no type of " + types[i].clazz.getSimpleName(), e);
73  }
74  } else if (Long.class.equals(types[i].clazz)) {
75  try {
76  array[i] = Long.parseLong(args[i]);
77  } catch (Exception e) {
78  throw new IllegalArgumentException(

80  " is no type of " + types[i].clazz.getSimpleName(), e);
81  }
82  } else if (Float.class.equals(types[i].clazz)) {

103  } else if (!file.canRead()) {

106  }

111  StringBuilder sb = new StringBuilder("[");
112  for (Enum<?> e : enumSet) {

117  sb.append(e.name() + ",");
118  }
119  if (array[i] == null) {
120  sb.setCharAt(sb.length() - 1, ']');
121  throw new RuntimeException(

123  " not found in " + sb.toString());
124  }
125  } catch (Exception e) {

127  }
