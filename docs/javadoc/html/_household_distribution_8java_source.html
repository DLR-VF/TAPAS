244  private double[] getDistribution(HashMap<Integer, ArrayList<Integer>> buildings) throws DistributionException {
245  double[] result = DEFAULT_DISTRIBUTION.clone();
246 
247  for (int i = 0; i < DEFAULT_DISTRIBUTION.length; i++) {
248  if (buildings.get(i).size() == 0 && result[i] > 0) {
249  for (int j = 0; j < DEFAULT_DISTRIBUTION.length; j++) {
250  if (i != j) {
251  result[j] = result[j] + result[j] * result[i] / (1 - result[i]);
252  }
253  }
254  result[i] = 0;
255  }
256  }
257 
258  double sum = 0;
259  for (int i = 0; i < DEFAULT_DISTRIBUTION.length; i++) {
260  result[i] *= buildings.get(i).size();
261  sum += result[i];
262  }
263  for (int i = 0; i < DEFAULT_DISTRIBUTION.length; i++) {
264  result[i] /= sum;
265  }
266 
267  // validate numbers
268 
269  sum = 0;
270  for (double d : result) {
271  if (Double.isNaN(d)) throw new DistributionException();
272  sum += d;
273  }
274 
275  if (Double.compare(Math.abs(sum - 1), MIN_PROB) > 0) throw new DistributionException();
276 
277  return result;
278  }

288  private int getLanduse(double[] locDist) {
289  double r = Math.random();
290  if (r < locDist[RESIDENTIAL]) return RESIDENTIAL;
291  else if (r < locDist[RESIDENTIAL] + locDist[MIXED]) return MIXED;
292  else if (r < locDist[RESIDENTIAL] + locDist[MIXED] + locDist[SPECIAL]) return SPECIAL;
293  else return INDUSTRIAL;
294  }

394  public void updateHouseholds(boolean verbose) {
395  for (int block : blocks) {
396  HashMap<Integer, ArrayList<Integer>> buildings = retrieveBuildingsByLanduse(block);
397  double[] locDist = DEFAULT_DISTRIBUTION;
398 
399  boolean foundAny = false;
400  for (ArrayList<Integer> al : buildings.values()) {
401  foundAny |= al.size() > 0;
402  }
403  if (!foundAny) {// no buildings found by landuse
404  // use simple approach
405  ArrayList<Integer> buildingsSimple = retrieveBuildingsSimple(block);
406 
407  if (buildingsSimple.size() == 0) {// block is empty
408  emptyBlocks.add(block);
409  addNoBuildingUpdate(block);
410  } else {// simple update
411  for (String key : keys) {
412  ArrayList<Integer> households = retrieveHouseholds(key, block);
413  for (int household : households) {
414  int building = buildingsSimple.get((int) (Math.random() * buildingsSimple.size()));
415  addRegularUpdate(building, household, key, verbose);
416  }
417  }
418  }
419  } else {// buildings found in normal approach
420  try {
421  locDist = getDistribution(buildings);
422  } catch (DistributionException e) {
423  // should not happen
424  System.err.println("Something went wrong while calculating distributions for block " + block);
425  continue;
426  }
427 
428  for (String key : keys) {
429  ArrayList<Integer> households = retrieveHouseholds(key, block);
430  for (int household : households) {
431  int type = getLanduse(locDist);
432  int building = buildings.get(type).get((int) (Math.random() * buildings.get(type).size()));
433  addRegularUpdate(building, household, key, verbose);
434  }
435  }
436  }
437  }
438  // update the rest
439  addNoBuildingUpdate(-1);
440  addRegularUpdate(-1, -1, null, true);
441  }
