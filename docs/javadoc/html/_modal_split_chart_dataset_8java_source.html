28  public ModalSplitChartDataset(ArrayList<ModalSplitData> data) {
29  data.sort(Comparator.comparingDouble(ModalSplitData::getReference));
30 
31  seriesCollection.put(Series.DATA, data);
32  addGEH(data);
33  addOptimal(data);
34  }

36  private void addGEH(ArrayList<ModalSplitData> data) {
37 
38  seriesCollection.put(Series.GEH5_HIGH, new ArrayList<>());
39  seriesCollection.put(Series.GEH5_LOW, new ArrayList<>());
40 
41  for (ModalSplitData d : data) {
42  double s = d.getReference();
43  double g5p = 0.25 * (4 * s + 5 * Math.sqrt(16 * s + 25) + 25);
44  double g5m = 0.25 * (4 * s - 5 * Math.sqrt(16 * s + 25) + 25);
45 
46  seriesCollection.get(Series.GEH5_LOW).add(new ModalSplitData(ccDummy, null, g5m, s));
47  seriesCollection.get(Series.GEH5_HIGH).add(new ModalSplitData(ccDummy, null, g5p, s));
48  }
49  }
50 
51  private void addOptimal(ArrayList<ModalSplitData> data) {
52  seriesCollection.put(Series.OPTIMAL, new ArrayList<>());
53  double max = data.get(data.size() - 1).getReference();
54  seriesCollection.get(Series.OPTIMAL).add(new ModalSplitData(ccDummy, null, 0, 0));
55  seriesCollection.get(Series.OPTIMAL).add(new ModalSplitData(ccDummy, null, max, max));
56  }
