86  protected TPS_Mode(String name, String[] attributes, boolean isFix, TPS_ParameterClass parameterClass) {
87  if (attributes.length % 3 != 0) {
88  throw new RuntimeException("Mode constant need n*3 attributes n*(name, code, type): " + attributes.length);
89  }
90  this.name = name;
91  this.parameterClass = parameterClass;
92 
93  TPS_InternalConstant<TPS_ModeCodeType> tic;
94  this.map = new EnumMap<>(TPS_ModeCodeType.class);
95 
96  for (int i = 0; i < attributes.length; i += 3) {
97  tic = new TPS_InternalConstant<>(attributes[i], Integer.parseInt(attributes[i + 1]),
98  TPS_ModeCodeType.valueOf(attributes[i + 2]));
99  this.map.put(tic.getType(), tic);
100  }
101 
102  for (TPS_ModeCodeType type : TPS_ModeCodeType.values()) {
103  if (!this.map.containsKey(type)) {
104  throw new RuntimeException(
105  "mode code for " + this.getName() + " for type " + type.name() + " not defined");
106  }
107  }
108  this.setAttribute(ModeType.valueOf(this.getName()));
109  this.setFix(isFix);
110  }
