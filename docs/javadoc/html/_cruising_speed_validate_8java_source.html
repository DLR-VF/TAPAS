93  private int calulateMinMaxValues(int index) {
94  int max;
95 
96  records[index].max = records[index].data[0][0];
97  records[index].min = records[index].max;
98  records[index].diagonalCount = 0;
99  for (int i = 0; i < records[index].data.length; i++) {
100  for (int j = 0; j < records[index].data[i].length; j++) {
101 
102  if (j >= i && (records[index].data[i][j] != 99999)) // value 99000 means, no way to this TAZ
103  {
104  records[index].diagonalCount++;
105 
106  if (records[index].max < records[index].data[i][j]) records[index].max = records[index].data[i][j];
107 
108  if (records[index].min > records[index].data[i][j]) records[index].min = records[index].data[i][j];
109  }
110  }
111  }
112  max = (int) records[index].max;
113  return (max + BINS_WIDTH - max % BINS_WIDTH) / BINS_WIDTH;
114  }
