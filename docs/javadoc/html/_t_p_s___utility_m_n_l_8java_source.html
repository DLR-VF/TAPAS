40  public double calculateDelta(TPS_Mode mode, TPS_Plan plan, double distanceNet, TPS_ModeChoiceContext mcc) {
41  double tt1 = mode.getTravelTime(mcc.fromStayLocation, mcc.toStayLocation, mcc.startTime,
42  SimulationType.SCENARIO, TPS_ActivityConstant.DUMMY, TPS_ActivityConstant.DUMMY, plan.getPerson(),
43  mcc.carForThisPlan);
44  double tt2 = 0;
45  if (mode.isUseBase()) { //differences in times
46  tt2 = mode.getTravelTime(mcc.fromStayLocation, mcc.toStayLocation, mcc.startTime, SimulationType.BASE,
47  TPS_ActivityConstant.DUMMY, TPS_ActivityConstant.DUMMY, plan.getPerson(), mcc.carForThisPlan);
48  } else {
49  tt2 = tt1;
50  }
51  double cost1 = this.getCostOfMode(mode, plan, tt1, mcc, SimulationType.SCENARIO);
52  double cost2 = this.getCostOfMode(mode, plan, tt2, mcc, SimulationType.BASE);
53  if (Double.isNaN(cost1)) {
54  return 0;
55  }
56  if (Double.isNaN(cost2)) {
57  return 0;
58  }
59  //chop to reasonable deltas
60  return Math.min(cost1 - cost2, 2.31);
61  }

72  public TPS_DiscreteDistribution<TPS_Mode> getDistributionSet(TPS_ModeSet modeSet, TPS_Plan plan, double distanceNet, TPS_ModeChoiceContext mcc) {
73  // init
74  TPS_DiscreteDistribution<TPS_Mode> dist = TPS_ModeDistribution.getDistribution(null);
75  double[] utilities = new double[dist.getValues().length];
76  double sumOfUtilities = 0;
77  // calculate utilities
78  for (int i = 0; i < utilities.length; ++i) {
79  // get the parameter set
80  TPS_Mode mode = TPS_Mode.get(TPS_Mode.MODE_TYPE_ARRAY[i]);
81  if (!mcc.isBikeAvailable && mode.isType(ModeType.BIKE) || //no bike
82  mcc.carForThisPlan == null && mode.isType(ModeType.MIT) || //no car
83  (mode.isType(ModeType.TAXI) && mode.getParameters().isFalse(ParamFlag.FLAG_USE_TAXI)) //disable TAXI
84  ) {
85  utilities[i] = minModeProbability;
86  } else {
87  //travel time
88  double travelTime = mode.getTravelTime(mcc.fromStayLocation, mcc.toStayLocation, mcc.startTime,
89  SimulationType.SCENARIO, TPS_ActivityConstant.DUMMY, TPS_ActivityConstant.DUMMY,
90  plan.getPerson(), mcc.carForThisPlan);
91  if (TPS_Mode.noConnection(travelTime)) { //no connection
92  utilities[i] = minModeProbability;
93  } else {
94  //calc the value
95  utilities[i] = this.getCostOfMode(mode, plan, travelTime, mcc,
96  SimulationType.SCENARIO);
97  if (Double.isNaN(utilities[i])) {
98  utilities[i] = minModeProbability;
99  } else {
100  utilities[i] = TPS_FastMath.exp(utilities[i]);
101  }
102  }
103  }
104  sumOfUtilities += utilities[i];
105  }
106 
107  if (sumOfUtilities > 0) {
108  // normalize
109  sumOfUtilities = 1.0 / sumOfUtilities;
110  for (int i = 0; i < utilities.length; ++i) {
111  utilities[i] *= sumOfUtilities;
112  }
113  // calc probabilities
114  for (int i = 0; i < utilities.length; ++i) {
115  //dist.setIndexedValue(TPS_Mode.get(TPS_Mode.MODE_TYPE_ARRAY[i]), utilities[i]/(sumOfUtilities-utilities[i]));
116  dist.setValueByKey(TPS_Mode.get(TPS_Mode.MODE_TYPE_ARRAY[i]),
117  utilities[i]); //according to http://en.wikipedia.org/wiki/Multinomial_logistic_regression this the mnl!
118  }
119  } else { //no possible mode!
120  // clear probabilities
121  for (int i = 0; i < utilities.length; ++i) {
122  dist.setValueByKey(TPS_Mode.get(TPS_Mode.MODE_TYPE_ARRAY[i]), 0);
123  }
124  dist.setValueByKey(TPS_Mode.get(ModeType.MIT_PASS),
125  1);// if we erased all possible modes we have to prepare an exit plan!
126  if (TPS_Logger.isLogging(HierarchyLogLevel.EPISODE, SeverenceLogLevel.SEVERE)) {
127  TPS_Logger.log(HierarchyLogLevel.EPISODE, SeverenceLogLevel.SEVERE,
128  "No possible mode! Enabling MIT_PASS!");
129  }
130  }
131 
132  boolean expertCheck = TPS_ExpertKnowledgeTree.applyExpertKnowledge(modeSet, plan, distanceNet, mcc, false,
133  dist);
134  if (!expertCheck) {
135  if (TPS_Logger.isLogging(HierarchyLogLevel.EPISODE, SeverenceLogLevel.SEVERE)) {
136  TPS_Logger.log(HierarchyLogLevel.EPISODE, SeverenceLogLevel.SEVERE, "No possible modes!");
137  dist.setValueByKey(TPS_Mode.get(ModeType.MIT_PASS), 1); // you have to find someone taking you there!
138  }
139  }
140  return dist;
141  }
