352  double relocateLocationMIVArr(TPS_Plan plan, double beelineDistance, TPS_ActivityConstant actCode, double durationStay, TPS_TrafficAnalysisZone comingFromTVZ, TPS_TrafficAnalysisZone goingToTVZ) {
353  double costs = 0.0;
354  double resultingTravelTimePlus = 0.0;
355  // determining if the destination traffic analysis zone requires paying a toll (not if trip is starting within
356  // the
357  // toll zone)
358  if (goingToTVZ.getSimulationTypeValues(SimulationType.SCENARIO).hasToll() &&
359  !comingFromTVZ.getSimulationTypeValues(SimulationType.SCENARIO).hasToll()) {// toll has to be payed
360  // -->
361  // cordon!
362  costs += goingToTVZ.getSimulationTypeValues(SimulationType.SCENARIO).getFeeToll();
363  }
364  if (goingToTVZ.getSimulationTypeValues(SimulationType.SCENARIO).hasParkingFee()) {
365  double parkingCosts = goingToTVZ.getSimulationTypeValues(SimulationType.SCENARIO).getFeeParking();
366  if (parkingCosts > 0) {
367  double stayingHours = durationStay * 2.7777777777e-4;// converting seconds into hours
368  parkingCosts = parkingCosts * stayingHours;
369  }
370  costs += parkingCosts;
371  if (TPS_Logger.isLogging(SeverenceLogLevel.FINE)) {
372  TPS_Logger.log(SeverenceLogLevel.FINE, "GoingTo hast parkingfee of" +
373  goingToTVZ.getSimulationTypeValues(SimulationType.SCENARIO).getFeeParking() + "; duration is " +
374  durationStay + "; parkingcosts: " + parkingCosts + "; sum resulting costs " + costs);
375  }
376  }
377  if (costs > 0.0) {
378  resultingTravelTimePlus = calculateVOTadds(costs, plan);
379  if (TPS_Logger.isLogging(SeverenceLogLevel.FINER)) {
380  TPS_Logger.log(SeverenceLogLevel.FINER,
381  "taz (id=" + goingToTVZ.getTAZId() + ") with costs:" + costs + " and values: " +
382  goingToTVZ.getSimulationTypeValues(SimulationType.SCENARIO));
383  }
384  }
385 
386  return resultingTravelTimePlus;
387  }

431  TPS_Location selectDefaultLocation(TPS_Plan plan, TPS_PlanningContext pc, TPS_LocatedStay locatedStay) {
432  TPS_ActivityConstant actCode = locatedStay.getEpisode().getActCode();
433  if (actCode.hasAttribute(TPS_ActivityConstantAttribute.DEFAULT)) {
434  TPS_Logger.log(SeverenceLogLevel.ERROR,
435  "Can't continue without locations for at least the default activity " +
436  this.PM.getParameters().getIntValue(ParamValue.DEFAULT_ACT_CODE_ZBE) + ". Driving home!");
437  // no location for the default activity: drive home!
438  return plan.getPerson().getHousehold().getLocation();
439  }
440  // if no location for the act type can be found; the search is done again with a default act type
441  locatedStay.getEpisode().setActCode(TPS_ActivityConstant.DEFAULT_ACTIVITY);
442  TPS_Location loc = selectLocation(plan, pc, locatedStay);
443  locatedStay.getEpisode().setActCode(actCode);

472  public TPS_Location selectLocation(TPS_Plan plan, TPS_PlanningContext pc, TPS_LocatedStay locatedStay) {
473  //if this is a working trip and the location is given select it!
474  int primaryLocId = plan.getPerson().getWorkLocationID();
475  TPS_ActivityConstant actCode = locatedStay.getStay().getActCode();
476  //is it set?
477  if (primaryLocId >= 0) {
478  if (actCode.isFix()) {
479  TPS_Location primaryLoc = this.getLocation(primaryLocId); //get location
480  if (primaryLoc != null) { // have we found a location?
481  return primaryLoc;
482  }
483  }
484  }
485 
486  // call occurs only in case of a tour part!
487  // check for existent location types for a specific activity code
488  Collection<TPS_LocationConstant> connectedLocCodesToActCode = TPS_TrafficAnalysisZone.ACTIVITY2LOCATIONS_MAP
489  .get(actCode);
490  //TPS_AbstractConstant.getConnectedConstants(actCode, TPS_LocationCode.class);
491  if (connectedLocCodesToActCode == null || connectedLocCodesToActCode.isEmpty()) {
492  TPS_Logger.log(SeverenceLogLevel.SEVERE,
493  "No location codes for activity code " + actCode.getId() + " " + "(ZBE:" +
494  actCode.getCode(TPS_ActivityCodeType.ZBE) + ") found");
495  return this.selectDefaultLocation(plan, pc, locatedStay);
496  }
497 
498  // get the location choice set
499  TPS_RegionResultSet resultSet = this.getLocationChoiceSet().getLocationRepresentatives(plan, pc, locatedStay,
500  this.PM.getParameters());
501  //locComingFrom, td.getArrivalDuration(), locGoingTo, td.getDepartureDuration(), actCode,tourpart);
502  //select a location from the choice set
503  return this.getLocationSelectModel().selectLocationFromChoiceSet(resultSet, plan, pc, locatedStay);
504  }
