45  public double getDistance(Locatable start, Locatable end, SimulationType simType, TPS_Car car) {
46  double distance;
47  double factor = 1;
48  int idStart = start.getTrafficAnalysisZone().getTAZId();
49  int idDest = end.getTrafficAnalysisZone().getTAZId();
50  double beelineDistanceLoc = TPS_Geometrics.getDistance(start, end,
51  this.getParameters().getDoubleValue(ParamValue.MIN_DIST));
52  if (idStart == idDest) {
53  // start and end locations are in the same traffic analysis zone
54  if (this.getParameters().isTrue(ParamFlag.FLAG_INTRA_INFOS_MATRIX)) {
55  // there exist information about distances inside a traffic
56  // analysis zone. This distance is retrieved from the
57  // distances matrix. The factor is 1.
58  distance = this.getParameters().paramMatrixClass.getValue(ParamMatrix.DISTANCES_STREET, idStart,
59  idDest);
60  } else {
61  // there exist no information about distances inside a traffic
62  // analysis zone. The distance is calculated by
63  // the beeline distance and a factor which is determined by the
64  // traffic analysis zone.
65  factor = start.getTrafficAnalysisZone().getSimulationTypeValues(simType).getBeelineFactorMIT();
66  distance = beelineDistanceLoc;
67  }
68  } else {
69  // if TAZes are different calculate beelinefactor
70  factor = beelineDistanceLoc / this.getParameters().paramMatrixClass.getValue(ParamMatrix.DISTANCES_BL,
71  idStart, idDest);
72  // start and end locations are in different traffic analysis zones.
73  // The distance is calculated by a distance
74  // retrieved from the distances matrix and a factor calculated by
75  // the beeline and the real distance of both
76  // traffic analysis zones.
77  distance = this.getParameters().paramMatrixClass.getValue(ParamMatrix.DISTANCES_STREET, idStart, idDest);
78  }
79 
80 // //autonomous vehicles get a virtual distance correction for adjusting the costs
81 // if(car!=null && car.getAutomation()>=this.getParameters().getIntValue(ParamValue.AUTOMATIC_VEHICLE_LEVEL)){
82 // factor *= distance>this.getParameters().getIntValue(ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_THRESHOLD)?
83 // this.getParameters().getDoubleValue(ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_FAR):
84 // this.getParameters().getDoubleValue(ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_NEAR);
85 // }
86 
87 
88  return Math.max(this.getParameters().getDoubleValue(ParamValue.MIN_DIST), distance * factor);
89  }

98  public double getTravelTime(Locatable start, Locatable end, int time, SimulationType simType, TPS_ActivityConstant actCodeFrom, TPS_ActivityConstant actCodeTo, TPS_Person person, TPS_Car car) {
99  int idStart = start.getTAZId();
100  int idDest = end.getTAZId();
101  double tt;
102  double beelineDistanceLoc = TPS_Geometrics.getDistance(start, end,
103  this.getParameters().getDoubleValue(ParamValue.MIN_DIST));
104 
105  if (idStart == idDest) {
106  if (!start.getTrafficAnalysisZone().getSimulationTypeValues(simType).isIntraMITTrafficAllowed()) {
107  // no intra traffic allowed!
108  return TPS_Mode.NO_CONNECTION;
109  }
110  // start and end locations are in the same traffic analysis zone
111  if (this.getParameters().isTrue(ParamFlag.FLAG_INTRA_INFOS_MATRIX)) {
112  // If there exists travel times inside a traffic analysis zone
113  // this value is used.
114  tt = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.TRAVEL_TIME_MIT, idStart, idDest,
115  simType, time);
116  if (tt < 0) { // no connection via MIT!
117  tt = beelineDistanceLoc * this.getParameters().getDoubleValue(ModeType.WALK.getBeelineFactor()) /
118  this.getParameters().getDoubleValue(get(ModeType.WALK).getVelocity());
119  if (travelTimeIsInvalid(tt)) {
120  TPS_Logger.log(SeverenceLogLevel.FATAL, "NaN detected");
121  }
122  }
123  } else {
124  // Otherwise the travel time is calculated by the beeline
125  // distance and a factor and the average speed inside
126  // this traffic analysis zone.
127  double factor = start.getTrafficAnalysisZone().getSimulationTypeValues(simType).getBeelineFactorMIT();
128  if (Double.isNaN(factor) || factor == 0.0) { //safety fix
129  factor = 1.4;
130  }
131  double avSpeed = start.getTrafficAnalysisZone().getSimulationTypeValues(simType).getAverageSpeedMIT();
132  if (Double.isNaN(avSpeed) || avSpeed == 0.0) {
133  avSpeed = 14;//safety fix
134  }
135  tt = ((beelineDistanceLoc * factor) / avSpeed);
136  }
137  } else {
138  // start and end locations are in different traffic analysis zones.
139  // The travel time is calculated by the travel
140  // time from a table and a factor retrieved by the beeline and the
141  // real distance.
142  tt = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.TRAVEL_TIME_MIT, idStart, idDest,
143  simType, time);
144  // if TAZes are different use beeline factor
145  if (idStart != idDest) {
146  tt *= beelineDistanceLoc / this.getParameters().paramMatrixClass.getValue(ParamMatrix.DISTANCES_BL,
147  idStart, idDest);
148  }
149  }
150 
151  if (car != null && car.getAutomation() >= this.getParameters().getIntValue(
152  ParamValue.AUTOMATIC_VEHICLE_LEVEL) && SimulationType.SCENARIO.equals(simType)) {
153  //calculate time perception modification
154  double rampUp = this.getParameters().getDoubleValue(ParamValue.AUTOMATIC_VEHICLE_RAMP_UP_TIME);
155  if (tt > rampUp) {
156  double distanceNet = this.getParameters().paramMatrixClass.getValue(ParamMatrix.DISTANCES_STREET,
157  idStart, idDest);
158  double timeMod = distanceNet > this.getParameters().getIntValue(
159  ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_THRESHOLD) ? this.getParameters().getDoubleValue(
160  ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_FAR) : this.getParameters().getDoubleValue(
161  ParamValue.AUTOMATIC_VEHICLE_TIME_MOD_NEAR);
162 
163  tt = rampUp + (timeMod * (tt - rampUp));
164  }
165  }
166 
167  double acc = 0, egr = 0;
168 
169  if (this.getParameters().isDefined(ParamString.DB_NAME_MATRIX_ACCESS_MIT) && simType.equals(
170  SimulationType.SCENARIO) || this.getParameters().isDefined(
171  ParamString.DB_NAME_MATRIX_ACCESS_MIT_BASE) && simType.equals(SimulationType.BASE))
172  acc = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.ARRIVAL_MIT, idStart, idDest,
173  simType, time);
174  if (this.getParameters().isDefined(ParamString.DB_NAME_MATRIX_EGRESS_MIT) && simType.equals(
175  SimulationType.SCENARIO) || this.getParameters().isDefined(
176  ParamString.DB_NAME_MATRIX_EGRESS_MIT_BASE) && simType.equals(SimulationType.BASE))
177  egr = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.EGRESS_MIT, idStart, idDest, simType,
178  time);
179 
180  if (car != null && car.getAutomation() >= this.getParameters().getIntValue(
181  ParamValue.AUTOMATIC_VALET_PARKING)) {
182  // calculate access modification
183  acc = Math.min(acc, this.getParameters().getDoubleValue(ParamValue.AUTOMATIC_PARKING_ACCESS));
184  // calculate egress modification
185  egr = Math.min(egr, this.getParameters().getDoubleValue(ParamValue.AUTOMATIC_PARKING_EGRESS));
186  }
187  tt += acc + egr;
188 
189  return tt;
190  }
