116  public static TPS_PersonGroup getPersonGroupOfPerson(TPS_Person person) {
117  if (USE_GROUP_COLUMN_FOR_PERSON_GROUPS) {
118  return PERSON_GROUP_MAP.get(person.getGroup());
119  }
120  else {
121  for (TPS_PersonGroup tpg : PERSON_GROUP_MAP.values()) {
122  if (tpg.fits(person)) {
123  return tpg;
124  }
125  }
126  }
127  if (TPS_Logger.isLogging(TPS_LoggingInterface.HierarchyLogLevel.PERSON, TPS_LoggingInterface.SeverenceLogLevel.ERROR)) {
128  TPS_Logger.log(
129  TPS_LoggingInterface.HierarchyLogLevel.PERSON, TPS_LoggingInterface.SeverenceLogLevel.ERROR,
130  "Person does not match any person group: " + person.toString());

158  public boolean fits(TPS_Person person) {
159  return this.getMinAge()<=person.getAge() &&
160  this.getMaxAge() >= person.getAge() &&
161  this.getSex().fits(person.getSex()) &&
162  this.getCarCode().fits(person.getHousehold().getNumberOfCars()) &&
163  this.getHasChildCode().fits(person.getHousehold().getNumChildren()) &&
164  this.getPersonType().equals(TPS_PersonType.getPersonTypeByStatus(person.getStatus()));
165  }
