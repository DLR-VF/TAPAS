196  public int balanceStarts() {
197 
198  int absAdaptation = 0;
199 
200  TPS_Episode reference = null, act;
201  TPS_AdaptedEpisode actAdapted = null;
202  for (TPS_SchemePart schemePart : this.scheme) {
203  if (schemePart.isTourPart()) {
204  TPS_TourPart tourpart = (TPS_TourPart) schemePart;
205  act = tourpart.getPriorisedStayIterable().iterator().next();
206  if (reference == null || ((TPS_Stay) reference).compareTo(((TPS_Stay) act)) < 0) {
207  reference = act;
208  }
209  }
210  }
211  if (reference != null) {
212  //adopt all previous
213  act = reference;
214  actAdapted = this.getAdaptedEpisode(act);
215  do {
216  //get estimated start
217  int endOfLastEpisode = actAdapted.getStart();
218  //go one back
219  act = act.getSchemePart().getPreviousEpisode(act);
220  if (act != null) {
221  actAdapted = this.getAdaptedEpisode(act);
222  //get duration
223  int duration = actAdapted.getDuration();
224  //statistics
225  this.adapt += actAdapted.getEnd() - endOfLastEpisode;
226  absAdaptation += Math.abs(actAdapted.getEnd() - endOfLastEpisode);
227  this.adaptAbsSum += absAdaptation;
228  //set new start
229  actAdapted.setStart(endOfLastEpisode - duration);
230  }
231  } while (act != null);
232 
233  //adopt allsucceding
234  act = reference;
235  actAdapted = this.getAdaptedEpisode(act);
236  do {
237  //get estimated start
238  int startOfNextEpisode = actAdapted.getEnd();
239  //go one further
240  act = act.getSchemePart().getNextEpisode(act);
241  if (act != null) {
242  actAdapted = this.getAdaptedEpisode(act);
243  //get duration
244  double start = actAdapted.getStart();
245  //statistics
246  this.adapt += start - startOfNextEpisode;
247  absAdaptation += Math.abs(start - startOfNextEpisode);
248 
249  this.adaptAbsSum += absAdaptation;
250  //set new start
251  actAdapted.setStart(startOfNextEpisode);
252  }
253  } while (act != null);
254  }
255  int lastStart = Integer.MIN_VALUE, start;
256  TPS_PlannedTrip pt;
257  for (TPS_TourPart tp : this.scheme.getTourPartIterator()) {
258  for (TPS_Trip t : tp.getTripIterator()) {
259  pt = this.getPlannedTrip(t);
260  start = pt.getStart();
261  if (start <= lastStart) {
262  TPS_Logger.log(SeverenceLogLevel.WARN,
263  "Same or earlier start of episode detected! last:" + lastStart + " act:" + start);
264  start = lastStart + 1;
265  pt.setStart(start);
266  }
267  lastStart = start;
268  }
269  }
270 
271  return absAdaptation;
272  }

281  public void calcPlanFeasiblity() {
282  feasible = true;
283  Timeline tl = new Timeline();
284  int start = 0, end = 0;
285  Vector<TPS_Episode> sortedEpisodes = new Vector<>();
286 
287  //copy episodes from iterator to vector
288  for (TPS_Episode e : this.scheme.getEpisodeIterator()) {
289  sortedEpisodes.add(e);
290  }
291 
292  //sort episodes according original start time
293  sortedEpisodes.sort(Comparator.comparingInt(TPS_Episode::getOriginalStart));
294  //insert episodes in the timeline
295  double dist = 0;
296  for (TPS_Episode e : sortedEpisodes) {
297  TPS_AdaptedEpisode es = this.getAdaptedEpisode(e);
298  if (es.isPlannedTrip()) dist += es.getDistance();
299  //set minimum duration!
300  if (es.getDuration() < this.PM.getParameters().getIntValue(ParamValue.SEC_TIME_SLOT)) es.setDuration(
301  this.PM.getParameters().getIntValue(ParamValue.SEC_TIME_SLOT));
302  start = es.getStart();
303  end = es.getDuration() + es.getStart();
304 
305  //if(start<0||end<0){ //quick bugfix for negative start times
306  // feasible = false;
307  // break;
308  //}
309  if (!tl.add(start, end)) {
310  feasible = false;
311  break;
312  }
313  }
314 
315  boolean bookCar = feasible;
316 
317  if (bookCar && //disable car booking for COST_OPTIMUM sorting
318  this.PM.getParameters().isDefined(ParamString.HOUSEHOLD_MEMBERSORTING) &&
319  this.PM.getParameters().getString(ParamString.HOUSEHOLD_MEMBERSORTING).equalsIgnoreCase(
320  TPS_Household.Sorting.COST_OPTIMUM.name())) {
321  bookCar = false;
322  }
323 
324  //now we check if the cars are used and if they have enough range to fullfill the plan
325  if (bookCar) {
326  Map<TPS_Car, Double> cars = new HashMap<>();
327  for (TPS_SchemePart schemePart : this.scheme) { //collect car specific distances
328  if (schemePart.isTourPart()) {
329  TPS_Car car = ((TPS_TourPart) schemePart).getCar();
330  if (car != null) { //trip with car
331  dist = 0;
332  if (cars.containsKey(car)) {
333  dist = cars.get(car);
334  }
335  cars.put(car, dist + ((TPS_TourPart) schemePart).getTourpartDistance());
336  }
337  }
338  }
339  //now check every used car, if it has enough range left
340  for (Entry<TPS_Car, Double> e : cars.entrySet()) {
341  if (e.getKey().getRangeLeft() < e.getValue()) { // not enough?
342  feasible = false;
343  break;
344  }
345  }
346  }
347  }
