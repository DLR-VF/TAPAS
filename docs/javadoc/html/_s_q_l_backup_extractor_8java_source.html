146  public void processLine(String lineToTest) {
147 
148  try {

150  System.out.println("Found end of file!");

158  }
159  }
160  if (startFound) { // we are in the block
161  if (lineToTest.contains(endToken)) { // we reached the end of the block
162  if (this.includeEnd) {
163  fw.append(lineToTest + "\n");
164  lineCount++;
165  }
166  System.out.println(
167  "Found end of block: <" + lineToTest + ">\nClosing file: " + this.generateFilename() +
168  " with " + this.lineCount + " lines written.");
169  fw.close();
170  fw = null;
171  startFound = false;
172  blockNumber++;
173  } else {
174  if (!(filter != null && !lineToTest.contains(filter))) {
175  fw.append(lineToTest + "\n");
176  lineCount++;
177  }
178  }
179 
180  } else {
181  if (lineToTest.contains(startToken)) { // we found a new starting block
182  lineCount = 0;
183  startFound = true;
184  System.out.println(
185  "Found start: <" + lineToTest + ">\nWriting block to: " + this.generateFilename());
186  fw = new BufferedWriter(new FileWriter(this.generateFilename()));
187  if (this.includeStart) {
188  fw.append(lineToTest + "\n");
189  lineCount++;
190  }
191  }
192  }
