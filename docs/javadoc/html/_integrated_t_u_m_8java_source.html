153  @Override
154  protected Void doInBackground() throws Exception {
155 
156  absstarttime = System.nanoTime();
157 
158  publish(header);
159 
160  int simcount = this.simulations.length;
161 
162  for (int i = 0; i < simcount; i++) {
163 
164  this.currenttripcnt = 0;
165  starttime = System.nanoTime();
166 
167  this.firePropertyChange("intermediate", null,
168  null); //reset progress bars in {@link SimulationMonitor} after one simulation is finished
169 
170  if (isCancelled()) //avoids to keep publishing after SwingWorker was cancelled because done() gets called before doInBackground() gets cancelled
171  break;
172 
173  publish(time, MultilanguageSupport.getString("TUM_CONSOLE_PROCESS") + " " + (i + 1) + " " +
174  MultilanguageSupport.getString("TUM_CONSOLE_OF") + " " + simcount + ": '" + simulations[i] + "'");
175  publish(time, MultilanguageSupport.getString("TUM_CONSOLE_TRIPTABLE") + " ");
176 
177  tripReader = new DBTripReader(simulations[i], null, TPS_SettlementSystemType.FORDCP, null, this.connection);
178 
179  //handle trip table changes coming from {@link DBTripReader}
180  tripReader.addPropertyChangeListener(evt -> {
181  if ("triptable".equals(evt.getPropertyName()))
182 
183  if ((boolean) evt.getNewValue() && !isCancelled()) {
184  publish(success);
185  this.triptablesuccess = true;
186  this.firePropertyChange("triptable", null, true);
187  } else {
188  if (!isCancelled()) publish(fail,
189  MultilanguageSupport.getString("TUM_CONSOLE_TRIPTABLE_ERROR"));
190 
191  this.triptablesuccess = false;
192  this.firePropertyChange("triptable", null, false);
193  }
194 
195  });
196 
197  GeneralAnalyzer ga = new GeneralAnalyzer(root, false, tripReader.region);
198  ga.setDescription(tripReader.description);
199 
200 
201  try {
202  if (tripReader.getIterator().hasNext() && this.triptablesuccess) {
203 
204  publish(time, MultilanguageSupport.getString("TUM_CONSOLE_REGIONANALYSIS") + " ");
205  ga.init(true);
206 
207  this.totaltrips = tripReader.getTotal();
208 
209  while (tripReader.getIterator().hasNext() && !isCancelled()) {
210 
211  this.currenttripcnt++;
212  setProgress((int) ((float) this.currenttripcnt / this.totaltrips * 100));
213  if (!ga.prepare(simulations[i], tripReader.getIterator().next(),
214  this.connection.getParameters()))
215  break; // the very first trip that can't be evaluated will stop further evaluation on the remaining trips
216  }
217 
218  // all trips have been evaluated -> success
219  if (this.currenttripcnt == this.totaltrips) {
220  publish(success);
221  } else {
222  if (!isCancelled()) {
223  publish(fail,
224  MultilanguageSupport.getString("TUM_CONSOLE_TRIP_ERROR") + this.currenttripcnt +
225  "... " + MultilanguageSupport.getString("TUM_CONSOLE_SKIPEXCEL"), summary);
226 
227  if (i < simcount - 1) publish(nextsim);
228  }
229  tripReader.close();
230  continue; // if a problem appeared during region analysis -> continue with next simulation
231  }
232  } else {
233  if (!isCancelled()) {
234 
235  if (i < simcount - 1) publish(nextsim);
236 
237  tripReader.close();
238  continue; // if for any reason a trip table hasn't successfully been created for a given simulation -> continue with next simulation
239  }
240  }
241 
242  tripReader.close();
243 
244  if (!isCancelled()) {
245  publish(time, MultilanguageSupport.getString("TUM_CONSOLE_EXCEL") + " " +
246  exportfiles[i].getAbsolutePath() + "... ");
247 
248  if (ga.finish(exportfiles[i])) publish(success, summary);
249  else {
250  publish(fail,
251  "Please check whether the Excel File you are trying to overwrite is currently NOT in use.",
252  summary);
253  if (i < simcount - 1) publish(nextsim);
254  continue; // if excel export for an analyzed simulation fails -> continue with next simulation
255  }
256  }
257  } catch (Exception e) {
258 
259  if (!isCancelled()) {
260  publish(fail, e.toString());
261  tripReader.close();
262  publish(summary);
263  }
264 
265  continue; // any thrown exception unless this {@link IntegratedTUM} was cancelled on purpose -> continue with next simulation
266  }
267 
268  Thread.sleep(
269  1000); // due to concurrency it might happen while process() is running asynchronously on the EDT that trip counts are already reset
270  // in the doInBackground() which could show wrong information. A short sleep on the Thread solves that problem.
271  }
272 
273  return null;
274  }

311  @Override
312  protected void process(List<String> chunks) {
313 
314  for (String s : chunks) {
315  try {
316  switch (s) {
317  case success:
318  this.console.insertString(this.console.getLength(),
319  MultilanguageSupport.getString("TUM_CONSOLE_SUCCESS"), this.console.getStyle("green"));
320  break;
321  case fail:
322  this.console.insertString(this.console.getLength(),
323  MultilanguageSupport.getString("TUM_CONSOLE_FAIL"), this.console.getStyle("redbold"));
324  break;
325  case abort:
326  this.console.insertString(this.console.getLength(),
327  MultilanguageSupport.getString("TUM_CONSOLE_ABORT"), this.console.getStyle("orange"));
328  break;
329  case time:
330  this.console.insertString(this.console.getLength(), "\n" + getTime() + " --> ", null);
331  break;
332  case nextsim:
333  this.console.insertString(this.console.getLength(),
334  "\n" + MultilanguageSupport.getString("TUM_CONSOLE_NEXTSIM"), null);
335  break;
336  case summary:
337  this.console.insertString(this.console.getLength(), summary(),
338  this.console.getStyle("summary"));
339  break;
340  case totalsummary:
341  this.console.insertString(this.console.getLength(), totalSummary(),
342  this.console.getStyle("totalsummary"));
343  break;
344  case header:
345  if (this.console.getLength() > 0) this.console.insertString(this.console.getLength(),
346  "\n\n" + s + "\n", this.console.getStyle("blackunderline"));
347  else this.console.insertString(this.console.getLength(), s + "\n",
348  this.console.getStyle("blackunderline"));
349  break;
350  default:
351  if (this.console.getText(this.console.getLength() - 4, 4).equals("FAIL"))
352  this.console.insertString(this.console.getLength(), " (" + s + ")",
353  this.console.getStyle("blackitalicsmall"));
354  else this.console.insertString(this.console.getLength(), s, this.console.getStyle("standard"));
355  break;
356  }
357  } catch (BadLocationException e) {
358 
359  e.printStackTrace();
360  }
361  }
