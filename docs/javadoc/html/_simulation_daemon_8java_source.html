59  private SimulationDaemon(String[] args) {
60  TPS_ParameterClass parameterClass = new TPS_ParameterClass();
61  try {
62  this.hostname = IPInfo.getHostname();
63  File propFile = new File("daemon.properties");
64  DaemonControlProperties prop = new DaemonControlProperties(propFile);
65  String runtimeConf = prop.get(DaemonControlPropKey.LOGIN_CONFIG);
66  if (runtimeConf == null || runtimeConf.length() == 0) {
67  runtimeConf = "runtime.csv";
68  prop.set(DaemonControlPropKey.LOGIN_CONFIG, runtimeConf);
69  prop.updateFile();
70  }
71  Object[] parameters = TPS_Argument.checkArguments(args, PARAMETERS);
72 
73  File tapasNetworkDirectory = ((File) parameters[0]).getParentFile();
74 
75  File runtimeFile = new File(new File(tapasNetworkDirectory, parameterClass.SIM_DIR), runtimeConf);
76  parameterClass.loadRuntimeParameters(runtimeFile);
77  parameterClass.setString(ParamString.FILE_WORKING_DIRECTORY, tapasNetworkDirectory.getPath());
78 
79  con = new TPS_DB_Connector(parameterClass);
80  con.executeUpdate("UPDATE " + parameterClass.getString(ParamString.DB_TABLE_SERVERS) +
81  " SET remote_boot = true WHERE server_name = '" + hostname + "'", this);
82 
83  this.timer = new Timer();
84  ServerProcessChecker spc = new ServerProcessChecker();
85  timer.schedule(spc, 10, 750);
86 
87  } catch (IOException | ClassNotFoundException e) {
88  e.printStackTrace();
89  }
90  this.arguments = args;
91  Runtime.getRuntime().addShutdownHook(new ShutDownProcedure());
92  }

94  public static void main(String[] args) {
95 
96  try {
97  //Set -Ddebug=true in the VM-options to enable debugging mode
98  String debugString = System.getProperty("debug");
99 
100  if (debugString != null && debugString.equalsIgnoreCase("true")) SimulationServer.main(args);
101  else {
102  RuntimeMXBean runtime = ManagementFactory.getRuntimeMXBean();
103 
104  List<String> command = new ArrayList<>();
105  command.add("java");
106  command.addAll(runtime.getInputArguments());
107  command.add("-cp");
108  command.add(System.getProperty("java.class.path", "."));
109  command.add(SimulationServer.class.getName());
110  command.addAll(Arrays.asList(args));
111 
112  p = new ProcessBuilder(command).inheritIO().redirectErrorStream(true).start();
113  processes.add(p);
114  }
115  } catch (Exception e) {
116  try {
117  File f = new File(
118  "SimulationDaemonException_" + SDF.format(new Date(System.currentTimeMillis()) + ".log"));
119  if (!f.exists()) f.createNewFile();
120  PrintWriter pw = new PrintWriter(f);
121  pw.flush();
122  pw.close();
123  Thread.sleep(2500);
124  e.printStackTrace();
125  } catch (Exception e1) {
126  e1.printStackTrace();
127  }
128  }
129 
130  if (daemon == null) daemon = new SimulationDaemon(args);
131  }

150  private class ShutDownProcedure extends Thread {
151 
152  @Override
153  public void run() {
154 
155  try {
156  con.executeUpdate("UPDATE " + con.getParameters().getString(ParamString.DB_TABLE_SERVERS) +
157  " SET remote_boot = false WHERE server_name = '" + hostname + "'",
158  this); //shutting down the daemon means no remote server start up
159  if (p.isAlive()) {
160  System.out.println(sysoutPrefix + "Waiting for Server to shut down...");
161  con.executeUpdate("UPDATE " + con.getParameters().getString(ParamString.DB_TABLE_PROCESSES) +
162  " SET shutdown = true WHERE host = '" + hostname + "' AND end_time IS NULL",
163  this); //set the shutdown flag on the server process
164  }
165  while (p.isAlive()) Thread.sleep(1000);
166  } catch (InterruptedException e) {
167  e.printStackTrace();
168  }
169  }
