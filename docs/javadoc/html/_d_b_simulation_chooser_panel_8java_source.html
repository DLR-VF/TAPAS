265  private boolean getDatabaseContent() {
266 
267  try {
268  String q = "SELECT s.sim_key, sp.param_value as region, sp2.param_value as hhkey, sim_finished, sim_description" +
269  " FROM simulations s join simulation_parameters sp on (s.sim_key = sp.sim_key)" +
270  " join simulation_parameters sp2 on (s.sim_key = sp2.sim_key) " +
271  "WHERE (sim_finished or (not sim_started and sim_progress>0)) and sp.param_key = 'DB_REGION' " +
272  "and sp2.param_key = 'DB_HOUSEHOLD_AND_PERSON_KEY' ORDER BY sim_key";
273 
274  simdata.clear();
275  simulations.clear();
276 
277  ResultSet rs = dbCon.executeQuery(q, this);
278  while (rs.next()) {
279  String simkey = rs.getString("sim_key");
280  String description = rs.getString("sim_description");
281  String region = rs.getString("region");
282  String hhkey = rs.getString("hhkey");
283  boolean finished = rs.getBoolean("sim_finished");
284 
285  if (description == null) description = "";
286  else description = " " + description;
287  String[] arr = {simkey, region, hhkey};
288  simdata.add(arr);
289  simulations.add(simkey + description + (finished ? "" : "\t(paused)"));
290  }
291  rs.close();
292  } catch (SQLException e) {
293  System.err.println("Could not fetch meta data. Check connection to database.");
294  System.err.println(e.getMessage());
295  e.printStackTrace();
296  return false;
297  }
298 
299  try {
300  filters.clear();
301  filters.addAll(TAZFilter.getMappingNames(dbCon, this));
302  } catch (SQLException e) {
303  System.err.println("Could not fetch filters.");
304  System.err.println(e.getMessage());
305  e.printStackTrace();
306  return false;
307  }
308 
309  // update content
310  simulationDataModel = new StringListModel(simulations);
311  lstSimulations.setModel(simulationDataModel);
312 
313  DefaultComboBoxModel<String> cbFilterModel = new DefaultComboBoxModel<>(new String[]{NO_PICK});
314  for (String f : filters)
315  cbFilterModel.addElement(f);
316  cbFilter.setModel(cbFilterModel);
317  updateLabel();
318 
319  return true;
320  }

322  private List<String> getOldTables() {
323 
324  ArrayList<String> result = new ArrayList<>();
325 
326  try {
327  ArrayList<String> tables = new ArrayList<>();
328  DatabaseMetaData metaData = dbCon.getConnection(this).getMetaData();
329  ResultSet res = metaData.getTables(null, null, null, new String[]{"TABLE"});
330 
331  while (res.next()) {
332  tables.add(res.getString("TABLE_NAME"));
333  }
334 
335  for (String t : tables) {
336  if (t.startsWith("tt_")) {
337  result.add(t);
338  }
339  }
340 
341  } catch (SQLException e) {
342  System.err.println("Could not fetch meta data. Check connection to database.");
343  }
344 
345  if (result.size() > 0) return result;
346  else return null;
347  }
