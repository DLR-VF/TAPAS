59  public void checkDBMatrix(String matrixName) {

64  ResultSet rs = dbCon.executeQuery(query, this);
65  if (rs.next()) {
66  int[] iArray = TPS_DB_IO.extractIntArray(rs, "matrix_values");
67  rs.close();
68  int len = (int) Math.sqrt(iArray.length);
69  double[][] matrix = new double[len][len];
70  int index = 0;
71  for (int i = 0; i < len; ++i) {
72  for (int j = 0; j < len; ++j) {
73  matrix[i][j] = iArray[index];

75  }

77  for (int i = 0; i < len; ++i) {
78  for (int j = 0; j < len; ++j) {

80  System.out.println("Not fixable connection from " + i + " to " + j);

83  }

138  public void readCSVFile(String fileName, boolean isPT) {
139  FileReader in;
140  BufferedReader input;
141  String line;
142  try {
143  int minIndex = 0x0fffffff;
144  int maxIndex = 0;
145  int from = 0, to = 0, size;
146  double time = 0, acc = 0, egr = 0, dist = 0;
147  // read the file for min/max-values
148  in = new FileReader(fileName);
149  input = new BufferedReader(in);
150  while ((line = input.readLine()) != null) {
151  StringTokenizer tok = new StringTokenizer(line);
152  if (tok.countTokens() == 7) {
153  from = Integer.parseInt(tok.nextToken());
154  to = Integer.parseInt(tok.nextToken());
155  }
156  minIndex = Math.min(minIndex, Math.min(from, to));
157  maxIndex = Math.max(maxIndex, Math.max(from, to));
158  }
159  input.close();
160  in.close();
161 
162  size = maxIndex - minIndex + 1;
163 
164  this.matrixTT = new double[size][size];
165  this.matrixAcc = new double[size][size];
166  this.matrixEgr = new double[size][size];
167  this.matrixDist = new double[size][size];
168  //fill with no connection
169  for (int i = 0; i < size; ++i) {
170  for (int j = 0; j < size; ++j) {
171  this.matrixTT[i][j] = this.matrixAcc[i][j] = this.matrixEgr[i][j] = this.matrixDist[i][j] = TPS_Mode.NO_CONNECTION;
172  }
173  }
174  // read the file for time-values
175  in = new FileReader(fileName);
176  input = new BufferedReader(in);
177  while ((line = input.readLine()) != null) {
178  StringTokenizer tok = new StringTokenizer(line);
179  if (tok.countTokens() == 7) {
180  from = Integer.parseInt(tok.nextToken()) - minIndex;
181  to = Integer.parseInt(tok.nextToken()) - minIndex;
182  time = Double.parseDouble(tok.nextToken());
183  dist = Double.parseDouble(tok.nextToken());
184  acc = Double.parseDouble(tok.nextToken());
185  egr = Double.parseDouble(tok.nextToken());
186  }
187 
188  if (isPT) {
189  this.matrixTT[from][to] = time;
190  //this.matrixTT[from][to]= time-acc-egr; // in public transport the travel time includes access and egress times!
191  } else {
192  this.matrixTT[from][to] = time;
193  }
194  this.matrixAcc[from][to] = acc;
195  this.matrixEgr[from][to] = egr;
196  this.matrixDist[from][to] = dist;
197 
198  }
199  input.close();
200  in.close();
201 
202  //check
203  for (int i = 0; i < this.matrixTT.length; ++i) {
204  for (int j = 0; j < this.matrixTT[i].length; ++j) {
205  if (this.matrixTT[i][j] < 1 && i != j) {
206  System.out.println("No connection from " + i + " to " + j);
207  }
208  }
209  }
210 
211  System.out.println("Input processed!");
212  } catch (IOException e) {
213  e.printStackTrace();
214  }
215  }
