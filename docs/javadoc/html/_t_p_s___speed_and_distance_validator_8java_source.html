110  public double[] calcStatistics(boolean speed, double distMin, double distMax) {
111  double[] stats = new double[4];
112  double val;
113  stats[STATVAL.MIN.ordinal()] = Double.POSITIVE_INFINITY;
114  stats[STATVAL.MAX.ordinal()] = Double.NEGATIVE_INFINITY;
115  stats[STATVAL.AVG.ordinal()] = 0;
116  stats[STATVAL.STDDEV.ordinal()] = 0;
117  int count = 0;
118  for (int i = 0; i < this.data.length; ++i) {
119  for (int j = 0; j < this.data[i].length; ++j) {
120  if (i == j | this.data[i][j].yVal <= 0) continue;
121  if (this.data[i][j].xVal > distMax | this.data[i][j].xVal < distMin) continue;
122  if (speed) val = this.data[i][j].getKMH();
123  else val = this.data[i][j].xVal / this.data[i][j].beeLine;
124  stats[STATVAL.MIN.ordinal()] = Math.min(val, stats[STATVAL.MIN.ordinal()]);
125  stats[STATVAL.MAX.ordinal()] = Math.max(val, stats[STATVAL.MAX.ordinal()]);
126  stats[STATVAL.AVG.ordinal()] += val;
127  count++;
128  }
129  }
130  stats[STATVAL.AVG.ordinal()] /= count;
131  for (int i = 0; i < this.data.length; ++i) {
132  for (int j = 0; j < this.data[i].length; ++j) {
133  if (i == j | this.data[i][j].yVal <= 0) continue;
134  if (this.data[i][j].xVal > distMax | this.data[i][j].xVal < distMin) continue;
135  if (speed) val = this.data[i][j].getKMH();
136  else val = this.data[i][j].xVal / this.data[i][j].beeLine;
137  stats[STATVAL.STDDEV.ordinal()] += Math.pow((val - stats[STATVAL.AVG.ordinal()]), 2);
138  }
139  }
140  stats[STATVAL.STDDEV.ordinal()] /= count - 1;
141  stats[STATVAL.STDDEV.ordinal()] = Math.sqrt(stats[STATVAL.STDDEV.ordinal()]);
142  return stats;
143  }

240  ratio = data[i][j].xVal / data[i][j].yVal;

242  ratio = data[i][j].yVal / data[i][j].xVal;
243  }
244 
245  if (Double.isInfinite(ratio)) continue;
246 
247  if (ratio > maxThreshold) {
248  OutlierEntry from = outliers.get(i);
249  OutlierEntry to = outliers.get(j);
250  if (from == null) {
251  from = new OutlierEntry(i);
252  outliers.put(i, from);
253  }
254  if (to == null) {
255  to = new OutlierEntry(j);

257  }
258  from.addOutlier(ratio);
259  to.addOutlier(ratio);
260  }
261  }
262  }
