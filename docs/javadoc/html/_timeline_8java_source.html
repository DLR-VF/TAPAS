102  public boolean add(int start, int end) {
103  if (start > end) {
104  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {

106  "Start is greater or equal to end (start=" + start + ", end=" + end + ")");

162  private Entry[] getSurroundingEntries(int start, int end) {
163  Entry e;
164  this.entries[0] = null;
165  this.entries[1] = null;
166  for (e = entrance; e != null && (this.entries[0] == null || this.entries[1] == null); e = e.next) {
167  if (this.entries[0] == null && e.end <= start && e.next.start >= end) this.entries[0] = e;
168  if (this.entries[1] == null && e.start >= end && e.previous.end <= start) this.entries[1] = e;
169  }
170 
171  if (this.entries[0] == null || this.entries[1] == null || this.entries[0].next != this.entries[1]) {
172  return null;
173  }

186  public boolean remove(int start, int end) {
187  boolean entryRemoved = false;
188 
189  for (Entry e = entrance.next; e != this.exit && !entryRemoved; e = e.next) {
190  if (e.start == start && e.end == end) {
191  //unlink this element from the list
192  e.previous.next = e.next;
193  e.next.previous = e.previous;
194  entryRemoved = true;
195  }
196  }

198  return entryRemoved;
