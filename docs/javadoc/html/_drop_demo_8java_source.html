50  @SuppressWarnings("unchecked")
51  public void drop(DropTargetDropEvent e) {
52  Transferable tr = e.getTransferable();
53  if (tr.isDataFlavorSupported(DataFlavor.javaFileListFlavor)) {
54  e.acceptDrop(DnDConstants.ACTION_COPY_OR_MOVE);
55  try {
56  List<File> files = (List<File>) tr.getTransferData(DataFlavor.javaFileListFlavor);
57  for (File file : files) {
58  System.out.println(file.getClass());
59  }
60  } catch (UnsupportedFlavorException | IOException e1) {
61  e1.printStackTrace();
62  }
63  }
64  }
