52  public void calcWeights(PTLineType minLine) {
53  List<Household> workingListHouseholds;
54  List<PTStop> worlingListPTStops;
55  for (Entry<Integer, List<Household>> hhs : this.taz2HouseholdMap.entrySet()) {
56  workingListHouseholds = hhs.getValue();
57 
58  worlingListPTStops = this.taz2StopMap.get(hhs.getKey());
59  if (worlingListPTStops == null) { //no pt stop!
60  continue;
61  }
62  if (workingListHouseholds == null) { // no households in this taz
63  continue;
64  }
65  for (Household hh : workingListHouseholds) {
66  PTStop shortestStop = null;
67  double minDist = Double.MAX_VALUE, dist;
68  for (PTStop stop : worlingListPTStops) {
69  dist = TPS_Geometrics.getDistance(hh.lat, hh.lon, stop.lat, stop.lon);
70  if (dist < minDist && stop.hasQualityOfServiceLevel(minLine)) {
71 
72  minDist = dist;
73  shortestStop = stop;
74  }
75  }
76  if (shortestStop != null) { // fopund something?
77  // now we have the shortest stop!
78  shortestStop.weight += hh.persons;
79  shortestStop.dist += minDist * hh.persons;
80  }
81  }
82  }
