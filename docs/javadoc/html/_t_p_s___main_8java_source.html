217  public void run(int threads) {
218  long t0, t1;
219  actStats.clear();
220  try {
221  waitForMe = true;
222  // initialise persistence manager
223  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
224  TPS_Logger.log(SeverenceLogLevel.INFO, "Initialize Persistence Manager ...");
225  }
226  t0 = System.currentTimeMillis();
227  PM.init();
228  t1 = System.currentTimeMillis();
229  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
230  TPS_Logger.log(SeverenceLogLevel.INFO, "... finished in " + (t1 - t0) * 0.001 + "s");
231  }
232  if (DEBUG) {
233  threads = 1;
234  }
235  threads = Math.min(48, threads); // limit to 48 cores
236  // now update the costs variables for the modes, which are not initialized again!
237  this.updateCosts();
238 
239  // start worker threads
240  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
241  TPS_Logger.log(SeverenceLogLevel.INFO, "Initialize " + threads + " working threads ...");
242  }
243 
244 
245  ExecutorService es = Executors.newFixedThreadPool(threads);
246  List<Future<Exception>> col = new LinkedList<>();
247  for (int activeThreads = 0; activeThreads < threads; activeThreads++) {
248  col.add(es.submit(new TPS_Worker(this.PM)));
249  }
250 
251  // wait for worker threads
252  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
253  TPS_Logger.log(SeverenceLogLevel.INFO, "... wait for all working threads ...");
254  }
255 
256  for (Future<Exception> future : col) {
257  future.get();
258  }
259  es.shutdown();
260  // reset unfinished households
261  //PM.resetUnfinishedHouseholds();
262 
263  waitForMe = false;
264  } catch (Exception e) {
265  waitForMe = false;
266  TPS_Logger.log(SeverenceLogLevel.FATAL, "Application shutdown: unhandable exception", e);
267  }
268  TPS_Logger.closeLoggers();
269  }

404  public void waitFor(long timeout) {
405  if (timeout > 0) {
406  long start = System.currentTimeMillis();
407  long duration = 0;
408  synchronized (this) {
409  while ((duration = System.currentTimeMillis() - start) < timeout && !this.isFinished()) {
410  try {
411  this.wait(timeout - duration + 10);
412  } catch (InterruptedException e) {
413  // nothing to do just wait
414  }
415  }
416  }
417  } else {
418  synchronized (this) {
419  while (!this.isFinished()) {
420  try {
421  this.wait();
422  } catch (InterruptedException e) {
423  // nothing to do just wait
424  }
425  }
426  }
