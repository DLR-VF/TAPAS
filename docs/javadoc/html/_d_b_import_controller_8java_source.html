199  public void dbimport(final JTable fileTable) {
200  final ExecutorService es = Executors.newFixedThreadPool(1);
201 
202  Callable<Boolean> callable = new Callable<Boolean>() {
203  public Boolean call() {
204 
205  boolean success = false;
206  control.print("Creation of Tables and Loading Table Contents");
207  control.setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
208 
209  try {
210  TPS_DB_Connector db = new TPS_DB_Connector("daniel", "daniel", control.getParameters());
211 
212 
213  control.getParameters().setString(ParamString.DB_SCHEMA_CORE, control.getSchema() + ".");
214  db.execute(TPS_DB_IOManager
215  .buildQuery(control.getParameters(), "create_region_based_tables", control.getRegion(),
216  control.getSchema()), this);
217 
218  db.execute("SET CONSTRAINTS ALL DEFERRED", this);
219 
220  DBImportMethod method;
221  List<ParamString> dbKeyList;
222  String fileName;
223  String tableName;
224  String name;
225 
226  for (int i = 0; i < DB_IMPORT_METHOD.size(); i++) {
227  method = DB_IMPORT_METHOD.get(i);
228  dbKeyList = DB_HIERARCHY.get(i);
229  for (ParamString dbKey : dbKeyList) {
230  for (int j = 0; j < fileTable.getRowCount(); j++) {
231  if (!fileTable.getValueAt(j, COL_T_KEY).equals(dbKey.name()) || method == null ||
232  Boolean.FALSE.equals(fileTable.getValueAt(j, COL_ACC))) {
233  continue;
234  }
235 
236  File tempFile = new File(control.getParameters().getString(ParamString.PATH_ABS_DB),
237  control.getParameters().INPUT_DIR);
238  fileName = new File(tempFile, fileTable.getValueAt(j, COL_F_NAME).toString()).getPath()
239  .replace(
240  '\\',
241  '/');
242  tableName = fileTable.getValueAt(j, COL_T_NAME).toString();
243  name = fileTable.getValueAt(j, COL_ID).toString();
244 
245  control.print("IMPORT " + fileName + " INTO " + tableName + " WITH NAME " + name);
246 
247  switch (method) {
248  case CSV:
249  db.execute(TPS_DB_IOManager
250  .buildQuery(control.getParameters(), "load_csvfile_to_table", fileName,
251  tableName), this);
252  break;
253  case EPS:
254  db.execute(TPS_DB_IOManager
255  .buildQuery(control.getParameters(), "load_episodes", fileName), this);
256  break;
257  case VOT:
258  db.execute(TPS_DB_IOManager
259  .buildQuery(control.getParameters(), "load_values_of_time", fileName,
260  name), this);
261  break;
262  case SCD:
263  db.execute(TPS_DB_IOManager
264  .buildQuery(control.getParameters(), "load_scheme_class_distributions",
265  fileName, name), this);
266  break;
267  case SCT:
268  db.execute(TPS_DB_IOManager
269  .buildQuery(control.getParameters(), "load_scheme_class_time_usage",
270  fileName), this);

272  // case TAZ:

274  // break;
275  case TIM:

329  }

331  }
