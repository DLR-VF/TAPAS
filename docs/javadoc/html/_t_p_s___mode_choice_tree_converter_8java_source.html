75  public void load() {
76  FileReader in = null;
77  BufferedReader input = null;
78  String line = "";
79  try {
80  in = new FileReader(inputFile);
81  input = new BufferedReader(in);
82  System.out.println("\t '--> File opened: " + this.inputFile.getAbsolutePath());
83  int id = -1, parent = -1, count;
84  String attVal = "", splitValue = "", dist = "", token;
85 
86  //read headers
87  input.readLine();
88 
89  while ((line = input.readLine()) != null) {
90  if (line.startsWith("$")) // comment
91  continue;
92  StringTokenizer tok = new StringTokenizer(line, ",");
93  //check format
94  if (tok.countTokens() != 7) continue;
95  count = 0;
96  while (tok.hasMoreTokens()) {
97  token = tok.nextToken();
98  token = token.replaceAll("\"", "");
99  if (token.equals("")) //empty string to NULL String
100  token = "\\N";
101 
102  switch (count) {
103  case 0: //id
104  id = Integer.parseInt(token);
105  break;
106  case 1: //level
107  break;
108  case 2: //size
109  break;
110  case 3: //idParent
111  parent = Integer.parseInt(token);
112  break;
113  case 4: //attval
114  attVal = token.replaceAll(" ", ",");
115  while (attVal.endsWith(",")) attVal = attVal.substring(0, attVal.length() - 1);
116  attVal = "{" + attVal + "}";
117  break;
118  case 5: //splitval
119  splitValue = map.get(token);
120  if (splitValue == null || splitValue.equals("")) splitValue = "\\N";
121  break;
122  case 6: //distribution
123  dist = token.replaceAll(" ", ",");
124  while (dist.endsWith(",")) dist = dist.substring(0, dist.length() - 1);
125  dist = "{" + dist + "}";
126  break;
127  }
128  count++;
129  }
130  this.treeEntries.add(
131  identifier + "\t" + id + "\t" + parent + "\t" + attVal + "\t" + splitValue + "\t" + dist);
132  }
133  System.out.println("\t '--> Found: " + this.treeEntries.size() + " nodes.");
134  } catch (Exception ex) {
135  System.out.println("\t '--> Error: " + ex.getMessage());
136  ex.printStackTrace();
137  }//catch
138  finally {
139  try {
140  if (input != null) input.close();
141  if (in != null) in.close();
142  System.out.println("\t '--> Input file closed: " + this.inputFile.getAbsolutePath());
143  }//try
