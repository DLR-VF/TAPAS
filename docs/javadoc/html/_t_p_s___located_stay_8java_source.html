184  public void selectLocation(TPS_Plan plan, TPS_PlanningContext pc) {
185  if (TPS_Logger.isLogging(SeverenceLogLevel.DEBUG)) {
186  TPS_Logger.log(SeverenceLogLevel.DEBUG,
187  "Start select procedure for stay (id=" + this.getStay().getId() + ")");
188  }
189 
190  TPS_ActivityConstant currentActCode = this.stay.getActCode();
191  plan.setAttributeValue(TPS_Attribute.CURRENT_EPISODE_ACTIVITY_CODE_MCT,
192  currentActCode.getCode(TPS_ActivityCodeType.MCT));
193  plan.setAttributeValue(TPS_Attribute.CURRENT_EPISODE_ACTIVITY_CODE_VOT,
194  currentActCode.getCode(TPS_ActivityCodeType.VOT));
195  plan.setAttributeValue(TPS_Attribute.CURRENT_EPISODE_ACTIVITY_CODE_TAPAS,
196  currentActCode.getCode(TPS_ActivityCodeType.TAPAS));
197 
198  // can never happen:
199  if (this.isLocated()) {
200  throw new RuntimeException(
201  "The location should not be set because in a previous call, because we only call this method once for every TPS_LocatedStay");
202  } else if (this.stay.isAtHome()) {
203  throw new RuntimeException(
204  "The initialisation of the home parts should be done in the constructor of TPS_Plan");
205  }
206 
207  // Has to be a tour part because all home parts are at home; home parts will never reach this method
208  TPS_TourPart tourpart = (TPS_TourPart) this.stay.getSchemePart();
209  TPS_Stay comingFrom = tourpart.getStayHierarchy(this.stay).getPrevStay();
210  TPS_Stay goingTo = tourpart.getStayHierarchy(this.stay).getNextStay();
211  if (!plan.isLocated(comingFrom) || !plan.isLocated(goingTo)) {
212  // this case should be impossible because we now iterate over the priorised stays,
213  // so every stay where we can come from or where we go to is located
214  throw new IllegalStateException("Found no location for a higher priorised stay");
215  }
216 
217  if (currentActCode.hasAttribute(TPS_ActivityConstantAttribute.E_COMMERCE_OUT_OF_HOME)) {
218  // Is this an activity that (if it not takes place at home anyway) should be executed in the very
219  // vicinity of the home residence?
220  if (TPS_Logger.isLogging(SeverenceLogLevel.DEBUG)) {
221  TPS_Logger.log(SeverenceLogLevel.DEBUG,
222  "Activity: " + currentActCode + " assumed to be performed at home");
223  }
224  this.setLocation(plan.getPerson().getHousehold().getLocation());
225  } else {
226  TPS_Region region = PM.getRegion();
227  this.setLocation(region.selectLocation(plan, pc, this));
228  if (this.getLocation() == null) {
229  TPS_Logger.log(SeverenceLogLevel.ERROR,
230  "End select procedure for stay (id=" + this.getStay().getId() + ") with no location");
231  //throw new RuntimeException("End select procedure for stay (id=" + this.getStay().getId() + ") with no location");
232  } else {
233  if (TPS_Logger.isLogging(SeverenceLogLevel.DEBUG)) {
234  TPS_Logger.log(SeverenceLogLevel.DEBUG,
235  "End select procedure for stay (id=" + this.getStay().getId() + ") with location (id=" +
236  this.getLocation().getId() + ")");
237  }
238  }
239  }
240 
241  // Get distance from home The MIV-mode is used to get distances on the net.
242  this.setDistance(TPS_Mode.get(ModeType.MIT)
243  .getDistance(plan.getLocatedStay(comingFrom).getLocation(), this.getLocation(),
244  SimulationType.SCENARIO, null));
245  }
