265  private int fetchNextSetOfHouseholds(TPS_Region region) {
266  String query = "";
267  try { //check if simulation is running
268  query = "SELECT sim_started FROM " + this.PM.getParameters().getString(ParamString.DB_TABLE_SIMULATIONS) +
269  " WHERE sim_key = '" + this.PM.getParameters().getString(ParamString.RUN_IDENTIFIER) + "'";
270  ResultSet sRs = PM.executeQuery(query);
271  if (!sRs.next()) {
272  TPS_Logger.log(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO, "simulation key removed from db!");
273  return 0;
274  }
275  if (sRs.getBoolean("sim_started")) {
276  if (TPS_Logger.isLogging(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO)) {
277  TPS_Logger.log(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO, "Fetching new set of households");
278  }
279  String hhtemptable = this.PM.getParameters().getString(ParamString.DB_TABLE_HOUSEHOLD_TMP);
280 
281 
282  int actualCount = numberToFetch;
283  List<Integer> hIDs = new ArrayList<>();
284  // first fetch number unfinished hh for this server
285  query = "SELECT count(*) as unfinished FROM " + hhtemptable +
286  " WHERE hh_started = true AND hh_finished = false AND server_ip = inet '" +
287  ADDRESS.getHostAddress() + "'";
288  ResultSet hRs = PM.executeQuery(query);
289  if (hRs.next()) {
290  actualCount -= hRs.getInt("unfinished");
291  }
292  hRs.close();
293 
294  //mark new set of households: This is done in a single statement.
295  //Deadlocks are avoided by sorting by prio, which is a random number initialized during
296  // simulation-setup. So its fix for the whole sim and guarantees, that no "ring lock" will happen!
297  if (actualCount > 0) {
298  Connection con = this.PM.getDbConnector().getConnection(this);
299 
300  //time critical lock
301  con.setAutoCommit(false);
302  Statement st = con.createStatement();
303  query = "LOCK TABLE " + hhtemptable + " IN EXCLUSIVE MODE";
304  st.execute(query);
305  query = "UPDATE " + hhtemptable + " SET hh_started = true, server_ip = inet '" +
306  ADDRESS.getHostAddress() + "' WHERE hh_started = false AND hh_id = ANY(SELECT hh_id FROM " +
307  hhtemptable + " WHERE hh_started = false ORDER BY prio, hh_id LIMIT " + actualCount + ")";
308  st.execute(query);
309  con.commit();
310  con.setAutoCommit(true);
311  if (TPS_Logger.isLogging(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO)) {
312  TPS_Logger.log(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO,
313  "Assigned " + actualCount + " new households this instance.");
314  }
315  }
316 
317  //get all hhId which should be processed in the next round
318  query = "SELECT hh_id FROM " + hhtemptable +
319  " WHERE hh_started = true AND hh_finished = false AND server_ip = inet '" +
320  ADDRESS.getHostAddress() + "' ORDER BY prio, hh_id";
321  hRs = PM.executeQuery(query);
322  int id;
323  while (hRs.next()) {
324  id = hRs.getInt("hh_id");
325  hIDs.add(id);
326  }
327  hRs.close();
328 
329  if (TPS_Logger.isLogging(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO)) {
330  TPS_Logger.log(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO,
331  "Fetching data for " + hIDs.size() + " households.");
332  }
333 
334  List<TPS_Household> hhs;
335  if (hIDs.size() == 0) {
336  this.houseHoldMap.clear();
337  } else {
338  hhs = this.loadHouseholds(region, hIDs);
339  for (TPS_Household hh : hhs) {
340  this.prefetchedHouseholds.offer(hh);
341  }
342  }
343  }
344  this.numberOfFetchedHouseholds = this.prefetchedHouseholds.size();
345  if (numberOfFetchedHouseholds > 0) {
346  if (TPS_Logger.isLogging(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO)) {
347  TPS_Logger.log(HierarchyLogLevel.THREAD, SeverenceLogLevel.INFO,
348  "Fetched " + numberOfFetchedHouseholds + " households");
349  }
350  }
351  } catch (SQLException e) {
352  TPS_Logger.log(HierarchyLogLevel.THREAD, SeverenceLogLevel.ERROR, "error during one of th sqls: " + query,
353  e);
354  TPS_Logger.log(HierarchyLogLevel.THREAD, SeverenceLogLevel.ERROR, "next exception:", e.getNextException());
355  }
356  return this.prefetchedHouseholds.size();
357  }
