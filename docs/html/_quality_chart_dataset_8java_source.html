154  private double setConfidenceInterval(ArrayList<QualityChartData> data) {
155  double pwa = 0;
156  for (QualityChartData d : data) {
157  if (d.getReference() > 0) {
158  pwa += (d.getModel() - d.getReference()) / d.getReference();
159  }
160  }
161 
162  // pwa = 0.01; // TODO confidence interval stub
163  pwa = 1.96 * pwa / data.size();
164 
165  seriesCollection.get(Key.LOWER_CONFIDENCE).clear();
166  seriesCollection.get(Key.UPPER_CONFIDENCE).clear();
167 
168  seriesCollection.get(Key.UPPER_CONFIDENCE).add(new QualityChartData(pwa, 0, ""));
169  seriesCollection.get(Key.UPPER_CONFIDENCE).add(new QualityChartData(max, max - pwa, ""));
170 
171  seriesCollection.get(Key.LOWER_CONFIDENCE).add(new QualityChartData(0, pwa, ""));
172  seriesCollection.get(Key.LOWER_CONFIDENCE).add(new QualityChartData(max - pwa, max, ""));
173  return pwa;
174  }

176  private void setGEH5(ArrayList<QualityChartData> data) {
177  for (QualityChartData d : data) {
178  double s = d.getReference();
179  double g5p = 0.25 * (4 * s + 5 * Math.sqrt(16 * s + 25) + 25);
180  double g5m = 0.25 * (4 * s - 5 * Math.sqrt(16 * s + 25) + 25);
181 
182  seriesCollection.get(Key.LOWER_GEH5).add(new QualityChartData(g5m, s, null));
183  seriesCollection.get(Key.UPPER_GEH5).add(new QualityChartData(g5p, s, null));
184  }
