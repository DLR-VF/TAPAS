231  public boolean loadMapping(String name) {
232  String query = "";
233  try {
234  query = "SELECT map_value, taz_values FROM core." + this.region + this.dbName + " WHERE name='" + name +
235  "'";
236  ResultSet rs = dbCon.executeQuery(query, this);
237  while (rs.next()) {
238  int map_id = rs.getInt("map_value");
239  int[] tazes = TPS_DB_IO.extractIntArray(rs, "taz_values");
240  boolean tazOK = tazes.length > 0;
241  int i;
242  for (i = 0; i < tazes.length && tazOK; ++i) {
243  tazOK &= this.taz.contains(tazes[i]);
244  }
245  if (tazOK) {
246  // sort values for later convenience, see getTAZes
247  Arrays.sort(tazes);
248  this.mapping.put(map_id, tazes);
249  } else {
250  if (tazes.length == 0) {
251  System.err.println(
252  this.getClass().getCanonicalName() + "loadMapping: Empty map for map_id: " + map_id);
253  } else {
254  System.err.println(
255  this.getClass().getCanonicalName() + "loadMapping: Unknown TAZ: " + tazes[i - 1]);
256  }
257  return false;
258  }
259  }
260  rs.close();
261  } catch (SQLException e) {
262  System.err.println(
263  this.getClass().getCanonicalName() + "loadMapping: SQL-Error during statement: " + query);
264  e.printStackTrace();
265  return false;
266  }
267  return taz.size() > 0;
268  }
