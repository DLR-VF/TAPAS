204  private void expandAll(JTree tree, TreePath parent, boolean expand) {
205  // Traverse children
206  TreeNode node = (TreeNode) parent.getLastPathComponent();
207  if (node.getChildCount() >= 0) {
208  for (Enumeration e = node.children(); e.hasMoreElements(); ) {
209  TreeNode n = (TreeNode) e.nextElement();
210  TreePath path = parent.pathByAddingChild(n);
211  expandAll(tree, path, expand);
212  }
213  }
214  // Expansion or collapse must be done bottom-up
215  if (expand) {
216  tree.expandPath(parent);
217  } else {
218  tree.collapsePath(parent);
219  }
220  }

581  private void updateNode(boolean add) {
582  if (tree == null || baseAnalyzerList == null) {
583  return;
584  }
585  TreePath path = tree.getSelectionPath();
586 
587  DefaultMutableTreeNode node;
588  if (path == null && !add) {
589  return;
590  } else if (path != null) {
591  node = (DefaultMutableTreeNode) path.getLastPathComponent();
592  } else {
593  node = root;
594  }
595 
596  String name = edtName.getText();
597  DefaultListModel<AnalyzerBase> analyzers = (DefaultListModel<AnalyzerBase>) activeAnalyzerList.getModel();
598 
599  AnalyzerCollection col;
600  if (analyzers.size() == 0) {
601  col = new AnalyzerCollection(name);
602  } else {
603  AnalyzerBase[] a = new AnalyzerBase[analyzers.size()];
604  for (int i = 0; i < analyzers.size(); i++) {
605  a[i] = analyzers.get(i);
606  }
607  col = new AnalyzerCollection(name, a);
608  }
609  if (add) {
610  treeModel.insertNodeInto(new DefaultMutableTreeNode(col), node, 0);
611  } else {
612  node.setUserObject(col);
613  treeModel.nodeChanged(node);
614  }

674  private static final long serialVersionUID = -5580351519596596981L;
675 
676  @Override
677  public boolean canImport(TransferSupport support) {
678  return support.isDataFlavorSupported(AnalyzerTransferable.ANALYZER_DATA_FLAVOR);
679  }
680 
681  @Override
682  public boolean importData(TransferSupport support) {
683  if (canImport(support)) {
684  Transferable transferable = support.getTransferable();
685  try {
686  AnalyzerBase analyzer = (AnalyzerBase) transferable.getTransferData(
687  AnalyzerTransferable.ANALYZER_DATA_FLAVOR);
688 
689  JTree.DropLocation dl = (JTree.DropLocation) support.getDropLocation();
690 
691  TreePath path = dl.getPath();
692  int childIndex = dl.getChildIndex();
693 
694  if (path == null) {
695  return false;
696  }
697 
698  DefaultMutableTreeNode node;
699  DefaultTreeModel model = (DefaultTreeModel) ((JTree) support.getComponent()).getModel();
700 
701  node = (DefaultMutableTreeNode) path.getPathComponent(path.getPathCount() - 1);
702 
703  if (childIndex < 0) {
704  AnalyzerCollection col = (AnalyzerCollection) node.getUserObject();
705  col.addAnalyzer(analyzer);
706  model.nodeChanged(node);
707  } else {
708  model.insertNodeInto(new DefaultMutableTreeNode(new AnalyzerCollection(analyzer)), node,
709  childIndex);
710  }
711  } catch (UnsupportedFlavorException | IOException e) {
712  e.printStackTrace();
713  return false;
714  }
715  return true;
716  }
717  return false;
718  }
