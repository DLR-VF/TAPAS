102  private Matrix readMatrix(String matrixName, String tableName) {
103  Matrix m = null;
104  String query = "SELECT matrix_values FROM " + tableName + " WHERE matrix_name='" + matrixName + "'";
105  try {
106  ResultSet rs = this.dbCon.executeQuery(query, this);
107  if (rs.next()) {
108  int[] iArray = TPS_DB_IO.extractIntArray(rs, "matrix_values");
109  int len = (int) Math.sqrt(iArray.length);
110  m = new Matrix(len, len);
111  for (int index = 0; index < iArray.length; index++) {
112  m.setRawValue(index, iArray[index]);
113  }
114  }
115  rs.close();
116  } catch (SQLException e) {
117  System.err.println("Error during sql-statement: " + query);
118  e.printStackTrace();
119  e.getNextException().printStackTrace();
120  }
121  return m;
122  }

159  public void run() {

168  double minDuration = -1;
169  for (int e1 = 0; e1 < parent.size; ++e1) {
170  if (parent.interchanges[e1][j] < 0) {
171  continue;
172  }
173  if (parent.carTT[i][e1] < 0) {
174  continue;
175  }
176  if (parent.ptTT[e1][j] < 0) {
177  continue;

180  duration += parent.carAccess[i][e1] + parent.carTT[i][e1] + parent.carEgress[i][e1] * 1.;
181  duration += parent.ptAccess[i][e1] * 1. + parent.ptTT[e1][j] + parent.ptEgress[i][e1];
182  if (minDuration < 0 || duration < minDuration) {
