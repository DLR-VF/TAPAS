254  private final SortedMap<Double, LAND_USE> finalFactor;
255  private final EnumMap<LAND_USE, ArrayList<Long>> addresses;
256  private final Random randomGenerator;

261  public Distributor(EnumMap<LAND_USE, ArrayList<Long>> addresses) {
262 
263  LAND_USE.RESIDENTIAL.getFactor();
264 
265  this.addresses = addresses;
266  double total = 0;
267  for (LAND_USE lu : LAND_USE.values()) {
268 
269  total += addresses.get(lu).size() * lu.getFactor();
270  }
271 
272  // create sorted map of factors
273  TreeMap<Double, LAND_USE> factor = new TreeMap<>();
274  for (LAND_USE lu : LAND_USE.values()) {
275  if (addresses.get(lu).size() > 0) {
276  factor.put(addresses.get(lu).size() * lu.getFactor() / total, lu);
277  } else {
278  factor.put(0.0, lu);
279  }
280  }
281 
282  // make cumulative map
283  finalFactor = new TreeMap<>();
284 
285  for (Entry<Double, LAND_USE> e : factor.entrySet()) {
286  double t = 0;
287  for (Double dt : factor.headMap(e.getKey()).keySet()) {
288  t += dt;
289  }
290  finalFactor.put(t, e.getValue());
291  }
292 
293  randomGenerator = new Random();
294  }

300  public long getRandomBuilding() {
301  double r = Math.random();
302  LAND_USE lu = finalFactor.get(finalFactor.headMap(r).lastKey());
303  // System.out.println("r = " + r + " lu = " + lu);
304  ArrayList<Long> ll = addresses.get(lu);
305  if (ll.size() > 0) {
306  return ll.get(randomGenerator.nextInt(ll.size()));
307  } else {
308  return -1;
309  }
