33  private Matrix readMatrix( String matrixName, String tableName) {
34  Matrix m = null;
35  String query= "SELECT matrix_values FROM " + tableName + " WHERE matrix_name='" + matrixName + "'";
36  try{
37  ResultSet rs = this.dbCon.executeQuery(query,this);
38  if (rs.next()) {
39  int[] iArray = TPS_DB_IO.extractIntArray(rs, "matrix_values");
40  int len = (int) Math.sqrt(iArray.length);
41  m = new Matrix(len, len);
42  for (int index = 0; index < iArray.length; index++) {
43  m.setRawValue(index, iArray[index]);
44  }
45  }
46  rs.close();
47  } catch(SQLException e){
48  System.err.println("Error during sql-statement: "+query);
49  e.printStackTrace();
50  e.getNextException().printStackTrace();
51  }
52  return m;
53  }
