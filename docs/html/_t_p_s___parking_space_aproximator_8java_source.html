101  public void calcParkingTimes(double minSearch, double maxSearch, double searchSpeedCar, double blockDistance, double walkSpeed, double saturationFactor) {
102  this.access = new double[this.numberOfTaz];
103  this.egress = new double[this.numberOfTaz];
104  double[] egressSort = new double[this.numberOfTaz];
105 
106  double highestDensity = 0, lowestDensity = 1e99, density;
107  int carsPerTaz, areaPerTaz;
108  //first calc the density and find the normalization factors
109  for (Integer taz : this.carsPerTaz.keySet()) {
110  carsPerTaz = this.carsPerTaz.get(taz);
111  areaPerTaz = this.areaPerTaz.get(taz);
112  density = ((double) carsPerTaz) * 1000000.0 / (double) areaPerTaz;
113  this.egress[taz - minTAZ] = density;
114  egressSort[taz - minTAZ] = density;
115  highestDensity = Math.max(highestDensity, density);
116  lowestDensity = Math.min(lowestDensity, density);
117  }
118 
119  saturationFactor = Math.max(0., Math.min(1.0, saturationFactor));
120 
121  Arrays.sort(egressSort);
122 
123  lowestDensity = egressSort[0];
124  int indexSaturation = (int) ((egressSort.length - 1) * saturationFactor);
125  //chop to reasonabele values
126  indexSaturation = Math.max(1, Math.min(egressSort.length - 1, indexSaturation));
127  highestDensity = egressSort[indexSaturation];
128 
129  //now normalize the density to the min/max searchradius
130  for (int i = 0; i < egress.length; ++i) {
131  //normalize from 0 to 1
132  this.egress[i] = (egress[i] - lowestDensity) / (highestDensity - lowestDensity);
133  //extract from min to max
134  this.egress[i] = egress[i] * (maxSearch - minSearch) + minSearch;
135  //in egress is now the search distance to find a parking lot
136  //now calculate the walking time back and put it in access
137  int round = 1;
138  double distanceCovered = 4 * blockDistance;
139  while (distanceCovered < egress[i]) {
140  round++;
141  distanceCovered += 8 * blockDistance + distanceCovered;
142  }
143  //ok we have now the number of rounds to circle around the destination
144  //now calculate the time to get there:
145  this.access[i] = round * blockDistance / walkSpeed;
146  this.maxAccess = Math.max(this.maxAccess, this.access[i]);
147  this.minAccess = Math.min(this.minAccess, this.access[i]);
148 
149  //ok now we calculate the egress time:
150  //search distance/search speed + walking time back
151  this.egress[i] = this.egress[i] / searchSpeedCar + this.access[i];
152  this.maxEgress = Math.max(this.maxEgress, this.egress[i]);
153  this.minEgress = Math.min(this.minEgress, this.egress[i]);
154  }
155  }

202  public void storeSQLMatrix(String db, String nameEgress, String nameAccess) {
203  //delete old values
204  String query = "DELETE FROM " + db + " WHERE matrix_name= '" + nameEgress + "'";
205  this.dbCon.execute(query, this);
206  query = "DELETE FROM " + db + " WHERE matrix_name= '" + nameAccess + "'";
207  this.dbCon.execute(query, this);
208 
209  //create the access/egress matrix:
210  //Egress: all values TO this cell have the same egress
211  //Access: all values FROM this cell have the same access
212  Matrix egressM = new Matrix(this.egress.length);
213  Matrix accessM = new Matrix(this.access.length);
214  for (int i = 0; i < this.egress.length; ++i) {
215  for (int j = 0; j < this.egress.length; ++j) {
216  egressM.setValue(i, j, this.egress[i]); //same destination (i)
217  accessM.setValue(i, j, this.access[j]); //same origin (j)
218  }
219  }
220 
221  //save to db!
222  query = "INSERT INTO " + db + " (\"matrix_name\",\"matrix_values\")" + " VALUES('" + nameEgress + "',";
223  // build array of matrix-ints
224  query += TPS_DB_IO.matrixToSQLArray(egressM, 0);
225  query += ")";
226  this.dbCon.execute(query, this);
227 
228  query = "INSERT INTO " + db + " (\"matrix_name\",\"matrix_values\")" + " VALUES('" + nameAccess + "',";
229  // build array of matrix-ints
230  query += TPS_DB_IO.matrixToSQLArray(accessM, 0);
231  query += ")";
232  this.dbCon.execute(query, this);
233  }
