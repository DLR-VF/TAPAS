290  private int getTimeSliceIndex(int startTimeMin) {
291  if (this.distribution == null) return -1;
292 
293  double time = startTimeMin / 60.0;
294 
295  //correct time
296  while (time < 0.0) time += 24.0;
297  while (time > 24.0) time -= 24.0;
298 
299  for (int i = 0; i < this.distribution.length; ++i) {
300  if (this.distribution[i] >= time) return i;
301  }
302  return -1;

318  public void mergeMatrixPair(Matrix oldVals, int index) {
319  int i, j;
320  double oldVal, newVal;
321  double error, maxError = 0, minError = 1e100, avgError = 0;
322  final int size = this.matrixArray[index].getNumberOfColums();
323  final double normalizationFactor = 1.0 / (1.0 + this.weight);
324  for (i = 0; i < size; ++i) {
325  for (j = 0; j < size; ++j) {
326  oldVal = oldVals.getValue(i, j);
327  newVal = this.matrixArray[index].getValue(i, j);
328  error = oldVal - newVal;
329  if (newVal > 0.0) {//ignore invalid cells (no trafic)
330  maxError = Math.max(maxError, error);
331  minError = Math.min(minError, error);
332  avgError += error;
333  newVal = (oldVal + this.weight * newVal) * normalizationFactor;
334  this.matrixArray[index].setValue(i, j, newVal);
335  }
336  }
337  }
338  avgError /= size * size;
339  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
340  TPS_Logger.log(SeverenceLogLevel.INFO,
341  "max error: " + maxError + " minError: " + minError + " avg error: " + avgError);
342  }
343  }

409  public void readCSVFile(String fileName) throws IOException {
410  FileReader in = null;
411  BufferedReader input = null;
412  String line = "";
413  try {
414  int fromTVZ, toTVZ, tvzCounter = 0, startTime, mode, distributionIndex;
415  double time, actDistance;
416  boolean hasDistance;
417  tvzCounter = this.idToIndex.size();
418  //prepare arrays
419  this.distance = new double[tvzCounter][tvzCounter];
420 
421  //open input
422  in = new FileReader(fileName);
423  input = new BufferedReader(in);
424  line = input.readLine();//header
425  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
426  TPS_Logger.log(SeverenceLogLevel.INFO, "File opened: " + fileName + " header: " + line);
427  }
428 
429  String actToken;
430  while ((line = input.readLine()) != null) {
431  if (line.startsWith("$")) // comment
432  continue;
433  StringTokenizer tok = new StringTokenizer(line, ",");
434  //check format
435  //check format
436  if (tok.countTokens() == 10) hasDistance = true;
437  else if (tok.countTokens() == 9) hasDistance = false;
438  else continue;
439  actToken = tok.nextToken(); //p_id
440  actToken = tok.nextToken(); //hh_id
441  actToken = tok.nextToken(); //start time
442  startTime = Integer.parseInt(actToken.trim());
443  distributionIndex = this.getTimeSliceIndex(startTime);
444  actToken = tok.nextToken(); //start taz
445  fromTVZ = Integer.parseInt(actToken.trim()) - 1;
446  actToken = tok.nextToken(); //end taz
447  toTVZ = Integer.parseInt(actToken.trim()) - 1;
448  actToken = tok.nextToken(); //start block
449  actToken = tok.nextToken(); //end block
450  actToken = tok.nextToken(); //mode
451  mode = Integer.parseInt(actToken.trim());
452  if (distributionIndex >= 0 && mode == 2) {
453  actToken = tok.nextToken(); //travel time
454  time = Double.parseDouble(actToken.trim());
455  //inc time
456  this.matrixArray[distributionIndex].setValue(fromTVZ, toTVZ,
457  this.matrixArray[distributionIndex].getValue(fromTVZ, toTVZ) + time);
458  //inc counter for this cell
459  this.counter[distributionIndex][fromTVZ][toTVZ]++;
460  this.counter[counter.length - 1][fromTVZ][toTVZ]++;
461  if (hasDistance) {
462  actToken = tok.nextToken(); //distance
463  actDistance = Double.parseDouble(actToken.trim());
464  //inc distance
465  this.distance[fromTVZ][toTVZ] += actDistance;
466  }
467  }
468  }
469  //calc average
470 
471  for (int i = 0; i < this.distribution.length; ++i) {
472  for (int j = 0; j < tvzCounter; ++j) {
473  for (int k = 0; k < tvzCounter; ++k) {
474  //only valid distances
475  if (i == 0 && this.counter[counter.length - 1][j][k] > 0) {
476  this.distance[j][k] /= this.counter[counter.length - 1][j][k];
477  }
478  int counterVal = this.counter[i][j][k];
479  //only valid traveltimes
480  if (counterVal > 0) {
481  double val = this.matrixArray[i].getValue(j, k);
482  this.matrixArray[i].setValue(j, k, val / counterVal);
483  }
484  }
485  }
486  }
487  } finally {
488  try {
489  if (input != null) input.close();
490  if (in != null) in.close();
491  }//try
492  catch (IOException ex) {
493  TPS_Logger.log(SeverenceLogLevel.ERROR, " Could not close : " + fileName);
494  throw new IOException(ex);
495  }//catch
496  }//finally
497  }

537  String query = "SELECT * FROM " + this.dbManager.getParameters().getString(ParamString.DB_TABLE_MATRICES) +
538  " WHERE \"matrix_name\" = '" + matrixName + "'";
539  try {
540  ResultSet rs = this.dbManager.executeQuery(query, this);
541  if (rs.next()) {
542  Object array = rs.getArray("matrix_values").getArray();
543  if (array instanceof Integer[]) {
544  //parse data to memory model
545  Integer[] matrixVal = (Integer[]) array;
546  int size = (int) Math.sqrt(matrixVal.length);
547  if (size != this.distance.length) {
548  TPS_Logger.log(SeverenceLogLevel.ERROR,
549  "Matrix " + matrixName + " has a different size: " + size + " Expected: " +
550  this.distance.length);
551  return null;
552  }
553  returnVal = new double[size][size];
554  int k = 0;
555  for (int i = 0; i < size; ++i) {
556  for (int j = 0; j < size; ++j, ++k) {
557  returnVal[i][j] = (double) matrixVal[k];
558  }
559  }
560  }
561  } else {
562  TPS_Logger.log(SeverenceLogLevel.ERROR, "Matrix " + matrixName + " does not exist!");
563  }
564  rs.close();
565  } catch (SQLException e) {
566  TPS_Logger.log(SeverenceLogLevel.ERROR, "SQL Error: " + query, e);
567  }
