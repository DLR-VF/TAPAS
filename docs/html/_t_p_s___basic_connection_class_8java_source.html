141  }
142 
143  if (props.get(ClientControlPropKey.LOGIN_CONFIG).length() > 0 && programPath.length() > 0 && !forceNewFile) {

147 
148  } else {

152  String baseDirectory;
153  File tmpFile;
154  // did they click on open
155  if (val == JFileChooser.APPROVE_OPTION) {
156  tmpFile = choose.getSelectedFile();
157  loginFile = tmpFile.getAbsolutePath();
158  props.set(ClientControlPropKey.LOGIN_CONFIG, loginFile);
159  if (System.getProperty("os.name").toLowerCase().contains("windows")) {
160  baseDirectory = loginFile.substring(0, loginFile.indexOf(":" + File.separator) + 2);
161  props.set(ClientControlPropKey.TAPAS_DIR_WIN, baseDirectory);
162  } else {
163  baseDirectory = loginFile.substring(0, loginFile.indexOf(File.separator, 1));
164  props.set(ClientControlPropKey.TAPAS_DIR_LINUX, tmpFile.getParent());
165  }
166  props.set(ClientControlPropKey.LOGIN_CONFIG, tmpFile.getAbsolutePath());
167  props.updateFile();
168  runtimeFile = new File(loginFile);
169  } else {
170  runtimeFile = null;
171  }
172  }
173  return runtimeFile;
174  }

212  public static String matrixToSQLArray(double[][] array, int decimalPlaces) {
213  StringBuilder buffer;
214  StringBuilder totalBuffer = new StringBuilder("'{");
215  for (int j = 0; j < array.length; ++j) {
216  buffer = new StringBuilder();
217  for (int k = 0; k < array[0].length; ++k) {
218  buffer.append(new BigDecimal(array[j][k]).setScale(decimalPlaces, RoundingMode.HALF_UP)).append(",");
219  }
220  if (j < array.length - 1) totalBuffer.append(buffer);
221  else totalBuffer.append(buffer.substring(0, buffer.length() - 1)).append("}'");
222  }
223  return totalBuffer.toString();
224  }

330  public void storeInDB(String matrixName, double[][] matrix, int decimalPlaces) {
331  if (decimalPlaces != 0) {
332  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {
333  TPS_Logger.log(SeverenceLogLevel.WARN,
334  "Decimal places are currently incompatible with the db (integers not doubles). Setting Decimal Places to 0!");
335  }
336  decimalPlaces = 0;
337  }
338  //load the data from the db
339  String query = "SELECT * FROM " + this.parameterClass.getString(ParamString.DB_TABLE_MATRICES) +
340  " WHERE \"matrix_name\" = '" + matrixName + "'";
341 
342  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
343  TPS_Logger.log(SeverenceLogLevel.INFO, "Preparing data for entry: " + matrixName + " in table " +
344  this.parameterClass.getString(ParamString.DB_TABLE_MATRICES));
345  }
346  if (checkMatrixName(matrixName)) {
347  //update!
348  query = "UPDATE " + this.parameterClass.getString(ParamString.DB_TABLE_MATRICES) + " SET matrix_values = ";
349  query += matrixToSQLArray(matrix, decimalPlaces) + " WHERE \"matrix_name\" = '" + matrixName + "'";
350  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
351  TPS_Logger.log(SeverenceLogLevel.INFO, "Updating data for entry: " + matrixName + " in table " +
352  this.parameterClass.getString(ParamString.DB_TABLE_MATRICES) + ".");
353  }
354  } else {
355  query = "INSERT INTO " + this.parameterClass.getString(ParamString.DB_TABLE_MATRICES) +
356  " (matrix_name, matrix_values) VALUES ('" + matrixName + "', ";
357  query += matrixToSQLArray(matrix, decimalPlaces) + ")";
358  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
359  TPS_Logger.log(SeverenceLogLevel.INFO, "Inserting data for entry: " + matrixName + " in table " +
360  this.parameterClass.getString(ParamString.DB_TABLE_MATRICES) + ".");
361  }
362  }
363  dbCon.execute(query, this);
364  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
365  TPS_Logger.log(SeverenceLogLevel.INFO, "Successful!");
366  }
367  }
