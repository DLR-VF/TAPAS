181  private static class QualityChartLegendGerator implements XYSeriesLabelGenerator {
182 
183  @Override
184  public String generateLabel(XYDataset dataset, int series) {
185  if (!(dataset instanceof QualityChartDataset)) {
186  throw new IllegalArgumentException("This dataset is not supported.");
187  }
188 
189  QualityChartDataset qcd = (QualityChartDataset) dataset;
190 
191  Key key = (Key) qcd.getSeriesKey(series);
192 
193  if (QualityChartDataset.DATASERIES.contains(key)) {
194  return Key.getQuality(key).toString();
195  } else if (key == Key.LOWER_CONFIDENCE) {
196  return "Confidence Interval";
197  } else if (key == Key.LOWER_GEH5) {
198  return "GEH5";
199  }
200  return "";
201  }
