101  private int binarySearch(double randomValue) {
102  // in case of randomValue==0.0
103  // return the first element with a non-zero probability
104  if (randomValue == 0.0) {
105  for (int i = 0; i < this.size(); i++) {
106  if (values[i] > 0.0) return i;
107  }
108  }
109  int low = 0;
110  int high = this.size() - 1;
111  int index = (low + high) / 2;
112  while (low < high) {
113  if (cumulativeValues[index] == randomValue) { //found the exact value
114  // roll back to the lowest index s.t. cumulativeValues[index] == randomValue
115  while (index > 0) {
116  if (cumulativeValues[index - 1] == randomValue) index--;
117  else break;
118  }
119  return index;
120  } else if (cumulativeValues[index] < randomValue) low = index + 1;
121  else if (cumulativeValues[index] > randomValue) {
122  if (index > 0 && cumulativeValues[index - 1] < randomValue) return index;
123  high = index;
124  }
125  index = (low + high) / 2;
126  }
127  if (low > high) {
128  if (cumulativeValues[index] < randomValue) TPS_Logger.log(SeverenceLogLevel.ERROR, "Something is wrong");
129  }
130  return index;
131  }

149  public int draw(double randomValue) {
150  if (this.size() == 0) {
151  TPS_Logger.log(SeverenceLogLevel.ERROR, "Empty distribution!");
152  return 0;
153  }
154  //safety min/max if randomValue is not between 0 and 1
155  randomValue = Math.max(0, Math.min(1, randomValue));
156  if (!isNormalized()) {
157  this.normalize(false);
158  }
159  if (!isNormalized()) {
160  TPS_Logger.log(SeverenceLogLevel.ERROR, "Distribution cannot be normalized because all" +
161  " values are 0 or there is an infinity or NaN value; sum of values: " + this.sum() +
162  " is Normalized: " + this.isNormalized);
163  }
164  return this.binarySearch(randomValue);
165  }

358  public boolean normalize(boolean normalizeValues) {
359  double sum = this.sum();
360  if (sum > 0.0) {
361  for (int i = 0; i < this.size(); i++) {
362  this.cumulativeValues[i] = (i > 0 ? this.cumulativeValues[i - 1] : 0) + this.values[i] / sum;
363  if (normalizeValues) this.values[i] = this.values[i] / sum; //normalize if true
364  }
365  this.isNormalized = true;
366  return true;
367  } else {
368  if (TPS_Logger.isLogging(SeverenceLogLevel.ERROR)) {
369  TPS_Logger.log(SeverenceLogLevel.ERROR, "Trying to normalize a distribution with a sum of zero!");
370  }
371  }
372  return false;
373  }
