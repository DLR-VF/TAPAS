64  public void readRepresentatives(String addressTable, String tazTable, int samples) {
65  String query = "";
66  try {
67  int centroidRepresentatives = 0;
68  int processed = 0;
69  for (Integer i : tazIds) {
70  query = "WITH adr as (SELECT gid, st_transform(geom,4326) as geom FROM " + addressTable + ") " +
71  "SELECT gid, st_X(geom) as x, st_Y(geom) as y FROM adr WHERE st_within(geom, (SELECT the_geom from " +
72  tazTable + " WHERE gid = " + i + ")) order by random() LIMIT " + samples;
73  ResultSet rs = this.dbCon.executeQuery(query, this);
74  int representativesFound = 0;
75  while (rs.next()) {
76  representativesFound++;
77  Representative elem = new Representative();
78  elem.id = rs.getInt("gid");
79  elem.taz = i;
80  elem.x = rs.getDouble("x");
81  elem.y = rs.getDouble("y");
82  representatives.add(elem);
83  }
84  rs.close();
85  if (representativesFound == 0) { // no address in this TAZ
86  centroidRepresentatives++; //one more centroid
87  query = "SELECT st_X(st_transform(st_centroid(the_geom),4326)) as x, st_Y(st_transform(st_centroid(the_geom),4326)) as y FROM " +
88  tazTable + " WHERE gid= " + i;
89  rs = this.dbCon.executeQuery(query, this);
90  while (rs.next()) {
91  Representative elem = new Representative();
92  elem.id = -centroidRepresentatives; //negative ID for centroids
93  elem.taz = i;
94  elem.x = rs.getDouble("x");
95  elem.y = rs.getDouble("y");
96  representatives.add(elem);
97  }
98  rs.close();
99  }
100  processed++;
101  if (processed % 10 == 0) System.out.println("Processed TAZ: " + processed + "/" + tazIds.size());
102  }
103  System.out.println("Processed TAZ: " + processed + "/" + tazIds.size());
104  System.out.println("Added " + centroidRepresentatives + " centroids, because TAZ had no address");
105  } catch (SQLException e) {
106  System.out.println("SQL error! Query: " + query);
107  e.printStackTrace();
108  }
109  }
