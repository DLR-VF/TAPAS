59  public static boolean applyExpertKnowledge(TPS_ModeSet modeSet, TPS_Plan plan, double distanceNet, TPS_ModeChoiceContext mcc, boolean checkForbiddenModes, TPS_DiscreteDistribution<TPS_Mode> dist) {
60 
61  if (checkForbiddenModes) {
62  //set modes with no service for this connection to zero
63  boolean changed = false;
64  int numEmpty = 0;
65  for (TPS_Mode.ModeType m : TPS_Mode.MODE_TYPE_ARRAY) {
66  TPS_Mode mode = TPS_Mode.get(m);
67  double tt = mode.getTravelTime(mcc.fromStayLocation, mcc.toStayLocation, mcc.startTime,
68  SimulationType.SCENARIO, TPS_ActivityConstant.DUMMY, TPS_ActivityConstant.DUMMY,
69  plan.getPerson(), mcc.carForThisPlan);
70  if (TPS_Mode.noConnection(tt) || (!mcc.isBikeAvailable && mode.isType(TPS_Mode.ModeType.BIKE)) ||
71  (mcc.carForThisPlan == null && mode.isType(TPS_Mode.ModeType.MIT)) || (mode.equals(
72  TPS_Mode.ModeType.TAXI) && mode.getParameters().isFalse(ParamFlag.FLAG_USE_TAXI)) //disable TAXI
73  ) { // FIXME TPS_Mode comparison with ModeType is always false, right?
74  dist.setValueByKey(mode, TPS_UtilityFunction.minModeProbability);
75  changed = true;
76  }
77  if (dist.getValueByKey(mode) == 0) {
78  numEmpty++;
79  }
80  }
81  if (changed || numEmpty == TPS_Mode.MODE_TYPE_ARRAY.length) {
82  if (numEmpty == TPS_Mode.MODE_TYPE_ARRAY.length || !dist.normalize()) {
83  if (TPS_Logger.isLogging(HierarchyLogLevel.EPISODE,
84  TPS_LoggingInterface.SeverenceLogLevel.SEVERE)) {
85  TPS_Logger.log(HierarchyLogLevel.EPISODE, TPS_LoggingInterface.SeverenceLogLevel.SEVERE,
86  "\"Check forbidden modes\" erased all possible modes!");
87  }
88  return false;
89  }
90  }
91  }
92  //now apply the expert
93 
94 
95  TPS_DiscreteDistribution<TPS_Mode> mod = modeSet.getExpertKnowledgeTree().getDistributionSet(plan)
96  .modifyDistribution(dist);
97  for (int i = 0; i < mod.size(); ++i) {
98  dist.setValueByPosition(i, mod.getValueByPosition(i));
99  }
100  dist.normalize();
101 
102  //reducing WALK for long distances!
103  double walkShare;
104  if (distanceNet > modeSet.getParameters().getIntValue(ParamValue.MAX_WALK_DIST)) {
105  walkShare = 0;
106  dist.setValueByKey(TPS_Mode.get(TPS_Mode.ModeType.WALK), walkShare);
107  return dist.normalize();
108  }
109  return true;
110  }
