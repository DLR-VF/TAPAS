76  public static String buildQuery(TPS_ParameterClass parameterClass, String function, Object... parameters) {
77  StringBuilder sb = new StringBuilder(
78  "SELECT " + parameterClass.getString(ParamString.DB_SCHEMA_CORE) + function + "(");
79  if (parameters != null) {
80  for (Object parameter : parameters) {
81  if (parameter instanceof ParamString) {
82  ParamString ps = (ParamString) parameter;
83  sb.append("'").append(parameterClass.getString(ps)).append("',");
84  } else if (parameter instanceof String) {
85  String string = (String) parameter;
86  sb.append("'").append(string).append("',");
87  } else if (parameter instanceof InetAddress) {
88  InetAddress address = (InetAddress) parameter;
89  sb.append("inet '").append(address.getHostAddress()).append("',");
90  } else {
91  sb.append(parameter.toString()).append(",");
92  }
93  }
94  }
95  sb.setCharAt(sb.length() - 1, ')');
96  return sb.toString();
97  }

99  private void batchStoreTrips() {
100  String statement = "";
101  PreparedStatement pS;
102  TPS_Plan plan;
103  TPS_Person p;
104  TPS_Household hh;
105  int index, numberOfPlans = 0;
106  int[] batchResultCounter;
107  TPS_Stay prevStay, nextStay;
108  TPS_Location prevLoc, nextLoc;
109  TPS_LocatedStay nextStayLocated;
110  TPS_PlannedTrip pt;
111  double duration;
112  boolean planStoreSuccessful = true;
113  final int chunkSize = 128;
114 
115  try {
116  Connection con = this.dbConnector.getConnection(this);
117 
118  statement = "INSERT INTO " + this.getParameters().getString(ParamString.DB_TABLE_TRIPS) +
119  " VALUES (?,?,?,?,?, ?,?,?,?,?, ?,?,?,?,?, ?,?,?,?,?, ?,?,?,?,?, ?,?,?,?,? ,?,?,?,?, ?)";
120  pS = con.prepareStatement(statement);
121  while (!this.plansToStore.isEmpty()) {
122  plan = this.plansToStore.poll();
123  plan.balanceStarts(); // because some location references could be reseted from other plans, balance starts again!
124 
125 
126  p = plan.getPerson();
127  hh = p.getHousehold();
128  int lastStart = -10000000, start;
129 
130  for (TPS_TourPart tp : plan.getScheme().getTourPartIterator()) {
131  for (TPS_Trip t : tp.getTripIterator()) {
132  index = 1;
133  pS.setInt(index++, p.getId());
134  pS.setInt(index++, hh.getId());
135  pS.setInt(index++, plan.getScheme().getId());
136  if (Double.isNaN(plan.getAcceptanceProbability()) || Double.isInfinite(
137  plan.getAcceptanceProbability())) {
138  TPS_Logger.log(SeverenceLogLevel.FATAL,
139  "NaN detected in getAcceptanceProbability for person " + p.getId());
140  }
141  pS.setDouble(index++, plan.getAcceptanceProbability());
142  if (Double.isNaN(plan.getBudgetAcceptanceProbability()) || Double.isInfinite(
143  plan.getBudgetAcceptanceProbability())) {
144  TPS_Logger.log(SeverenceLogLevel.FATAL,
145  "NaN detected in getBudgetAcceptanceProbability for person " + p.getId());
146  }
147  pS.setDouble(index++, plan.getBudgetAcceptanceProbability());
148  if (Double.isNaN(plan.getTimeAcceptanceProbability()) || Double.isInfinite(
149  plan.getTimeAcceptanceProbability())) {
150  TPS_Logger.log(SeverenceLogLevel.FATAL,
151  "NaN detected in getTimeAcceptanceProbability for person " + p.getId());
152  }
153  pS.setDouble(index++, plan.getTimeAcceptanceProbability());
154 
155  prevStay = tp.getPreviousStay(t);
156  nextStay = tp.getNextStay(t);
157  prevLoc = plan.getLocatedStay(prevStay).getLocation();
158  nextStayLocated = plan.getLocatedStay(nextStay);
159  nextLoc = nextStayLocated.getLocation();
160  pS.setInt(index++, prevLoc.getTrafficAnalysisZone().getTAZId());
161  pS.setBoolean(index++,
162  prevLoc.getTrafficAnalysisZone().hasToll(this.getParameters().getSimulationType()));
163  pS.setInt(index++, (prevLoc.hasBlock() ? prevLoc.getBlock().getId() : -1));
164  pS.setInt(index++, prevLoc.getId());
165  pS.setDouble(index++, prevLoc.getCoordinate().getValue(0));
166  pS.setDouble(index++, prevLoc.getCoordinate().getValue(1));
167 
168  pS.setInt(index++, nextLoc.getTrafficAnalysisZone().getTAZId());
169  pS.setBoolean(index++,
170  nextLoc.getTrafficAnalysisZone().hasToll(this.getParameters().getSimulationType()));
171  pS.setInt(index++, (nextLoc.hasBlock() ? nextLoc.getBlock().getId() : -1));
172  pS.setInt(index++, nextLoc.getId());
173  pS.setDouble(index++, nextLoc.getCoordinate().getValue(0));
174  pS.setDouble(index++, nextLoc.getCoordinate().getValue(1));
175 
176  pt = plan.getPlannedTrip(t);
177  start = pt.getStart();
178  if (start - lastStart < 1.0) {
179  start = lastStart + 1;
180  }
181  lastStart = start;
182 
183  pS.setInt(index++, (int) ((start * 1.66666666e-2) + 0.5)); //sec to min incl round
184  if (Double.isNaN(pt.getDuration()) || Double.isInfinite(pt.getDuration())) {
185  TPS_Logger.log(SeverenceLogLevel.FATAL,
186  "NaN detected in getDuration for person " + p.getId());
187  }
188  pS.setDouble(index++, pt.getDuration());
189  pS.setInt(index++, pt.getMode().getMCTCode());
190  pS.setInt(index++, tp.getCar() == null ? -1 : tp.getCar().getId());
191  if (Double.isNaN(pt.getDistanceBeeline()) || Double.isInfinite(pt.getDistanceBeeline())) {
192  TPS_Logger.log(SeverenceLogLevel.FATAL,
193  "NaN detected in getDistanceBeeline for person " + p.getId());
194  }
195  pS.setDouble(index++, pt.getDistanceBeeline());
196  if (Double.isNaN(pt.getDistance()) || Double.isInfinite(pt.getDistance())) {
197  TPS_Logger.log(SeverenceLogLevel.FATAL,
198  "NaN detected in getDistance for person " + p.getId());
199  }
200  pS.setDouble(index++, pt.getDistance());
201  pS.setInt(index++, nextStay.getActCode().getCode(TPS_ActivityCodeType.ZBE));
202  pS.setBoolean(index++, nextStay.isAtHome());
203  pS.setInt(index++,
204  (int) ((nextStayLocated.getStart() * 1.66666666e-2) + 0.5)); // secs to min incl round
205  duration = nextStayLocated.getDuration();
206  //sum durations of concurring stays
207  while (tp.getNextEpisode(nextStay).isStay()) {
208  nextStay = (TPS_Stay) tp.getNextEpisode(nextStay);
209  duration += plan.getLocatedStay(nextStay).getDuration();
210  }
211  pS.setInt(index++, (int) ((duration * 1.66666666e-2) + 0.5)); // secs to min incl round
212  if (tp.getCar() != null) {
213  pS.setInt(index++, tp.getCar().index);
214  pS.setBoolean(index++, tp.getCar().isRestricted());
215  } else {
216  pS.setInt(index++, -1);
217  pS.setBoolean(index++, false);
218  }
219  pS.setInt(index++, p.getPersGroup().getCode());
220  pS.setInt(index++,
221  prevLoc.getTrafficAnalysisZone().getBbrType().getCode(TPS_SettlementSystemType.FORDCP));
222  pS.setInt(index++, hh.getLocation().getTrafficAnalysisZone().getBbrType()
223  .getCode(TPS_SettlementSystemType.FORDCP));
224  pS.setInt(index++, nextStay.locationChoiceMotive.code);// the location_selection_motive
225  pS.setInt(index++, nextStay.locationChoiceMotiveSupply.code);// the location_selection_motive
226  pS.addBatch();
227  numberOfPlans++;
228  if (numberOfPlans > 0 && numberOfPlans % chunkSize == 0) {
229  batchResultCounter = pS.executeBatch();
230  for (index = 0; index < batchResultCounter.length && planStoreSuccessful; ++index) {
231  planStoreSuccessful &= batchResultCounter[index] != PreparedStatement.EXECUTE_FAILED;
232  }
233  }
234  }
235  }
236  plan.reset();
237  }
238 
239  batchResultCounter = pS.executeBatch();
240 
241  for (index = 0; index < batchResultCounter.length && planStoreSuccessful; ++index) {
242  planStoreSuccessful &= batchResultCounter[index] != PreparedStatement.EXECUTE_FAILED;
243  }
244 
245  if (!planStoreSuccessful) {
246  TPS_Logger.log(SeverenceLogLevel.ERROR, "Storing of plans failed!");
247  }
248  pS.close();
249  } catch (SQLException e) {
250  TPS_Logger.log(SeverenceLogLevel.ERROR, "Error during sql-statement: " + statement);
251  TPS_Logger.log(SeverenceLogLevel.ERROR, e.getMessage(), e);
252  TPS_Logger.log(SeverenceLogLevel.ERROR, "Next exception:");
253  TPS_Logger.log(SeverenceLogLevel.ERROR, e.getNextException().getMessage(), e.getNextException());
254  }
255  }

552  private void storeFinishedHouseholdIDsToDB() {
553  ArrayList<Integer> tmpIds = new ArrayList<>();
554  String statement = "";
555  PreparedStatement pS;
556  int numberOfPlans = 0, numberOfHouseholds = 0;
557  int[] batchResultCounter;
558  int index, chunkSize = 256;
559  boolean planStoreSuccessful = true, householdStoreSuccesful = true;
560 
561  try {
562  Connection con = this.dbConnector.getConnection(this);
563  con.setAutoCommit(false);
564  //write households
565  if (this.insertedHouseHolds.size() > 0) {
566  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
567  TPS_Logger.log(SeverenceLogLevel.INFO,
568  "Finishing " + this.insertedHouseHolds.size() + " households.");
569  }
570 
571  //sort to avoid dead locks!
572  while (this.insertedHouseHolds.size() > 0) {
573  tmpIds.add(this.insertedHouseHolds.poll());
574  }
575  Collections.sort(tmpIds);
576 
577  statement = "UPDATE " + this.getParameters().getString(ParamString.DB_TABLE_HOUSEHOLD_TMP) +
578  " SET hh_started = true, hh_finished = true WHERE hh_id = ?";
579  pS = con.prepareStatement(statement);
580  for (Integer i : tmpIds) {
581  pS.setInt(1, i);
582  pS.addBatch();
583  numberOfHouseholds++;
584  if (numberOfHouseholds > 0 && numberOfHouseholds % chunkSize == 0) {
