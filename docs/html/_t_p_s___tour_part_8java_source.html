504  public void updateActualTravelDistances(TPS_Plan plan) {
505  for (TPS_Stay stay : this.getPriorisedStayIterable()) {
506  TPS_PlannedTrip arrivalTrip = null;
507  TPS_PlannedTrip departureTrip = null;
508  int minTime = (int) (plan.getPM().getParameters().getDoubleValue(ParamValue.MIN_DIST) /
509  plan.getPM().getParameters().getDoubleValue(TPS_Mode.get(ModeType.WALK).getVelocity()));
510  //get arrival trip
511  if (!this.isFirst(stay)) {
512  arrivalTrip = plan.getPlannedTrip(this.getPreviousTrip(stay));
513  }
514  //get departure trip
515  if (!this.isLast(stay)) {
516  departureTrip = plan.getPlannedTrip(this.getNextTrip(stay));
517  }
518  if (arrivalTrip.getMode() != null) {
519  this.travelDurationsMap.get(stay).arrivalDuration = Math.max(arrivalTrip.getDuration(), minTime);
520  }
521  if (departureTrip.getMode() != null) {
522  this.travelDurationsMap.get(stay).departureDuration = Math.max(departureTrip.getDuration(), minTime);
523  }
524  }
