101  public StringBuffer createStringBuffer(ResultSet rs) throws SQLException {
102  StringBuffer sb = new StringBuffer("ARRAY[");

131  public int deleteHouseholdsFromTriptable(Connection connection) {
132  int isolationLevel;
133  boolean autoCommit;
134  String query = "";
135 
136  try {
137  synchronized (connection) {
138  autoCommit = connection.getAutoCommit();
139  isolationLevel = connection.getTransactionIsolation();
140  Statement statement = connection.createStatement();
141  // AutoCommit to FALSE
142  connection.setAutoCommit(false);
143  connection.setTransactionIsolation(Connection.TRANSACTION_REPEATABLE_READ);
144 
145  // get all ids for all locations
146  StringBuilder sb_hhId = new StringBuilder("ARRAY[");
147  if (hh_id.size() == 0) {
148  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {
149  TPS_Logger.log(SeverenceLogLevel.WARN, "No households given. Program stops.");
150  }
151  resetConnection(connection, autoCommit, isolationLevel, statement);
152  return 0;
153  }
154  for (Integer hh_id : this.hh_id) {
155  sb_hhId.append(hh_id + ",");
156  }
157  sb_hhId.setCharAt(sb_hhId.length() - 1, ']');
158 
159  // get all Person_IDs from HH-id
160  //query = "SELECT p_id from " + ParamString.DB_TABLE_PERSON.getString() + " WHERE p_hh_id = ANY("
161  // + sb_hhId.toString() + ") AND p_key = '" + ParamString.DB_HOUSEHOLD_AND_PERSON_KEY.getString()+ "'";
162  query = TPS_DB_IOManager.buildQuery(this.parameterClass, "getpersonidsfromhh_trip",
163  this.parameterClass.getString(ParamString.DB_TABLE_PERSON), sb_hhId.toString(),
164  this.parameterClass.getString(ParamString.DB_HOUSEHOLD_AND_PERSON_KEY));
165  ResultSet rs_persons = statement.executeQuery(query);
166 
167  // put all person_ids in an array
168  StringBuffer sb_personIds = createStringBuffer(rs_persons);
169  rs_persons.close();
170  if (sb_personIds == null) {
171  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {
172  TPS_Logger.log(SeverenceLogLevel.WARN, "No matching person IDs found. Program stops.");
173  }
174  resetConnection(connection, autoCommit, isolationLevel, statement);
175  return 0;
176  }
177  // check if all households in a trip have been processed
178  for (Integer iterator_elem : hh_id) {
179  query = "Select * FROM " + this.parameterClass.getString(ParamString.DB_TABLE_HOUSEHOLD_TMP) +
180  " WHERE hh_id =" + iterator_elem;
181  ResultSet rs_hhTest = statement.executeQuery(query);
182  if (!rs_hhTest.next()) {
183  hh_id.remove(iterator_elem);
184  }
185  rs_hhTest.close();
186  if (hh_id.isEmpty()) {
187  resetConnection(connection, autoCommit, isolationLevel, statement);
188  return 0;
189  }
190  }
191 
192  // Get location_ids from person_ids
193  //query = "SELECT loc_id_end FROM " + ParamString.DB_TABLE_TRIPS.getString() + " WHERE p_id = ANY("
194  // + sb_personIds.toString() + ")";
195  query = TPS_DB_IOManager.buildQuery(this.parameterClass, "getlocidsfromperson",
196  this.parameterClass.getString(ParamString.DB_TABLE_TRIPS), sb_personIds.toString());
197  ResultSet rs_locIds = statement.executeQuery(query);
198 
199  Map<String, Integer> sortedIds = new HashMap<>();
200  int sum = 0;
201  // Loc_IDs sortieren:
202  if (rs_locIds.next()) {
203  if (Integer.parseInt(rs_locIds.getString(1)) > 0) {
204  sortedIds.put(rs_locIds.getString(1), 1);
205  } else {
206  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
207  TPS_Logger.log(SeverenceLogLevel.INFO, "1. Weg nach Hause, Programm fortsetzen");
208  }
209  }
210  while (rs_locIds.next()) {
211  if (!sortedIds.containsKey(rs_locIds.getString(1)) && Integer.parseInt(rs_locIds.getString(1)) >
212  0) {
213  sortedIds.put(rs_locIds.getString(1), 1);
214  } else if (sortedIds.containsKey(rs_locIds.getString(1)) && Integer.parseInt(
215  rs_locIds.getString(1)) > 0) {
216  int oldValue = sortedIds.get(rs_locIds.getString(1));
217  sortedIds.remove(rs_locIds.getString(1));
218  sortedIds.put(rs_locIds.getString(1), oldValue + 1);
219  }
220  sum++;
221  }
222  } else {
223  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
224  TPS_Logger.log(SeverenceLogLevel.INFO, "No locations found; continuing.");
225  }
226  }
227  rs_locIds.close();
228 
229  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
230  TPS_Logger.log(SeverenceLogLevel.INFO,
231  "Found " + sortedIds.size() + " locations with " + sum + " occupancies to remove");
232  }
233  while (!sortedIds.isEmpty()) {
234  Map<String, Integer> sortedIdsUse = new HashMap<>();
235  Iterator<String> it = sortedIds.keySet().iterator();
236  StringBuilder sb_LocIds = new StringBuilder("ARRAY[");
237 
238  if (!it.hasNext()) {
239  break;
240  }
241  while (it.hasNext()) {
242  Object nextKey = it.next();
243  sb_LocIds.append(nextKey.toString() + ",");
244  int val = sortedIds.get(nextKey.toString()) - 1;
245  if (val > 0) {
246  sortedIdsUse.put(nextKey.toString(), val);
247  } else {
248  it.remove();
249  }
250  }
251  sb_LocIds.setCharAt(sb_LocIds.length() - 1, ']');
252 
253  // Decrement locations
254  // System.out.println("sbLocID: " + sb_LocIds.toString());
255 
256  query = TPS_DB_IOManager.buildQuery(this.parameterClass, "incr_occupancy",
257  this.parameterClass.getString(ParamString.DB_TABLE_LOCATION_TMP), sb_LocIds.toString(), -1,
258  !TPS_DB_IOManager.BEHAVIOUR.equals(Behaviour.FAT));
259  statement.execute(query);
260  sortedIds = sortedIdsUse;
261  }
262 
263  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
264  TPS_Logger.log(SeverenceLogLevel.INFO, "deleting hhs");
265  }
266 
267  // delete hh_id from triptable
268  query = TPS_DB_IOManager.buildQuery(this.parameterClass, "deletehhfromtrip",
269  this.parameterClass.getString(ParamString.DB_TABLE_TRIPS), sb_hhId.toString());
270  statement.execute(query);
271 
272  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
273  TPS_Logger.log(SeverenceLogLevel.INFO, "resetting hhs");
274  }
275 
276  // set households on started = false/finished = false & server_ip =
277  // null
278  query = TPS_DB_IOManager.buildQuery(this.parameterClass, "sethhfalse",
279  this.parameterClass.getString(ParamString.DB_TABLE_HOUSEHOLD_TMP), sb_hhId.toString());
280  statement.execute(query);
281 
282  // decrement sim_progress and set flags "started" & "finished" =
283  // FALSE
284  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
285  TPS_Logger.log(SeverenceLogLevel.INFO, "decreasing progress bar");
286  }
287  query = TPS_DB_IOManager.buildQuery(this.parameterClass, "decrProgress", key, hh_id.size());
288  statement.execute(query);
289  connection.commit();
290  connection.setAutoCommit(autoCommit);
291  connection.setTransactionIsolation(isolationLevel);
292  //vacuum trips tmp household and locations
293  query = "VACUUM " + this.parameterClass.getString(ParamString.DB_TABLE_HOUSEHOLD_TMP);
294  statement.execute(query);
295  query = "VACUUM " + this.parameterClass.getString(ParamString.DB_TABLE_LOCATION_TMP);
296  statement.execute(query);
297  query = "VACUUM " + this.parameterClass.getString(ParamString.DB_TABLE_TRIPS);
298  statement.execute(query);
299 
300 
301  statement.close();
302  // connection.close();
303  }
304  } catch (SQLException e) {
305  TPS_Logger.log(SeverenceLogLevel.ERROR, "Exception during SQL! Query: " + query, e);
306  }
307  return hh_id.size();
308  }
