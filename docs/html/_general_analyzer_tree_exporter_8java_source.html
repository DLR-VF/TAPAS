154  @SuppressWarnings("rawtypes")
155  public static DefaultMutableTreeNode readTree(String filename) throws IOException {
156 
157  File xmlFile = new File(filename);
158  DocumentBuilderFactory docFactory = DocumentBuilderFactory.newInstance();
159 
160  DocumentBuilder docBuilder;
161  Document doc;
162  try {
163  docBuilder = docFactory.newDocumentBuilder();
164  doc = docBuilder.parse(xmlFile);
165  } catch (ParserConfigurationException | SAXException e1) {
166  e1.printStackTrace();
167  return null;
168  }
169 
170  doc.getDocumentElement().normalize();
171 
172  if (!doc.getDocumentElement().getAttribute(ATTRIBUTE_VERSION).equals(VERSION)) {
173  System.err.println("Version does not match! Expected version is " + VERSION);
174  return null;
175  }
176 
177  Node domRoot = doc.getDocumentElement();
178  DocumentTraversal traversal = (DocumentTraversal) doc;
179  TreeWalker walker = traversal.createTreeWalker(domRoot, NodeFilter.SHOW_ELEMENT, null, false);
180 
181  DefaultMutableTreeNode treeRoot = new DefaultMutableTreeNode();
182 
183  Node currentDOMNode;
184  DefaultMutableTreeNode parentTreeNode = treeRoot;
185  currentDOMNode = walker.nextNode();
186  HashSet<AnalyzerBase> analyzerSet = new HashSet<>();
187  String currentName = "";
188 
189  while (currentDOMNode != null) {
190  if (currentDOMNode.getNodeType() != Node.ELEMENT_NODE) {
191  System.err.println("TreeExporter read something strange while processing " + currentDOMNode);
192  System.err.println("The node will be ignored.");
193  continue;
194  }
195 
196  if (ELEMENT_ANALYZER.equals(currentDOMNode.getNodeName())) {
197  analyzerSet.clear();
198  // TODO add support DC special cases
199 
200  AnalyzerBase analyzer;
201  try {
202  analyzer = (AnalyzerBase) Class.forName(currentDOMNode.getTextContent()).getDeclaredConstructor()
203  .newInstance();
204  } catch (InstantiationException | ClassNotFoundException | IllegalAccessException | IllegalArgumentException | InvocationTargetException | NoSuchMethodException | SecurityException | DOMException e) {
205  e.printStackTrace();
206  return null;
207  }
208  analyzerSet.add(analyzer);
209  currentDOMNode = walker.nextNode();
210  while (currentDOMNode != null && ELEMENT_ANALYZER.equals(currentDOMNode.getNodeName())) {
211  try {
212  analyzer = (AnalyzerBase) Class.forName(currentDOMNode.getTextContent())
213  .getDeclaredConstructor().newInstance();
214  } catch (InstantiationException | ClassNotFoundException | IllegalAccessException | IllegalArgumentException | InvocationTargetException | NoSuchMethodException | SecurityException | DOMException e) {
215  e.printStackTrace();
216  return null;
217  }
218  analyzerSet.add(analyzer);
219  currentDOMNode = walker.nextNode();
220  }
221 
222  AnalyzerCollection col = new AnalyzerCollection(currentName, analyzerSet.toArray(new AnalyzerBase[0]));
223 
224  parentTreeNode.setUserObject(col);
225 
226  while (currentDOMNode != null && getLevel(walker, currentDOMNode) <= parentTreeNode.getLevel()) {
227  parentTreeNode = (DefaultMutableTreeNode) parentTreeNode.getParent();
228  }
229  } else {
230  currentName = currentDOMNode.getAttributes().getNamedItem(ATTRIBUTE_NAME).getNodeValue();
231  DefaultMutableTreeNode node = new DefaultMutableTreeNode(new AnalyzerCollection(currentName));
232  parentTreeNode.add(node);
233  parentTreeNode = node;
234  currentDOMNode = walker.nextNode();
235  }
236  }
237 
238  return treeRoot;
239  }

250  @SuppressWarnings("rawtypes")
251  public static boolean writeTree(DefaultMutableTreeNode root, String filename) throws FileNotFoundException {
252 
253  DocumentBuilderFactory docFactory = DocumentBuilderFactory.newInstance();
254  DocumentBuilder docBuilder;
255  try {
256  docBuilder = docFactory.newDocumentBuilder();
257  } catch (ParserConfigurationException e1) {
258  e1.printStackTrace();
259  return false;
260  }
261  Document doc = docBuilder.newDocument();
262 
263  Element rootElement = doc.createElement(ELEMENT_ROOT);
264  rootElement.setAttribute(ATTRIBUTE_VERSION, VERSION);
265  doc.appendChild(rootElement);
266  Element currentElement = null;
267  Element currentFile = null;
268  Element currentTab = null;
269  Element currentSplit;
270 
271  Enumeration en = root.preorderEnumeration();
272  while (en.hasMoreElements()) {
273  DefaultMutableTreeNode node = (DefaultMutableTreeNode) en.nextElement();
274 
275  AnalyzerCollection analyzers = (AnalyzerCollection) node.getUserObject();
276  // System.out.println(node.getUserObject());
277 
278  int level = node.getLevel();
279  if (level == 0) {
280  continue; // ignore root
281  }
282 
283  switch (level) {
284  case 1:
285  currentFile = doc.createElement(ELEMENT_FILE);
286  currentElement = currentFile;
287  rootElement.appendChild(currentFile);
288  break;
289  case 2:
290  currentTab = doc.createElement(ELEMENT_TAB);
291  currentElement = currentTab;
292  currentFile.appendChild(currentTab);
293  break;
294  case 3:
295  currentSplit = doc.createElement(ELEMENT_SPLIT);
296  currentElement = currentSplit;
297  currentTab.appendChild(currentSplit);
298  break;
299  }
300 
301  currentElement.setAttribute(ATTRIBUTE_NAME, analyzers.getName());
302  for (AnalyzerBase a : analyzers.getAnalyzers()) {
303  Element e = doc.createElement(ELEMENT_ANALYZER);
304  e.appendChild(doc.createTextNode(a.getClass().getName()));
305  currentElement.appendChild(e);
306  }
307  }
308 
309  TransformerFactory transformerFactory = TransformerFactory.newInstance();
310  Transformer transformer;
311  try {
312  transformer = transformerFactory.newTransformer();
313  transformer.setOutputProperty(OutputKeys.INDENT, "yes");
314  transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
315  DOMSource source = new DOMSource(doc);
316  StreamResult result = new StreamResult(new FileOutputStream(filename));
317  transformer.transform(source, result);
318  } catch (TransformerException e) {
319  e.printStackTrace();
320  return false;
321  }
322 
323  return true;
324 
325  }
