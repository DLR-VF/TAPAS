56  public void analyzeSegments() {
57  for (Trip trip : trips) {
58  if (trip.mode != 2) {
59  continue;
60  }
61  CarSize size = TPS_Car.getCarSize(trip.car_kba);
62  FuelType fuel = trip.fuelType;
63  EcoMoveCar ref = null;
64  boolean ecoCar = trip.car_kba >= 100;
65  for (EcoMoveCar car : this.knownCarTypes) {
66  if (car.fuelType.equals(fuel) && car.size.equals(size) && car.ecoCar == ecoCar) {
67  ref = car;
68  break;
69  }
70  }
71  //new car found
72  if (ref == null) {
73  ref = new EcoMoveCar();
74  ref.fuelType = fuel;
75  ref.size = size;
76  ref.ecoCar = ecoCar;
77  //store new element in lists/maps
78  this.knownCarTypes.add(ref);
79  this.segmentCount.put(ref, 0);
80  this.segmentDistances.put(ref, 0.0);
81 
82  }
83 
84  //adopt numbers
85  this.segmentCount.put(ref, this.segmentCount.get(ref) + 1);
86  this.segmentDistances.put(ref, this.segmentDistances.get(ref) + trip.dist);
87  int tripByThisCar = 1;
88  if (this.segmentCarTrips.containsKey(trip.car_id)) {
89  tripByThisCar += this.segmentCarTrips.get(trip.car_id);
90  }
91  segmentCarTrips.put(trip.car_id, tripByThisCar);
92  this.carFleet.put(trip.car_id, ref);
93 
94  int histogammSegment = (int) (trip.dist + 0.5); // incl. round
95  histogammSegment -= histogammSegment % meterStepsForHistogram; // drop the remainer
96  int number = 1;
97  if (this.distHistogram.containsKey(histogammSegment)) {
98  number = this.distHistogram.get(histogammSegment) + 1;
99  }
100  this.distHistogram.put(histogammSegment, number);
101  }
102 
103  for (Entry<Integer, EcoMoveCar> entry : this.carFleet.entrySet()) {
104  EcoMoveCar ref = entry.getValue();
105  int count = 1;
106  if (this.segmentCars.containsKey(ref)) {
107  count += this.segmentCars.get(ref);
108  }
109  this.segmentCars.put(ref, count);
110  }
111 
112  }

114  public void printResults() {

116  for (Entry<EcoMoveCar, Double> entry : this.segmentDistances.entrySet()) {

120  entry.getKey().fuelType.toString(), entry.getKey().ecoCar ? "ecoCar" : "NonEco",

156  public void readTrips() {
157  String query = "";
158  try {
159  boolean recordsFound;
160  int step = 0;
161  int chunk = 1000000;
162  int pID = -1, hhID = -1;
163  do {
164  recordsFound = false;
165  query = "SELECT trip.p_id, trip.hh_id, trip.start_time_min, trip.mode, trip.car_type, trip.distance_real_m as dist, cars.kba_no, cars.engine_type FROM " +
166  regionName + "_trips_" + key + " as trip LEFT OUTER JOIN core." + regionName +
167  "_cars as cars ON (trip.car_type=cars.car_id AND cars.car_key='" + fleet_key +
168  "') WHERE trip.mode =2 ORDER BY hh_id, p_id, start_time_min LIMIT " + chunk + " OFFSET " +
169  (step * chunk);
170  ResultSet rs = dbCon.executeQuery(query, this);
171  step++;
172  while (rs.next()) {
173  recordsFound = true;
174  int actPID = rs.getInt("p_id");
175  int actHHID = rs.getInt("hh_id");
176  if (pID != actPID || hhID != actHHID) {
177  this.tripCounter++;
178  pID = actPID;
179  hhID = actHHID;
180  }
181  Trip trip = new Trip(actPID, actHHID, rs.getInt("start_time_min"));
182  trip.car_id = rs.getInt("car_type");
183  trip.car_kba = rs.getInt("kba_no");
184  trip.fuelType = TPS_Car.FUEL_TYPE_ARRAY[rs.getInt("engine_type")];
185  trip.dist = rs.getDouble("dist");
186  trip.mode = rs.getInt("mode");
187 
188  this.trips.add(trip);
189 
190  }
191  rs.close();
192  } while (recordsFound);
193 
194  } catch (SQLException e) {
195  System.err.println(this.getClass().getCanonicalName() + " readTrips: SQL-Error during statement: " + query);
196  e.printStackTrace();
197  }
198  }
