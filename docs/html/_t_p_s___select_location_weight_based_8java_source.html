46  @Override
47  public TPS_Location selectLocationFromChoiceSet(TPS_RegionResultSet choiceSet, TPS_Plan plan, TPS_PlanningContext pc, TPS_LocatedStay locatedStay) {
48  if (this.PM == null) {
49  TPS_Logger.log(SeverenceLogLevel.FATAL,
50  "TPS_LocationSelectModel not properly initialized! Persistance manager is null?!?! Driving home!");
51  return plan.getPerson().getHousehold().getLocation();
52  }
53  if (this.region == null) {
54  TPS_Logger.log(SeverenceLogLevel.FATAL,
55  "TPS_LocationSelectModel not properly initialized! Region is null?!?! Driving home!");
56  return plan.getPerson().getHousehold().getLocation();
57  }
58  TPS_Stay stay = locatedStay.getStay();
59  TPS_ActivityConstant actCode = stay.getActCode();
60  TPS_TourPart tourpart = (TPS_TourPart) locatedStay.getStay().getSchemePart();
61  TPS_Stay comingFrom = tourpart.getStayHierarchy(stay).getPrevStay();
62  TPS_Stay goingTo = tourpart.getStayHierarchy(stay).getNextStay();
63  TPS_Location locComingFrom = plan.getLocatedStay(comingFrom).getLocation();
64  TPS_Location locGoingTo = plan.getLocatedStay(goingTo).getLocation();
65  TPS_SettlementSystem regionType = locComingFrom.getTrafficAnalysisZone().getBbrType();
66 
67  // different cnf4-params according to the type of trip
68  double cfn4 = region.getCfn().getCFN4Value(regionType, actCode);
69  double cnfX = region.getCfn().getCFNXValue(regionType);
70  double param = 1;
71  TPS_CFN pot = region.getPotential();
72  if(pot!=null){
73  double val = pot.getParamValue(regionType, actCode);
74  if (val>=0)
75  param = val;
76  };
77 
78  SortedSet<WeightedResult> weightedChoiceSet = new TreeSet<>();
79  TPS_Location locRepresentant = null;
80  double sumWeight = 0;
81  for (Result result : choiceSet.getResultIterable()) {
82  if (result.sumWeight <= 0) continue;
83  // Draw a location at random. This location represents the zone for this round.
84  TPS_TrafficAnalysisZone taz = result.taz;
85  locRepresentant = result.loc;
86  //here we can switch between different travel time models
87  TPS_ModeChoiceContext prevMCC = new TPS_ModeChoiceContext();
88  prevMCC.isBikeAvailable = pc.isBikeAvailable;
89  prevMCC.carForThisPlan = pc.carForThisPlan;
90  prevMCC.duration = stay.getOriginalDuration();
91  prevMCC.startTime = stay.getOriginalStart();
92  prevMCC.fromStayLocation = locComingFrom;
93  prevMCC.fromStay = comingFrom;
94  prevMCC.toStayLocation = locRepresentant;
95  prevMCC.toStay = stay;
96  TPS_ModeChoiceContext nextMCC = new TPS_ModeChoiceContext();
97  nextMCC.isBikeAvailable = pc.isBikeAvailable;
98  nextMCC.carForThisPlan = pc.carForThisPlan;
99  nextMCC.duration = stay.getOriginalDuration();
100  nextMCC.startTime = stay.getOriginalStart();
101  nextMCC.fromStayLocation = locRepresentant;
102  nextMCC.fromStay = stay;
103  nextMCC.toStayLocation = locGoingTo;
104  nextMCC.toStay = goingTo;
105 
106  double weightedTT = this.getTravelTime(plan, pc, prevMCC, nextMCC, taz, this.PM.getParameters());
107  if (weightedTT > 0) {
108  //here we can switch between different Opportunity-weighting
109  WeightedResult weightedResult = this.createLocationOption(result,
110  weightedTT,param); //weight wird mit steigender Reisezeit abgewertet
111  sumWeight += weightedResult.getAdaptedWeight();
112  weightedChoiceSet.add(weightedResult);
113  }
114  }
115 
116  //Collections.sort(weightedChoiceSet); //sort descending!!
117 
118  if (weightedChoiceSet.size() > 0) {
119  double rand = Randomizer.random(); //uniform distribution from 0 to 1
120  double posMicro = rand;
121 // if(actCode.getCode(TPS_ActivityCodeType.TAPAS)==3) {
122 // rand = 0; //hit
123 // }
124 
125  //double posMicro = ( Math.log(1.0 - rand) / Math.log(1.0 - cfn4));
126  if (this.PM.getParameters().isDefined(ParamValue.GAMMA_LOCATION_WEIGHT)) {
127  posMicro = Math.pow(posMicro, this.PM.getParameters().getDoubleValue(ParamValue.GAMMA_LOCATION_WEIGHT));
128  }
129  //double posMicro = 1.0 - rand;
130  posMicro *= sumWeight; // norm to max weight
131  posMicro *= cnfX; //apply region-specific restiction factor
132  posMicro *= cfn4; // apply activity based restiction factor
133 
134  double weightPos = 0;
135  for (WeightedResult entry : weightedChoiceSet) {
136  weightPos += entry.getAdaptedWeight();
137  if (weightPos >= posMicro) {
138  locRepresentant = entry.result.loc;
139  break;
140  }
141  }
142  } else {
143  TPS_Logger.log(SeverenceLogLevel.WARN,
144  "No specific location found for activity " + actCode.getCode(TPS_ActivityCodeType.ZBE));
145  return region.selectDefaultLocation(plan, pc, locatedStay);
146  }
147  return locRepresentant;
148  }
