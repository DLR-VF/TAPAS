95  private HashMap<DefaultMutableTreeNode, Analyzer> buildAnalyzerList() {
96  HashMap<DefaultMutableTreeNode, Analyzer> result = new HashMap<>();
97  Enumeration en = root.depthFirstEnumeration();
98  while (en.hasMoreElements()) {
99  DefaultMutableTreeNode node = (DefaultMutableTreeNode) en.nextElement();
100  if (node.isLeaf()) {
101 
102  Object[] oPath = node.getUserObjectPath();
103  ArrayList<AnalyzerBase> analyzerPath = new ArrayList<>();
104  for (Object o : oPath) {
105  analyzerPath.addAll(Arrays.asList(((AnalyzerCollection) o).getAnalyzers()));
106  }
107  result.put(node, new Analyzer(analyzerPath.toArray(new AnalyzerBase[0])));
108  }
109  }
110 
111  if (writeToDB) {// add the calibration results analyzer
112  ModeAnalyzer mo = new ModeAnalyzer();
113  TripIntentionAnalyzer ti = new TripIntentionAnalyzer();
114  DefaultDistanceCategoryAnalyzer dc = new DefaultDistanceCategoryAnalyzer();
115  PersonGroupAnalyzer pg = new PersonGroupAnalyzer(mo, ti, dc);
116 
117  Analyzer analyzer = new Analyzer(mo, pg, ti, dc);
118  result.put(null, analyzer);
119  }
120 
121  return result;
122  }

142  } catch (IOException | WriteException e) {

144  } finally {
145  if (excelSuccess) {
