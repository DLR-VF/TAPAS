103  public static JMenu init(final Class<?> guiClass) {
104  JMenu menu = new JMenu("-");
105  try {
106 
107  String user = System.getProperty("user.name");
108  Properties userProperties = load(MultilanguageSupport.class);
109  if (userProperties.containsKey(user)) {
110  setLocale(new Locale(userProperties.getProperty(user)));
111  } else {
112  setLocale(new Locale(Locale.getDefault().getLanguage()));
113  store(locale);
114  }
115  ResourceBundle MLSRB = ResourceBundle.getBundle(MultilanguageSupport.class.getSimpleName() + "Labels",
116  locale);
117  menu.setText(MLSRB.getString("menu"));
118  Properties languages = new Properties();
119  String resource = "MultilanguageSupport.properties";
120  InputStream is = ClassLoader.getSystemResourceAsStream(resource);
121  languages.load(is);
122  StringTokenizer st = new StringTokenizer(languages.getProperty("languages"), ", ");
123  while (st.hasMoreTokens()) {
124  String language = st.nextToken();
125  JMenuItem item = new JMenuItem(MLSRB.getString("menu.item." + language));
126  item.setName(language);
127  item.addActionListener(e -> {
128  change(new Locale(((JMenuItem) e.getSource()).getName()));
129  JOptionPane.showMessageDialog(null, MLSRB.getString("message"));
130  });
131  menu.add(item);
132  }
133  RB = ResourceBundle.getBundle(guiClass.getSimpleName() + "Labels", locale);
134  } catch (IOException e) {
135  TPS_Logger.log(SeverenceLogLevel.ERROR, e);
136  JOptionPane.showMessageDialog(null, e.getMessage(), "Error storing user language information",
137  JOptionPane.ERROR_MESSAGE);
138  }
139 
140  return menu;
141  }

163  private static void store(Locale locale) throws IOException {
164  Properties userProperties = load(MultilanguageSupport.class);
165  setLocale(locale);

167  String comment = "Auto generated users properties file " + "storing language information for each user. " +
