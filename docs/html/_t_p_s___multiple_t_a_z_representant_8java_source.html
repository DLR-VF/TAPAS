54  public TPS_RegionResultSet getLocationRepresentatives(TPS_Plan plan, TPS_PlanningContext pc, TPS_LocatedStay locatedStay, TPS_ParameterClass parameterClass) {
55 
56  long time = System.nanoTime();
57  TPS_RegionResultSet regionRS = new TPS_RegionResultSet();
58  TPS_TourPart tourpart = (TPS_TourPart) locatedStay.getStay().getSchemePart();
59 
60  TPS_Stay stay = locatedStay.getStay();
61  TPS_ActivityConstant activityCode = locatedStay.getStay().getActCode();
62  TPS_Stay comingFrom = tourpart.getStayHierarchy(stay).getPrevStay();
63  TPS_Stay goingTo = tourpart.getStayHierarchy(stay).getNextStay();
64  TPS_Location locComingFrom = plan.getLocatedStay(comingFrom).getLocation();
65  TPS_Location locGoingTo = plan.getLocatedStay(goingTo).getLocation();
66  TravelDurations td = tourpart.getTravelDurations(stay);
67  double arrivalDuration = td.getArrivalDuration(plan.getPM().getParameters());
68  double departureDuration = td.getDepartureDuration(plan.getPM().getParameters());
69 
70  //store some statistics:
71  Integer[] tupel = TPS_Main.actStats.get(activityCode.getCode(TPS_ActivityCodeType.ZBE));
72 
73  if (tupel == null) {
74  tupel = new Integer[3];
75  tupel[0] = 0;
76  tupel[1] = 0;
77  tupel[2] = 0;
78  }
79 
80  tupel[0]++;
81  tupel[1] += (int) arrivalDuration;
82  tupel[2] += (int) departureDuration;
83 
84  TPS_Main.actStats.put(activityCode.getCode(TPS_ActivityCodeType.ZBE), tupel);
85 
86 
87  int i;
88  // normal case
89  // speed slices from MAX_SYSTEM_SPEED/ MAX_TRIES_LOCATION_SELECTION to
90  // MAX_SYSTEM_SPEED
91  double incFactor = 1.0 / parameterClass.getDoubleValue(ParamValue.MAX_TRIES_LOCATION_SELECTION);
92  //double incFactorTime = 1.0 / ParamValue.MAX_TRIES_LOCATION_SELECTION.getDoubleValue();
93  double arrivalDistance = 0, departureDistance = 0, actDistance;
94 
95  // TODO: make this better!
96  // some trips start per default at different i and not at i=1!
97 
98  switch (activityCode.getCode(TPS_ActivityCodeType.TAPAS)) {
99  case 1: // work
100  // always look at the maximum range!
101  i = parameterClass.getIntValue(ParamValue.MAX_TRIES_LOCATION_SELECTION);
102  break;
103  case 0:
104  case 2: // school
105  case 3: // shopping
106  case 4: // private matters
107  case 5:// free time
108  case 6: // other
109  case 7: // university
110  default:
111  i = 1; // start as close as possible
112  break;
113  }
114 
115  for (; i <= parameterClass.getIntValue(ParamValue.MAX_TRIES_LOCATION_SELECTION) + 1 &&
116  regionRS.size() < 2 * numOfTazRepresentants; i++) {
117 
118  regionRS.clear();
119 
120  for (TPS_TrafficAnalysisZone taz : region) {
121  if (taz.getData() == null) {
122  // empty taz data
123  continue;
124  }
125  if (!taz.allowsActivity(activityCode)) {
126  // no activities of this type
127  continue;
128  }
129  // filter restricted tazes if a restricted car is used
130  if (tourpart.getCar() != null && tourpart.isCarUsed() && tourpart.getCar().isRestricted() &&
131  taz.isRestricted()) continue;
132 
133  if (parameterClass.isTrue(ParamFlag.FLAG_FILTER_SHOPPING_CHOICE_SET) || !stay.isShopping()) {
134  actDistance = TPS_Mode.get(ModeType.WALK).getDistance(taz, locGoingTo, SimulationType.SCENARIO,
135  null);
136  actDistance += TPS_Mode.get(ModeType.WALK).getDistance(locComingFrom, taz, SimulationType.SCENARIO,
137  null);
138 
139 
140  if (i <= parameterClass.getIntValue(ParamValue.MAX_TRIES_LOCATION_SELECTION)) {
141  arrivalDistance = arrivalDuration * parameterClass.getDoubleValue(ParamValue.MAX_SYSTEM_SPEED) *
142  incFactor * (double) i;
143  departureDistance = departureDuration * parameterClass.getDoubleValue(
144  ParamValue.MAX_SYSTEM_SPEED) * incFactor * (double) i;
145  if (actDistance > (departureDistance + arrivalDistance)) {
146  continue;
147  }
148  } else {
149  // desperate last try: drop distance constraint and keep on going!
150  }
151  }
152 
153 
154  double weight = taz.getActivityWeightSum(activityCode);
155  if (weight <= 0) {
156  // do not process "zero weight"-locations
157  continue;
158  }
159  List<TPS_Location> locList = generateLocationRepr(activityCode, taz);
160  // first sum the weight of the selected representatives (the list may be empty)
161  double sumWeight = 0;
162  for (TPS_Location loc : locList) {
163  sumWeight += loc.getData().getWeight();
164  }
165  // now weight the total weight of the cell by the normalized weight of this location
166  // in the set of representants
167  for (TPS_Location loc : locList) {
168  regionRS.add(taz, loc, weight * loc.getData().getWeight() / sumWeight);
169  }
170  }
171  }
172 
173  time = System.nanoTime() - time;
174  if (TPS_Logger.isLogging(SeverenceLogLevel.DEBUG)) {
175  TPS_Logger.log(SeverenceLogLevel.DEBUG,
176  "Selected traffic analysis zones (size=" + regionRS.size() + ") in " + (time / 1000000.0) + "ms");
177  }
178  return regionRS;
179  }
