54  public HashMap<Long, PlanDeviation> calculatePlanDerivation(int iteration) {
55  HashMap<Long, PlanDeviation> map = new HashMap<>();
56 
57  String query = "";
58  String key = this.parameterClass.getString(ParamString.RUN_IDENTIFIER);
59  String region = this.parameterClass.getString(ParamString.DB_REGION);
60  try {
61  boolean recordsFound;
62  int step = 0;
63  int chunk = 1000000;
64  Connection con = this.dbManager.getConnection(this);
65  Statement stat = con.createStatement();
66  ResultSet rs;
67  int hhID, pID, mode;
68  long keyVal;
69  PlanDeviation timeDiff;
70  double ttNew, distNew, ttOld, distOld;
71 
72  do {
73  recordsFound = false;
74  query = "SELECT trip.p_id, trip.hh_id, trip.start_time_min, " +
75  "it.travel_time_sec as tt_new, trips.travel_time_sec as tt_old " +
76  "it.travel_distance_m as dist_new, trips.travel_distance_m as dist_old, trips.mode " + "FROM " +
77  region + "_trips_" + key + " AS trips " + "LEFT OUTER JOIN temp.trips_" + key + " AS it " +
78  "ON (it.p_id = trips.p_id AND it.hh_id = trips.hh_id AND it.start_time_min == trips.start_time_min AND it.iteration=" +
79  iteration + ") " + "ORDER BY trips.hh_id, trips.p_id, trips.start_time_min LIMIT " + chunk +
80  " OFFSET " + (step * chunk);
81  rs = stat.executeQuery(query);
82 
83  step++; // for the next chunk
84 
85  while (rs.next()) {
86  recordsFound = true;
87  hhID = rs.getInt("hh_id");
88  pID = rs.getInt("p_id");
89  keyVal = hhAndPersonIDToHash(hhID, pID);
90 
91  if (map.containsKey(keyVal)) { // is this person known?
92  timeDiff = map.get(keyVal);
93  } else {
94  timeDiff = new PlanDeviation(); //new person
95  }
96 
97  mode = rs.getInt("mode");
98  ttOld = rs.getDouble("tt_old");
99  distOld = rs.getDouble("dist_old");
100  if (rs.getObject("tt_new") != null) {
101  ttNew = rs.getDouble("tt_new");
102  } else {
103  ttNew = ttOld; //no update information
104  }
105 
106  if (rs.getObject("dist_new") != null) {
107  distNew = rs.getDouble("dist_new");
108  } else {
109  distNew = distOld; //no update information
110  }
111 
112  timeDiff.newTime += ttNew;
113  timeDiff.oldTime += ttOld;
114  timeDiff.newDist[mode] += distNew; // mode dependent distance! : change of costs!
115  timeDiff.oldDist[mode] += distOld;
116  map.put(keyVal, timeDiff);
117 
118  }
119  } while (recordsFound);
120 
121 
122  } catch (SQLException e) {
123  TPS_Logger.log(SeverenceLogLevel.ERROR, "Exception during SQL! Query: " + query, e);
124  }
125 
126 
127  return map;
128  }

178  public void resetPlansForRecalculation(ArrayList<Long> plans) {
179  String query = "";
180 
181  try {
182  TPS_TripDeleter worker = new TPS_TripDeleter(this.parameterClass);
183  Connection con = this.dbManager.getConnection(worker);
184 
185  worker.setKey(this.parameterClass.getString(ParamString.RUN_IDENTIFIER));
186  // read HH-IDs
187  //performance tweak: 10000 -chunks!
188  int counter = 0;
189  for (Long id : plans) {
190  worker.putHHID(this.hashToHHID(id));
191  counter++;
192  if (counter == 10000) {
193  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
194  TPS_Logger.log(SeverenceLogLevel.INFO, "Deleting " + counter + " households from triptable");
195  }
196  worker.deleteHouseholdsFromTriptable(con);
197  worker.clearHHID();
198  counter = 0;
199  }
200  }
201  //rest
202  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
203  TPS_Logger.log(SeverenceLogLevel.INFO, "Deleting " + counter + " households from triptable");
204  }
205  worker.deleteHouseholdsFromTriptable(con);
206  worker.clearHHID();
207 
208  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
209  TPS_Logger.log(SeverenceLogLevel.INFO, "Done!");
210  }
211  } catch (SQLException e) {
212  TPS_Logger.log(SeverenceLogLevel.ERROR, "Exception during SQL! Query: " + query, e);
