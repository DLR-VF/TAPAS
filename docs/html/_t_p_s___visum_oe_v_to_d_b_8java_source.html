171  public void calcBeelines() {
172  double bl;
173  Integer from, to;
174  for (Entry<Integer, TAZ> i : this.TAZes.entrySet()) {
175  for (Entry<Integer, TAZ> j : this.TAZes.entrySet()) {
176  if (i.getValue().ignore || j.getValue().ignore) continue;
177  from = i.getValue().taz_id - 1;
178  to = j.getValue().taz_id - 1;
179  bl = 0;
180  this.nodes[from][to].idStart = i.getKey();
181  this.nodes[from][to].idEnd = j.getKey();
182  this.nodes[to][from].idStart = j.getKey();
183  this.nodes[to][from].idEnd = i.getKey();
184  if (!from.equals(to)) {
185  bl = this.TAZes.get(i.getKey()).getDistance(this.TAZes.get(j.getKey()));
186  }
187  bl /= 1000.0; //km
188  //symmetric!
189  this.nodes[from][to].bld = this.nodes[to][from].bld = bl;
190  }
191  }

212  public void calcTTWithoutWaiting(int secPerTransit) {
213  double factor = secPerTransit;
214  factor /= 100.0;
215  int numTWTbetter = 0, numNRTbetter = 0;
216  double ttTWT = 0, ttNRT = 0, ttWait = 0;
217  for (PTNode[] node : this.nodes) {
218  for (PTNode ptNode : node) {
219  if (ptNode.ivt >= 0 && ptNode.twt >= 0 && ptNode.bdh >= 0) {
220  //sum bdh (instead of initial waiting time), ivt (in vehicle time) and twt (total waiting time)
221  //nrt = number of transfers is transfers*100, therefore the factor normalizes this integer
222  ptNode.sumTT = ptNode.bdh + ptNode.ivt +
223  //this.nodes[i][j].swt
224  +Math.min(ptNode.twt, (ptNode.nrt * factor)); //constant time per transfer
225 
226  if (ptNode.nrt > 0) {
227  ttWait += Math.min(ptNode.twt, (ptNode.nrt * factor));
228  ttTWT += ptNode.twt;
229  ttNRT += ptNode.nrt * factor;
230  if (ptNode.twt >= (ptNode.nrt * factor)) {
231  numTWTbetter++;
232 
233  } else {
234  numNRTbetter++;
235  }
236  }
237  } else ptNode.sumTT = -1; //no connection
238  }
239  }
240  System.out.println("TWT better: " + numTWTbetter + " nrt better: " + numNRTbetter);
241  System.out.println("Wait sum: " + ttWait + " TWT sum: " + ttTWT + " nrt sum: " + ttNRT);

247  public void calcTop3(double factor) {
248 
249  int i, j;
250  List<Double> row = null;
251  if (this.nodes.length < 4) return;
252  factor /= 3.0; //incl normalization of three elements
253  for (i = 0; i < this.nodes.length; ++i) {
254  //access time
255  row = new ArrayList<>();
256  for (j = 0; j < this.nodes[i].length; ++j) {
257  if (i != j && this.nodes[i][j].act >= 0) row.add(this.nodes[i][j].act);
258  }
259  this.nodes[i][i].act = this.getTop3FromRow(row, factor);
260  //egress time
261  row.clear();
262  for (j = 0; j < this.nodes[i].length; ++j) {
263  if (i != j && this.nodes[i][j].egt >= 0) row.add(this.nodes[i][j].egt);
264  }
265  this.nodes[i][i].egt = this.getTop3FromRow(row, factor);
266  //waiting time
267  row.clear();
268  for (j = 0; j < this.nodes[i].length; ++j) {
269  if (i != j && this.nodes[i][j].twt >= 0) row.add(this.nodes[i][j].twt);
270  }
271  this.nodes[i][i].twt = this.getTop3FromRow(row, factor);
272  //travel time
273  row.clear();
274  for (j = 0; j < this.nodes[i].length; ++j) {
275  if (i != j && this.nodes[i][j].ivt >= 0) row.add(this.nodes[i][j].ivt);
276  }
277  this.nodes[i][i].ivt = this.getTop3FromRow(row, factor);
278  //travel dist
279  row.clear();
280  for (j = 0; j < this.nodes[i].length; ++j) {
281  if (i != j && this.nodes[i][j].jrd >= 0) row.add(this.nodes[i][j].jrd);
282  }
283  this.nodes[i][i].jrd = this.getTop3FromRow(row, factor);
284 
285  //sum travel time
286  row.clear();
287  for (j = 0; j < this.nodes[i].length; ++j) {
288  if (i != j && this.nodes[i][j].sumTT >= 0) row.add(this.nodes[i][j].sumTT);
289  }
290  this.nodes[i][i].sumTT = this.getTop3FromRow(row, factor);
291 
292  }

510 
511  for (Integer entry : ids.toArray(new Integer[0])) {
