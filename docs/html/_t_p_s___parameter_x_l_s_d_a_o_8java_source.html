145  wb = new HSSFWorkbook();
146  CellStyle[] cellStyles = new CellStyle[header.length];
147  for (int i = 0; i < header.length; i++) {
148  CellStyle style = wb.createCellStyle();
149  style.setFillForegroundColor(indexedColors[i].getIndex());
150  style.setFillPattern(FillPatternType.SOLID_FOREGROUND);

158  HSSFSheet[] sheets = new HSSFSheet[prefixes.length];
159  int[] rows = new int[prefixes.length];
160  Arrays.fill(rows, 0);
161  Font font = wb.createFont();
162  font.setBold(true);
163 
164  for (int i = 0; i < sheets.length; i++) {
165  sheets[i] = wb.createSheet(prefixes[i] + name);
166  row = sheets[i].createRow(rows[i]++);

173  cell.setCellStyle(cellStyle);

178 
179  for (int i = 0; i < params.length; i++) {
180  row = sheets[0].createRow(rows[0]++);
181  cell = row.createCell(0);
182  cell.setCellValue(params[i].name());
183  cell.setCellStyle(cellStyles[0]);

186  cell.setCellStyle(cellStyles[1]);

188 
189  row = sheets[0].createRow(rows[0]++);
190  cell = row.createCell(0);
191  cell.setCellValue(ParamString.CLASS_DATA_SCOURCE_ORIGIN.name());
192  cell.setCellStyle(cellStyles[0]);
193  cell = row.createCell(1);
194  cell.setCellValue(TPS_DB_IOManager.class.getName());
195  cell.setCellStyle(cellStyles[1]);
196 
197  try {
198  breader = new BufferedReader(new FileReader(oldFile));
199  String[] entry = new String[header.length];
200  Enum<?> e;
201  ParamType pt = null;
202  boolean defined = false;
203  String definition = null;
204 
205  for (String line = breader.readLine(); line != null; line = breader.readLine()) {
206  if (line.startsWith("#")) {
207  line = line.replaceAll("#", "");
208  line = line.replaceAll("\n", " ");
209  line = line.replaceAll("\t", " ");
210  int startIndex = 0;
211  while (startIndex < line.length() && line.charAt(startIndex) == ' ') {
212  startIndex++;
213  }
214  if (startIndex < line.length()) entry[2] += line.substring(startIndex) + " ";
215  } else if (line.length() > 2) {
216  StringTokenizer st = new StringTokenizer(line, " =,");
217  if (st.countTokens() < 2) {
218  System.err.println("Skipped [no value] in " + oldFile.getName() + ": " + line);
219  entry[2] = "";
220  continue;
221  }
222  entry[0] = st.nextToken();
223  entry[1] = st.nextToken();
224  e = this.map.get(entry[0]);
225 
226  if (e == null) {
227  System.err.println("Skipped [deprecated] in " + oldFile.getName() + ": " + line);
228  entry[2] = "";
229  continue;
230  }
231 
232  if (e instanceof ParamFlag) {
233  ParamFlag pf = (ParamFlag) e;
234  pt = parameterClass.getType(pf);
235  if (defined = parameterClass.isDefined(pf)) definition = Boolean.toString(
236  parameterClass.isTrue(pf));
237  } else if (e instanceof ParamString) {
238  ParamString ps = (ParamString) e;
239  pt = parameterClass.getType(ps);
240  if (defined = parameterClass.isDefined(ps)) definition = parameterClass.getString(ps);
241  } else if (e instanceof ParamValue) {
242  ParamValue pv = (ParamValue) e;
243  pt = parameterClass.getType(pv);
244  if (defined = parameterClass.isDefined(pv)) definition = Double.toString(
245  parameterClass.getDoubleValue(pv));
246  } else {
247  System.err.println("Skipped [???] in " + oldFile.getName() + ": " + line);
248  entry[2] = "";
249  continue;
250  }
251 
252  row = sheets[pt.getIndex()].createRow(rows[pt.getIndex()]++);
253 
254  if (defined) {
255  String text =
256  "Parameter " + e.name() + " is already defined with the default value '" + definition +
257  "'. Would you like to overwrite this parameter with the new value '" +
258  entry[1] + "'?";
259  int option = JOptionPane.showConfirmDialog(null, text,
260  "Confirm overwrite default parameter value", JOptionPane.YES_NO_OPTION);
261  if (option == JOptionPane.NO_OPTION) {
262  entry[1] = definition;
263  }
264  }
265 
266  for (int i = 0; i < entry.length; i++) {
267  cell = row.createCell(i);
268  cell.setCellValue(entry[i]);
269  cell.setCellStyle(cellStyles[i]);
270  }
271 
272  entry[2] = "";
273  }
274  }
275 
276  breader.close();
277 
278  wb.write(new FileOutputStream(new File(newPath, name + ".xls")));
279  wb.close();
280  } catch (IOException ioe) {
281  ioe.printStackTrace();
282  }
