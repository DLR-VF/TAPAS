216  public void redistributeCars() {
217  int numOfExchangedCars = 0;
218  //loop over all house holds
219  for (HouseHoldWithCars badCarHH : this.householdsWithBadCars) {
220  //loop over all cars
221  for (int i = 0; i < badCarHH.carIDs.length; ++i) {
222  //is this a restricted car?
223  if (this.carRestriction.get(badCarHH.carIDs[i]) && this.householdsWithGoodCars.size() > 0) {
224  //find a random unrestricted partner household
225  int exchangePartner = (int) (Math.random() * this.householdsWithGoodCars.size());
226  boolean hasMoreGoodCars = false, carExchanged = false;
227  HouseHoldWithCars goodCarHH = this.householdsWithGoodCars.get(exchangePartner);
228  //find the first good car
229  for (int j = 0; j < goodCarHH.carIDs.length; ++j) {
230  if (!this.carRestriction.get(goodCarHH.carIDs[j])) {
231  if (!carExchanged) {
232  //exchange these cars
233  int tmp = goodCarHH.carIDs[j];
234  goodCarHH.carIDs[j] = badCarHH.carIDs[i];
235  badCarHH.carIDs[i] = tmp;
236  carExchanged = true;
237  numOfExchangedCars++;
238  } else { //this house holds more good cars!
239  hasMoreGoodCars = true;
240  }
241  }
242  }
243  //remove household from the good-list, if no good car is left
244  if (!hasMoreGoodCars) {
245  this.householdsWithGoodCars.remove(exchangePartner);
246  }
247  if (!this.householdsToUpdate.contains(goodCarHH)) { //add to update list if not previously known
248  this.householdsToUpdate.add(goodCarHH);
249  }
250  }
251  }
252  this.householdsToUpdate.add(badCarHH);
253  }
254  System.out.println("Exchanged cars: " + numOfExchangedCars);
255  System.out.println("Households to update: " + this.householdsToUpdate.size());
256  }

302  public void transformCars() {
303  int numOfExchangedCars = 0;
304 
305  //build a probability-vector
306  List<CarProbabaility> probs = new ArrayList<>();
307  double sum = 0;
308  for (Entry<Integer, Integer> e : this.carIDFrequency.entrySet()) {
309  if (!this.carRestriction.get(e.getKey())) { // an unrestricted car!
310  CarProbabaility tmp = new CarProbabaility();
311  tmp.id = e.getKey();
312  tmp.freqency = e.getValue();
313  if (tmp.freqency > 0) {
314  sum += e.getValue();
315  probs.add(tmp);
316  }
317  }
318  }
319  double accumulation = 0;
320  for (CarProbabaility car : probs) {
321  car.prob = accumulation + (double) car.freqency / sum;
322  accumulation = car.prob;
323  }
324 
325  //loop over all house holds
326  for (HouseHoldWithCars badCarHH : this.householdsWithBadCars) {
327  //loop over all cars
328  for (int i = 0; i < badCarHH.carIDs.length; ++i) {
329  //is this a restricted car?
330  if (this.carRestriction.get(badCarHH.carIDs[i])) {
331  double random = Math.random();
332  int id = badCarHH.carIDs[i];
333  for (CarProbabaility car : probs) {
334  if (car.prob > random) {
335  id = car.id;
336  numOfExchangedCars++;
337  break;
338  }
339  }
340 
341  badCarHH.carIDs[i] = id;
342  }
343  }
344  this.householdsToUpdate.add(badCarHH);
345  }
346  System.out.println("Exchanged cars: " + numOfExchangedCars);
