42  public ArrayList<Long> selectPlansForRecalculation(HashMap<Long, PlanDeviation> derivationMap, int actIter, int maxIter) {
43  ArrayList<Long> plans = new ArrayList<>();
44 
45  String query = "";
46  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
47  TPS_Logger.log(SeverenceLogLevel.INFO, "Selecting households for recalculation");
48  }
49 
50  try {
51  ResultSet rs;
52  // Anschmiegeparameter der EVA1-Funktion
53  double EBottom = this.parameterClass.getDoubleValue(ParamValue.OVERALL_TIME_E);
54  // EVA1-Parameter, der das Absinken der Akzeptanz bei geringer Überschreitung beeinflußt
55  double FTop = this.parameterClass.getDoubleValue(ParamValue.OVERALL_TIME_F);
56  double TurningPoint = this.parameterClass.getDoubleValue(ParamValue.MAX_TIME_DIFFERENCE);
57  //time slot factor für die Tagespläne
58  int timeSlotFactor = this.parameterClass.getIntValue(ParamValue.SEC_TIME_SLOT);
59 
60  for (Entry<Long, PlanDeviation> entry : derivationMap.entrySet()) {
61  int hh_id = this.hashToHHID(entry.getKey());
62  int p_id = this.hashToPersonID(entry.getKey());
63  int schemeID;
64  int origTT;
65  double oldAcc, newAcc, ratio, randomRate;
66  //get scheme id
67  query = "SELECT scheme_id FROM " + this.parameterClass.getString(ParamString.DB_TABLE_TRIPS) +
68  " WHERE hh_id =" + hh_id + " AND p_id = " + p_id + " LIMIT 1";
69  rs = dbManager.executeQuery(query, this);
70  if (rs.next()) {
71  schemeID = rs.getInt("scheme_id");
72  } else {
73  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {
74  TPS_Logger.log(SeverenceLogLevel.WARN,
75  "Household " + hh_id + " person " + p_id + " not found in trip-table. Query: " + query);
76  }
77  continue;
78  }
79 
80  //get scheme-travel time
81  origTT = 0;
82  query = "SELECT duration FROM " + this.parameterClass.getString(ParamString.DB_TABLE_EPISODE) +
83  " WHERE scheme_id =" + schemeID + " AND act_code_zbe = ANY(ARRAY[]) AND key = '" +
84  this.parameterClass.getString(ParamString.DB_EPISODE_KEY)+ "'";
85  rs = dbManager.executeQuery(query, this);
86  while (rs.next()) {
87  origTT += rs.getInt("duration");
88  }
89 
90  origTT *= timeSlotFactor;
91 
92 
93  //calc old acceptance
94  ratio = entry.getValue().oldTime / origTT;
95  ratio = Math.abs(ratio - 1.0);
96  oldAcc = TPS_GX.calculateEVA1Acceptance(ratio, EBottom, FTop, TurningPoint);
97 
98 
99  //calc new acceptance
100  ratio = entry.getValue().newTime / origTT;
101  ratio = Math.abs(ratio - 1.0);
102  newAcc = TPS_GX.calculateEVA1Acceptance(ratio, EBottom, FTop, TurningPoint);
103 
104  //calc random rate
105  randomRate = (1 - newAcc) * (1 - Math.abs(oldAcc - newAcc));
106 
107  if (Randomizer.random() <= randomRate) {
108  plans.add(this.hhAndPersonIDToHash(hh_id, 0));
109  }
110  }
111  } catch (SQLException e) {
112  TPS_Logger.log(SeverenceLogLevel.ERROR, "Exception during SQL! Query: " + query, e);
113  }
114  return plans;
115  }
116 }
