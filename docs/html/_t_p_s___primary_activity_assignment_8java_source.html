403  private TPS_DiscreteDistribution<Integer> probDistributionWorkplace() {
404  double beta1 = 0.108801184642458;
405  double beta2 = 0.229681522400213;
406  workplaceKeys = distanceCategories.keySet().toArray(new Integer[0]);
407  Double[] workplaceValuesDist = new Double[distanceCategories.size()];
408 
409  //fill the array
410  for (int i = 0; i < distanceCategories.size(); i++) {
411  workplaceValuesDist[i] = Math.exp(-beta1 * (workplaceKeys[i] / 1000)) - Math.exp(
412  -beta2 * (workplaceKeys[i] / 1000));
413  }
414  return new TPS_DiscreteDistribution<>(new ArrayList<>(distanceCategories.keySet()),
415  Arrays.asList(workplaceValuesDist));
416  }
