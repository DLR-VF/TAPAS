83  private boolean checkTableName(int row) {

118  @Override
119  public boolean isCellEditable(int row, int column) {
120  // select column
121  if (column == COL_ACC) {
122  return true;
123  }
124  // user generated rows
125  if (row >= separator) {
126  if (column == COL_T_KEY) {
127  return true;
128  } else if (column == COL_ID) {
129  return !this.getValueAt(row, column).toString().startsWith("-");
130  } else if (column == COL_T_NAME || column == COL_F_NAME) {
131  try {
132  ParamString.valueOf(this.getValueAt(row, 0).toString());
133  return true;
134  } catch (Exception e) {
135  return false;
136  }
137  }
138  }
139 
140  return false;
141  }

150  @Override
151  public void setValueAt(Object value, int row, int column) {
152  super.setValueAt(value, row, column);
153 
154  if (column == COL_T_KEY || column == COL_T_NAME) this.check(row);
155 
156  if (row >= separator) {
157  if (column == COL_T_KEY) {
158  ParamString ps = ParamString.valueOf(value.toString());
159  if (this.parameterClass.isDefined(ps)) this.setValueAt(this.parameterClass.getString(ps), row,
160  COL_T_NAME);
161  if (!DBImportController.DB_IDENTIFIABLE_TABLES.contains(ps)) this.setValueAt("-----", row, COL_ID);
162  else this.setValueAt("", row, COL_ID);
163  this.setValueAt("-----", row, COL_F_KEY);
164  } else if (column == COL_T_NAME && isCellEditable(row, column)) {
165  this.parameterClass.setString(ParamString.valueOf(this.getValueAt(row, COL_T_KEY).toString()),
166  value.toString());
167  } else if (column == COL_F_NAME && getValueAt(row, COL_ID).equals("")) {
168  String s = value.toString();
169  setValueAt(s.substring(s.lastIndexOf('/') + 1, s.lastIndexOf('.')), row, COL_ID);
170  }
171  }
172  if (row == this.getRowCount() - 1 && column == COL_ACC && Boolean.TRUE.equals(value)) {
173  this.addEditableRow();
174  }
175  }
