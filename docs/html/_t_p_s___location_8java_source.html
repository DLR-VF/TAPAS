263  public void calculateWeight() {
264  double val;
265  if (this.capacity == 0) {
266  val = 0;
267  } else if (this.occupancy <= 0) {
268  val = this.capacity;
269  this.occupancy = 0;
270  } else if (fixCapacity) {
271  val = this.capacity - this.occupancy;
272  } else {
273  if (this.occupancy < 10.0 * this.capacity) {
274  val = this.capacity * TPS_FastMath.exp(
275  -this.parameterClass.getDoubleValue(ParamValue.WEIGHT_OCCUPANCY) * this.occupancy /
276  this.capacity);
277  } else {
278  val = 0;
279  }
280  }
281  this.weight = val;
282  }
