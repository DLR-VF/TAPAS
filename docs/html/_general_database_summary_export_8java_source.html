46  public GeneralDatabaseSummaryExport(Analyzer analyzer, String source) throws IOException, ClassNotFoundException {
47  TPS_ParameterClass parameterClass = new TPS_ParameterClass();
48  parameterClass.loadRuntimeParameters(TPS_BasicConnectionClass.getRuntimeFile());
49  dbCon = new TPS_DB_Connector(parameterClass);
50  this.analyzer = analyzer;
51  this.source = source;
52 
53  Categories[] cats = analyzer.getCategories();
54 
55  // check if analyzer is valid for this
56  if (cats.length != 4) {
57  throw new IllegalArgumentException("Not the right number of Analyzers");
58  }
59  if (cats[0] != Categories.Mode) {
60  throw new IllegalArgumentException("Analyzer[0] is not Mode");
61  }
62  if (cats[1] != Categories.PersonGroup) {
63  throw new IllegalArgumentException("Analyzer[1] is not PersonGroup");
64  }
65  if (cats[2] != Categories.TripIntention) {
66  throw new IllegalArgumentException("Analyzer[2] is not TripIntention");
67  }
68  if (cats[3] != Categories.DistanceCategoryDefault) {
69  throw new IllegalArgumentException("Analyzer[3] is not DistanceCategoryDefault");
70  }
71 
72  if (!analyzer.hasPersonGroupStatistics()) {
73  throw new IllegalArgumentException("Analyzer has no PersonGroup statistics");
74  }
75 
76  try {
77  PreparedStatement ps = dbCon.getConnection(this).prepareStatement(
78  "SELECT * FROM calibration_results WHERE sim_key = ?");
79  ps.setString(1, source);
80  ResultSet rs = ps.executeQuery();
81  if (rs.next()) {
82  System.err.println("Simkey for export exists and will be overwritten.");
83  update = true;
84  }
85  rs.close();
86  } catch (SQLException e) {
87  // should never happen
88  e.printStackTrace();
89  }
90  }

183  "distance_category = ? AND " + "trip_mode = ? AND trip_intention = ?");
184 
185  } else {
186  updateStatement = con.prepareStatement(
187  "INSERT INTO calibration_results " + "(cnt_persons, cnt_trips, avg_dist, avg_time, sim_key, " +
188  "person_group, distance_category, trip_mode, trip_intention) " +
189  "VALUES (?,?,?,?,?,?,?,?,?)");
190  }
191
