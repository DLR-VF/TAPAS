42  public void addTrip(TapasTrip tt) {
43  if (uniquePersons.add(tt.getIdPers())) {
44  PersonGroup pg = personGroupAnalyzer.assignTrip(tt);
45  cntPersons.put(pg, cntPersons.get(pg) + 1);
