252  public void setValues(String[] values) {
253  if (keyMapping == null) throw new IllegalArgumentException("KeyMapping must be generated first.");
254 
255  for (int i = 0; i < keyMapping.length; i++) {
256  String s = keyMapping[i];
257  header h = header.fromString(s);
258  if (h == null) {
259  continue; // unknown header ignored
260  }
261 
262  switch (h) {
263  case TAZ_ID_START:
264  tazIdStart = Integer.parseInt(values[i]);
265  break;
266  case TAZ_ID_END:
267  tazIdEnd = Integer.parseInt(values[i]);
268  break;
269  case LOC_ID_START:
270  locIdStart = Integer.parseInt(values[i]);
271  break;
272  case LOC_ID_END:
273  locIdEnd = Integer.parseInt(values[i]);
274  break;
275  case BBR_TYPE_START:
276  bbrTypeStart = Integer.parseInt(values[i]);
277  break;
278  case BBR_TYPE_HOME:
279  bbrTypeHome = Integer.parseInt(values[i]);
280  break;
281  case DIST:
282  distBL = Double.parseDouble(values[i]);
283  break;
284  case DIST_NET:
285  dist = Double.parseDouble(values[i]);
286  break;
287  case MODE_ID:
288  mode = Integer.parseInt(values[i]);
289  break;
290  case PERSON_ID:
291  pId = Integer.parseInt(values[i]);
292  break;
293  case HH_ID:
294  hhId = Integer.parseInt(values[i]);
295  break;
296  case JOB_ID:
297  job = Integer.parseInt(values[i]);
298  break;
299  case START_TIME:
300  startTime = Integer.parseInt(values[i]);
301  break;
302  case TRAVEL_TIME:
303  travelTime = Double.parseDouble(values[i]);
304  break;
305  case HOME:
306  isBackHome = Boolean.parseBoolean(values[i]);
307  break;
308  case ACTIVITY_CODE:
309  activityCode = Integer.parseInt(values[i]);
310  break;
311  case ACTIVITY_START:
312  activityCode = Integer.parseInt(values[i]);
313  break;
314  case ACTIVITY_DURATION:
315  activityCode = Integer.parseInt(values[i]);

317  }
