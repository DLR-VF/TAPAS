966  private TPS_Location select(double weightSum, List<TPS_Location> from) {
967  /*
968  * A microscopic sum of weights causes strange floating point issues,
969  * which cannot be caught by the 2nd loop: Namely hundreds of
970  * tiny weights sum up to a number barely above zero. But each
971  * weight is interpreted as zero by itself! -> Crash!
972  */
973 
974  if (weightSum < 1e-15 && from.size() > 0) return from.get(0);
975 
976  double rPos = Randomizer.random() * (double) (int) weightSum;
977  // use binary search? -> no!
978  for (TPS_Location loc : from) {
979  double locWeight = loc.getData().getWeight();
980  if (rPos < locWeight) {
981  return loc;
982  }
983  rPos -= locWeight;
984  }
985  // ok, catching potentially occuring floating point issues
986  for (TPS_Location loc : from) {
987  double locWeight = loc.getData().getWeight();
988  if (locWeight != 0) {
989  return loc;
990  }
991  }
992  //last resort
993  if (from.size() > 0) return from.get(0);
994  else return null;
995  //throw new RuntimeException("Ran over available weights; Input was: weightSum=" + weightSum);
996  }

1004  public ArrayList<TPS_Location> selectActivityLocations(int number) {
1005  ArrayList<TPS_Location> ret = new ArrayList<>();
1006  if (this.locations.size() <= number) {
1007  ret.addAll(locations);
1008  } else {
1009  double cCapacitySum = freeWeightSum;
1010  Vector<TPS_Location> cLocations = new Vector<>(locations);
1011  for (int i = 0; i < number; ++i) {
1012  TPS_Location selected = select(cCapacitySum, cLocations);
1013  if (selected == null) {
1014  break;
1015  }
1016  ret.add(selected);
1017  cCapacitySum -= selected.getData().getWeight();
1018  cLocations.remove(selected);
1019  }
1020  }
1021  return ret;
1022  }
