386  public double primaryDriver() {
387  if (driverScore < -0.1) {
388  int members = 0;
389  //int numberOfAdhults=0;
390  int numberOfChilds = 0;
391  int numberOfWorkingMembers = 0;
392  int numberOfMembersWithDrivingLicense = 0;
393  int ageYoungestAdhult = 1000; //I hope no one becomes older, ever!
394  int ageYoungestChild = 1000;
395  int numberOfCars = this.getHousehold().getCarNumber();
396  //shortcut:
397  if (numberOfCars == 0 || !this.hasDrivingLicense()) {
398  driverScore = 0;
399  return 0;
400  }
401 
402  driverScore = -25.6959310091439; //constant
403  driverScore += this.age * -0.104309375757471;
404  driverScore += Math.log(this.age) * 3.53025970082386;
405  driverScore += this.sex == TPS_Sex.MALE ? -1.34768311274 : 0;
406  driverScore += this.sex == TPS_Sex.MALE ? this.age * 0.0449898677627131 : 0;
407  driverScore += this.working > 0 ? 1.02298146009324 : 0;
408  driverScore += this.hasDrivingLicense() ? 10.9458075841947 : 0;
409 
410  //sum up all statistics above
411  for (TPS_Person p : this.getHousehold().getMembers(TPS_Household.Sorting.NONE)) {
412  members++;
413  if (p.working > 0) numberOfWorkingMembers++;
414  if (p.hasDrivingLicense()) numberOfMembersWithDrivingLicense++;
415  if (p.age >= 18) {
416  //numberOfAdhults++;
417  ageYoungestAdhult = Math.min(ageYoungestAdhult, p.age);
418  } else {
419  numberOfChilds++;
420  ageYoungestChild = Math.min(ageYoungestChild, p.age);
421  }
422  }
423 
424  if (numberOfChilds == 0) { //keine Kinder
425  if (members == 1) { //ein Erwachsener
426  if (ageYoungestAdhult < 30) {
427  driverScore += 0.355021416722072;
428  } else if (ageYoungestAdhult < 60) {
429  driverScore += 0.171869345495256;
430  } else {
431  driverScore += 1.22484387968388;
432  }
433  } else if (members == 2) { //zwei erwachsene
434  if (ageYoungestAdhult < 30) {
435  driverScore += -0.225129329476169;
436  } else if (ageYoungestAdhult < 60) {
437  driverScore += -0.526338766819736;
438  } else {
439  driverScore += 0.00349663057568653;
440  }
441  } else { // mehr als 3
442  driverScore += -0.383355892620622;
443  }
444  } else { //Haushalte mit Kindern
445  if (ageYoungestChild < 6) {
446  driverScore += -0.422373712001074;
447  } else if (ageYoungestChild < 14) {
448  driverScore += -0.457447530364251;
449  } else {
450  driverScore += -0.362948716707795;
451  }
452  }
453 
454  driverScore += numberOfWorkingMembers * -0.358492255297056;
455  driverScore += numberOfMembersWithDrivingLicense * 1.20018551149737;
456  driverScore += numberOfCars * -1.69765399821086;
457  if (numberOfMembersWithDrivingLicense > 0) //avoid div by zero! //TODO is this Integer division correct?
458  driverScore += (numberOfCars / numberOfMembersWithDrivingLicense) * 10.6624425307085;
459 
460 
461  driverScore = 1.0 / (1.0 + Math.exp(-driverScore));
462 
463  //finally cap it (for safety)!
464  driverScore = Math.max(0.0, Math.min(1.0, driverScore)); //cap to 0-1
465  }
466 
467 
468  return driverScore;
469  }

471  private ShoppingPreferenceAccessibility selectPreferenceAccessibility(ShoppingClass s) {
472  ShoppingPreferenceAccessibility accessibilityPreference = ShoppingPreferenceAccessibility.Sonstige; //default value!
473  double[] utilities = new double[ShoppingPreferenceAccessibility.valueArray.length];
474  int index = 0;
475  double sum = 0;
476  boolean oneIsSet = false;
477  if (s.equals(ShoppingClass.UEL)) {
478  index = 0;
479  }
480  Map<ShoppingPreferenceAccessibility, PreferenceSet> accessForShoppingClass = preferenceValues.preferencesAccess
481  .get(s);
482  PreferenceSet prefSet;
483  //calc utilities with a new random uncertantiy (normal distributed) for every choice
484  for (ShoppingPreferenceAccessibility e : ShoppingPreferenceAccessibility.valueArray) {
485  prefSet = accessForShoppingClass.get(e);
486  if (prefSet.isUsed) {
487  utilities[index] = TPS_FastMath.exp(prefSet.utility + //utility
488  Randomizer.randomGaussianDistribution(() -> Randomizer.randomGaussian(), 0, prefSet.sigma)
489  //uncertainty
490  + this.errorTerm); //errorterm
491  sum += utilities[index];
492  oneIsSet = true;
493  } else {
494  utilities[index] = 0;
495  }
496  index++;
497  }
498 
499  if (!oneIsSet) //nothing to estimate!
500  return accessibilityPreference; //return the default
501 
502  //normalize to get probabilities
503  for (index = 0; index < utilities.length; ++index) {
504  utilities[index] /= sum;
505  }
506 
507 
508  //draw the choice
509  double rand = Randomizer.random();
510  sum = 0;
511  index = 0;
512  for (ShoppingPreferenceAccessibility e : ShoppingPreferenceAccessibility.valueArray) {
513  prefSet = accessForShoppingClass.get(e);
514  if (prefSet.isUsed) {
515  sum += utilities[index];
516  index++;
517  if (sum >= rand) {
518  accessibilityPreference = e;
519  break; //finish!
520  }
521  } else {//skip this index
522  index++;
523  }
524  }
525  return accessibilityPreference;
526  }
