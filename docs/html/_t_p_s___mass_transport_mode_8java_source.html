123  public double getTravelTime(Locatable start, Locatable end, int time, SimulationType simType, TPS_ActivityConstant actCodeFrom, TPS_ActivityConstant actCodeTo, TPS_Person person, TPS_Car car) {
124 
125  double tt = -1.0; // this is the indicator, of an invalid tt -time
126  int TVZfrom = start.getTrafficAnalysisZone().getTAZId();
127  int TVZto = end.getTrafficAnalysisZone().getTAZId();
128 
129  double beelineDistanceLoc = TPS_Geometrics.getDistance(start, end,
130  this.getParameters().getDoubleValue(ParamValue.MIN_DIST));
131  double beelineDistanceTAZ;
132 
133  if (TVZfrom != TVZto) {
134  beelineDistanceTAZ = this.getParameters().paramMatrixClass.getValue(ParamMatrix.DISTANCES_BL, TVZfrom,
135  TVZto);
136  } else {
137  beelineDistanceTAZ = beelineDistanceLoc;
138  }
139 
140  double beelineFaktor = beelineDistanceLoc / beelineDistanceTAZ;
141 
142  if (!start.getTrafficAnalysisZone().equals(end.getTrafficAnalysisZone())) {
143  // start and destination are not within the same traffic zone
144  double scoreFrom = 0;
145  double scoreTo = 0;
146 
147 
148  double ttBuf = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.TRAVEL_TIME_PT, TVZfrom,
149  TVZto, simType, time);
150 
151 
152  // if the pt matrix has no valid entry for the relation and the existence of a school bus should be
153  // simulated
154  if (TPS_Mode.noConnection(ttBuf)) {
155  ttBuf = TPS_Mode.NO_CONNECTION;
156  if (TPS_PersonType.PUPIL.equals(person.getPersGroup().getPersType()) && this.getParameters().isTrue(
157  ParamFlag.FLAG_USE_SCHOOLBUS) && (actCodeFrom.hasAttribute(
158  TPS_ActivityConstantAttribute.SCHOOL) || actCodeTo.hasAttribute(
159  TPS_ActivityConstantAttribute.SCHOOL))) {
160  // getting average speed per combination of bbr-codes
161  int fromBBR = start.getTrafficAnalysisZone().getBbrType().getCode(TPS_SettlementSystemType.FORDCP);
162  int toBBR = end.getTrafficAnalysisZone().getBbrType().getCode(TPS_SettlementSystemType.FORDCP);
163  double schoolBusSpeed = this.getParameters().paramMatrixMapClass.getValue(
164  ParamMatrixMap.AVERAGE_SPEED_SCHOOLBUS, fromBBR, toBBR, simType, time);
165 
166  // Assumtion: traveltime plus time for access and egress and waiting
167  // beta * beeline / schulBusSpeed + Zugang + Abgang + Wartezeit
168  tt = beelineFaktor * beelineDistanceLoc / schoolBusSpeed + this.getParameters().getDoubleValue(
169  ParamValue.DEFAULT_SCHOOL_BUS_ACCESS) + this.getParameters().getDoubleValue(
170  ParamValue.DEFAULT_SCHOOL_BUS_EGRESS) + this.getParameters().getDoubleValue(
171  ParamValue.DEFAULT_SCHOOL_BUS_WAIT);
172  if (travelTimeIsInvalid(tt)) {
173  TPS_Logger.log(SeverenceLogLevel.DEBUG,
174  "Invalid travel time detected: " + tt + " from " + TVZfrom + " to " + TVZto +
175  " mode: " + getName());
176  tt = TPS_Mode.NO_CONNECTION;
177  }
178  } else {
179  //no connection via PT!
180  tt = TPS_Mode.NO_CONNECTION;
181  // tt= beelineDistanceLoc * ModeType.WALK.getBeelineFactor().getDoubleValue() /
182  // get(ModeType.WALK).getVelocity().getDoubleValue();
183  // if(travelTimeIsInvalid(tt))
184  // TPS_Logger.log(SeverenceLogLevel.FATAL, "NaN detected");
185  }
186 
187  } else {
188  tt = beelineFaktor * ttBuf;
189  // access time
190  double TT1 = 0;
191  if (this.getParameters().isDefined(ParamMatrixMap.ARRIVAL_PT))
192  TT1 = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.ARRIVAL_PT, TVZfrom, TVZto,
193  simType, time);
194  // egress time
195  double TT2 = 0;
196  if (this.getParameters().isDefined(ParamMatrixMap.EGRESS_PT))
197  TT2 = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.EGRESS_PT, TVZfrom, TVZto,
198  simType, time);
199 
200  if (this.getParameters().isTrue(ParamFlag.FLAG_USE_BLOCK_LEVEL)) {
201  scoreFrom = this.getScore(start.getTrafficAnalysisZone()) - this.getScore(start.getBlock());
202  scoreTo = this.getScore(end.getTrafficAnalysisZone()) - this.getScore(end.getBlock());
203  //travel time
204  tt *= 1.0 + ((scoreFrom + scoreTo) * this.getParameters().getDoubleValue(ParamValue.PT_TT_FACTOR));
205  // access time
206  TT1 *= 1.0 + (scoreFrom * this.getParameters().getDoubleValue(ParamValue.PT_ACCESS_FACTOR));
207  // egress time
208  TT2 *= 1.0 + (scoreTo * this.getParameters().getDoubleValue(ParamValue.PT_EGRESS_FACTOR));
209  }
210 
211  tt += TT1 + TT2;
212  if (travelTimeIsInvalid(tt)) {
213  TPS_Logger.log(SeverenceLogLevel.DEBUG,
214  "Invalid travel time detected: " + tt + " from " + TVZfrom + " to " + TVZto + " mode: " +
215  getName());
216  tt = TPS_Mode.NO_CONNECTION;
217  }
218  }
219  } else {
220  if (!start.getTrafficAnalysisZone().getSimulationTypeValues(simType).isIntraPTTrafficAllowed()) {
221  //no intra traffic allowed!
222  return TPS_Mode.NO_CONNECTION;
223  }
224 
225  // start and destination within the same zone
226  if (this.getParameters().isTrue(ParamFlag.FLAG_USE_BLOCK_LEVEL)) {
227  if ((start.hasBlock() && start.getBlock().equals(end.getBlock())) ||
228  (!start.hasBlock() && !end.hasBlock())) {
229  // start and destination within the same block or no block information: No valid tt-time!
230  // tt = this.getParameters().getDoubleValue(ParamValue.PT_MINIMUM_TT);
231  } else {
232  // start and destination not within the same block
233  if (!this.getParameters().isTrue(ParamFlag.FLAG_INTRA_INFOS_MATRIX)) {
234  ScenarioTypeValues stv = start.getTrafficAnalysisZone().getSimulationTypeValues(simType);
235  double bla0 = stv.getAverageSpeedPT();
236  if (Double.isNaN(bla0))//safety fix
237  bla0 = 8;
238  tt = beelineFaktor * beelineDistanceLoc / bla0;
239  } else {
240  double bla1 = this.getParameters().getDoubleValue(get(ModeType.PT).getVelocity());
241  if (Double.isNaN(bla1))//safety fix
242  bla1 = 8;
243  tt = beelineFaktor * beelineDistanceLoc / bla1;
244  }
245 
246  // access distance to pt
247  double distFrom = start.hasBlock() ? start.getBlock().getNearestPubTransStop() : this
248  .getParameters().getDoubleValue(ParamValue.AVERAGE_DISTANCE_PT_STOP);
249 
250  // egress distance to pt
251  double distTo = end.hasBlock() ? end.getBlock().getNearestPubTransStop() : this.getParameters()
252  .getDoubleValue(
253  ParamValue.AVERAGE_DISTANCE_PT_STOP);
254 
255  // !!!dk
256  if (distTo < 0) {
257  distTo = 0;
258  }
259  double ptSpeed = this.getParameters().getDoubleValue(get(ModeType.WALK).getVelocity());
260  if (ptSpeed == 0 || Double.isNaN(ptSpeed)) {
261  ptSpeed = 1.0;
262  }
263  double beelineFactor = this.getParameters().getDoubleValue(ModeType.WALK.getBeelineFactor());
264  if (Double.isNaN(beelineFactor) || beelineFactor == 0) {
265  beelineFactor = 1.4;
266  }
267  // !!!dk
268  tt += (distFrom + distTo) * beelineFactor / ptSpeed;
269  if (travelTimeIsInvalid(tt)) {
270  TPS_Logger.log(SeverenceLogLevel.DEBUG,
271  "Invalid travel time detected: " + tt + " from " + TVZfrom + " to " + TVZto +
272  " mode: " + getName());
273  tt = TPS_Mode.NO_CONNECTION;
274  }
275  }
276  } else {
277  // no block level
278  if (this.getParameters().isTrue(ParamFlag.FLAG_INTRA_INFOS_MATRIX)) {
279  // information about situation within traffic analysis zones are in the travel time matrices
280  double ttBuf = 0;
281  ttBuf = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.TRAVEL_TIME_PT, TVZfrom,
282  TVZto, simType, time);
283 
284  if (TPS_Mode.noConnection(ttBuf)) {
285  // no connection via PT!
286  return TPS_Mode.NO_CONNECTION;
287  }
288 
289  // now only valid times! zero means walk, because there is no pt
290  // intra-cell trafic!
291 
292  tt = beelineFaktor * ttBuf;
293  // access time
294  double TT1 = 0;
295  if (this.getParameters().isDefined(ParamString.DB_NAME_MATRIX_ACCESS_PT) && simType.equals(
296  SimulationType.SCENARIO) || this.getParameters().isDefined(
297  ParamString.DB_NAME_MATRIX_ACCESS_PT_BASE) && simType.equals(SimulationType.BASE))
298  TT1 = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.ARRIVAL_PT, TVZfrom,
299  TVZto, simType, time);
300  // egress time
301  double TT2 = 0;
302  if (this.getParameters().isDefined(ParamString.DB_NAME_MATRIX_EGRESS_PT) && simType.equals(
303  SimulationType.SCENARIO) || this.getParameters().isDefined(
304  ParamString.DB_NAME_MATRIX_EGRESS_PT_BASE) && simType.equals(SimulationType.BASE))
305  TT2 = this.getParameters().paramMatrixMapClass.getValue(ParamMatrixMap.EGRESS_PT, TVZfrom,
306  TVZto, simType, time);
307 
308  tt += TT1 + TT2;
309  if (travelTimeIsInvalid(tt)) {
310  TPS_Logger.log(SeverenceLogLevel.DEBUG,
311  "Invalid travel time detected: " + tt + " from " + TVZfrom + " to " + TVZto +
312  " mode: " + getName());
313  tt = TPS_Mode.NO_CONNECTION;
314  }
315 
316  } else {
317  // no infos to situation within traffic analysis zones; using standard minimum time
318  // tt = this.getParameters().getDoubleValue(ParamValue.PT_MINIMUM_TT);
319  // Otherwise the travel time is calculated by the beeline distance and a factor and the average speed inside
320  // this traffic analysis zone.
321  double factor = start.getTrafficAnalysisZone().getSimulationTypeValues(simType)
322  .getBeelineFactorMIT();
323  if (Double.isNaN(factor) || factor == 0)//safety fix
324  factor = 1.4;
325  double avSpeed = start.getTrafficAnalysisZone().getSimulationTypeValues(simType)
326  .getAverageSpeedPT();
327  if (Double.isNaN(avSpeed) || avSpeed == 0)//safety fix
328  avSpeed = 8;
329  tt = ((beelineDistanceLoc * factor) / avSpeed);
330  //tt = -1;
331  }
332  }
333  }
334  return (tt);
335  }
