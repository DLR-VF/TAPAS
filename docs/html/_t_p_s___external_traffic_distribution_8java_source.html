217  public void distributeOverDay() {
218  //go through the OD-List
219  for (ODPair p : this.ODPairs.values()) {
220 
221  //check if the origin is outside berlin -> distribute according to start list
222  if (p.inBound) {
223  for (Entry<Integer, Double> e : this.dayDistributionStart.entrySet()) {
224  ODPair pair = new ODPair();
225  pair.idOrigin = p.idOrigin;
226  pair.idDestination = p.idDestination;
227  pair.inBound = p.inBound;
228  pair.internalTraffic = p.internalTraffic;
229  pair.hourOfDay = e.getKey();
230  pair.volume = p.volume * e.getValue();
231  this.ODPairsDayDistributed.add(pair);
232  }
233  } else if (p.internalTraffic) {
234  for (Entry<Integer, Double> e : this.dayDistributionIntern.entrySet()) {
235  ODPair pair = new ODPair();
236  pair.idOrigin = p.idOrigin;
237  pair.idDestination = p.idDestination;
238  pair.inBound = p.inBound;
239  pair.internalTraffic = p.internalTraffic;
240  pair.hourOfDay = e.getKey();
241  pair.volume = p.volume * e.getValue();
242  this.ODPairsDayDistributed.add(pair);
243  }
244  } else {
245  //check if the origin is inside berlin -> distribute according to return list
246  for (Entry<Integer, Double> e : this.dayDistributionReturn.entrySet()) {
247  ODPair pair = new ODPair();
248  pair.idOrigin = p.idOrigin;
249  pair.idDestination = p.idDestination;
250  pair.inBound = p.inBound;
251  pair.internalTraffic = p.internalTraffic;
252  pair.hourOfDay = e.getKey();
253  pair.volume = p.volume * e.getValue();
254  this.ODPairsDayDistributed.add(pair);
255  }
256  }
257  }

260  public void distributeOverLocations(double calib) {
261  double way = 0;
262  Map<Integer, Integer> timeDistCheck = new HashMap<>();
263  for (int i = 0; i < 24; ++i)
264  timeDistCheck.put(i, 0);
265  TAZ o;
266  Map<Integer, TAZ> dSet, dSet2;
267  for (ODPair p : this.ODPairsDayDistributed) {
268  if (!p.internalTraffic) {
269  //first calculate the weight of the locations
270  //it doesnt matter if o and d are exchanged: "gravity" is symmetric
271  if (p.inBound) {
272  o = this.externalOrigins.get(p.idOrigin);
273  dSet = this.berlinDModel.get(p.idDestination);
274  } else {
275  o = this.externalOrigins.get(p.idDestination);
276  dSet = this.berlinDModel.get(p.idOrigin);
277  }
278  double sum = 0;
279  for (TAZ d : dSet.values()) {
280  if (o.useGravity) d.tempWeight = d.cappa * Math.pow(TPS_Geometrics
281  .getDistance(d.coordDestination.getValue(0), d.coordDestination.getValue(1),
282  o.coordSource.getValue(0), o.coordSource.getValue(1)), -calib);
283  else d.tempWeight = d.cappa;
284  sum += d.tempWeight;
285  }
286  //normalize
287  sum = 1.0 / sum;
288  for (TAZ d : dSet.values()) {
289  d.tempWeight *= sum;
290  }
291 
292  //now distribute the volume
293  int totalVolume = 0;
294  for (TAZ d : dSet.values()) {
295 
296  way += d.tempWeight * p.volume;
297  if (way >= 1.0) {
298  //we have some trips!
299  ODPair distributedPair = new ODPair();
300  distributedPair.inBound = p.inBound;
301  distributedPair.hourOfDay = p.hourOfDay;
302  if (p.inBound) {
303  distributedPair.idOrigin = p.idOrigin;
304  distributedPair.idDestination = d.id;
305  } else {
306  distributedPair.idOrigin = d.id;
307  distributedPair.idDestination = p.idDestination;
308 
309  }
310 
311  distributedPair.volume = (int) way; // this is the integer-part
312  totalVolume += distributedPair.volume;
313  way -= distributedPair.volume; // this is the remainer
314  this.ODPairsGravityDistributed.add(distributedPair);
315  }
316  }
317  timeDistCheck.put(p.hourOfDay, timeDistCheck.get(p.hourOfDay) + totalVolume);
318  //System.out.println("Pair o:"+p.idOrigin+" d:"+p.idDestination+" h:"+p.hourOfDay+" v:"+p.volume+" distributed:"+totalVolume);
319  } else {
320  //first calculate the weight of the locations
321  //just the capacity for internal traffics!
322  dSet = this.berlinDModel.get(p.idOrigin);
323  dSet2 = this.berlinDModel.get(p.idDestination);
324  double sum = 0;
325  for (TAZ d : dSet.values()) {
326  d.tempWeight = d.cappa;
327  sum += d.tempWeight;
328  }
329  //normalize
330  sum = 1.0 / sum;
331  for (TAZ d : dSet.values()) {
332  d.tempWeight *= sum;
333  }
334 
335  //now distribute the volume
336  int totalVolume = 0;
337  for (TAZ oSet : dSet.values()) {
338  if (way + oSet.tempWeight * p.volume > 1.0) {
339  for (TAZ d : dSet2.values()) {
340 
341  way += d.tempWeight * oSet.tempWeight * p.volume;
342  if (way >= 1.0 && oSet.id != d.id) {
343  //we have some trips!
344  ODPair distributedPair = new ODPair();
345  distributedPair.inBound = p.inBound;
346  distributedPair.internalTraffic = p.internalTraffic;
347  distributedPair.hourOfDay = p.hourOfDay;
348  distributedPair.idOrigin = oSet.id;
349  distributedPair.idDestination = d.id;
350  distributedPair.volume = (int) way; // this is the integer-part
351  totalVolume += distributedPair.volume;
352  way -= distributedPair.volume; // this is the remainer
353  this.ODPairsGravityDistributed.add(distributedPair);
354  }
355  }
356  } else {
357  way += oSet.tempWeight * p.volume;
358  }
359  timeDistCheck.put(p.hourOfDay, timeDistCheck.get(p.hourOfDay) + totalVolume);
360  //System.out.println("Pair o:"+p.idOrigin+" d:"+p.idDestination+" h:"+p.hourOfDay+" v:"+p.volume+" distributed:"+totalVolume);
361  }
362  }
363  }

365 // System.out.println("Hour "+i+" volume: "+timeDistCheck.get(i));
366 // }

423  public void loadCordonDistricts(String tableName, String dmod, int berlinId) {
424  String query = "";
425  try {
426  int id, taz;
427  double sLon, sLat, dLon, dLat;
428 
429  //load the external mapping
430  query = "SELECT z.tapas_taz_id,z.vbz_no, vbz_6561, z.nuts_id ISNULL as isberlin, " +
431  "st_X(st_transform(st_centroid(z.the_geom),4326)) as lon, " +
432  "st_Y(st_transform(st_centroid(z.the_geom),4326)) as lat " + " from " + tableName + " as z join " +
433  dmod + " as d on z.vbz_no=d.ver_nr";
434  ResultSet rs = this.dbCon.executeQuery(query, this);
435  while (rs.next()) {
436  if (!rs.getBoolean("isberlin")) {
437  id = (int) (rs.getDouble("vbz_6561") + 0.1);
438  sLat = rs.getDouble("lat");
439  sLon = rs.getDouble("lon");
440  dLat = rs.getDouble("lat");
441  dLon = rs.getDouble("lon");
442  taz = rs.getInt("tapas_taz_id");
443  dModellZonen.put(id, taz);
444 
445  //create point informations
446  TAZ entry = new TAZ();
447  //now convert this id to a zone of the above mapping
448  entry.id = taz;
449  entry.taz_id = taz;
450  entry.coordSource.setValues(sLon, sLat);
451  entry.coordDestination.setValues(dLon, dLat);
452  entry.cappa = 0;
453  this.externalOrigins.put(entry.id, entry);
454  this.internalDestinations.put(entry.id, entry);
455  }
456  }
457  rs.close();
458 
459 
460  System.out.println("found " + this.externalOrigins.size() + " origin entires");
461  berlinZone = berlinId;
462 
463 
464  } catch (SQLException e) {
465  System.err.println("Error in sqlstatement: " + query);
466  e.printStackTrace();
467  }
468  }
