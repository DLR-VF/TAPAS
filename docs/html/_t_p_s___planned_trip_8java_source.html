116  boolean setTravelTime(TPS_LocatedStay pComingFrom, TPS_LocatedStay pGoingTo) {
117  // Do not get confused with 'pComingFrom' and 'myPComingFrom'. 'pComingFrom' is needed for the hierarchy of the
118  // trips in the tour.
119  if (pComingFrom.getStay().isAtHome() && pGoingTo.getStay().isAtHome() && mode == null) {
120  // This shouldn't be a big problem. Skip the message; Use mode 'walk' as this are most of the time walks with dogs.
121  mode = TPS_ExtMode.simpleWalk;
122  }
123 
124  if (!pComingFrom.isLocated()) {
125  throw new RuntimeException("tpsTrip.setTravelTime: missing location of the preceding stay: " +
126  pComingFrom.getStay().getActCode());
127  }
128 
129  if (!pGoingTo.isLocated()) {
130  throw new RuntimeException("tpsTrip.setTravelTime: missing location of the following stay: " +
131  pGoingTo.getStay().getActCode());
132  }
133 
134  TPS_Location pLocComingFrom = pComingFrom.getLocation();
135  TPS_Location pLocGoingTo = pGoingTo.getLocation();
136  this.setDistanceBeeline(TPS_Geometrics
137  .getDistance(pLocComingFrom, pLocGoingTo, plan.getParameters().getDoubleValue(ParamValue.MIN_DIST)));
138 
139  TPS_TrafficAnalysisZone goingToTVZ = pLocGoingTo.getTrafficAnalysisZone();
140  TPS_TrafficAnalysisZone comingFromTVZ = pLocComingFrom.getTrafficAnalysisZone();
141  double tt = -1;
142 
143  TPS_Car carForTrip = null;
144  //pick cars used for this plan
145  for (TPS_SchemePart schemePart : this.plan.getScheme()) {
146  if (!schemePart.isHomePart()) {
147  TPS_TourPart tourpart = (TPS_TourPart) schemePart;
148  if (tourpart.getCar() != null) {
149  carForTrip = tourpart.getCar();
150  break;
151  }
152  }
153  }
154 
155 
156  /*
157  * Beachte: pModeComingFrom = mode, vom Stay pComingFrom wegzukommen (getPModeGo) pModeGoingTo = mode, zum Stay
158  * pGoingTo hinzukommen (getPModeCome)
159  */
160  TPS_ExtMode pModeComingFrom = pComingFrom.getModeDep();
161  TPS_ExtMode pModeGoingTo = pGoingTo.getModeArr();
162 
163  // If the person comes from home or is going home, then pModeGoingTo or pModeComingFrom are 0. Usually at least
164  // one
165  // of them should be set. Note that 'pComingFrom' and 'pGoingTo' have a different meaning here compared to
166  // 'Stay'.
167  // Here they point to the adjacent episodes.
168 
169  // If different modes are used for leave and to get to the locations of these episodes then the priority of the
170  // episodes determines the mode choice.
171 
172  if (pComingFrom.getStay().getPriority() < pGoingTo.getStay().getPriority()) {
173  if (pComingFrom.getStay().isAtHome()) {
174  setMode(pModeGoingTo);
175  } else {
176  setMode(pModeComingFrom);
177  }
178  } else {
179  if (pGoingTo.getStay().isAtHome()) {
180  setMode(pModeComingFrom);
181  } else {
182  setMode(pModeGoingTo);
183  }
184  }
185 
186  this.setDistanceEmptyNet(
187  TPS_Mode.get(ModeType.WALK).getDistance(pLocComingFrom, pLocGoingTo, SimulationType.SCENARIO, null));
188 
189  /*
190  * Da anders als in der Vorlage die Klassen für die verschiedenen Modes nicht in einer Datei definiert werden
191  * können, bin ich auch bei der Vererbung und dem Zugriff auf die richtige Funktion skeptisch. Deshalb wird hier
192  * explizit nach dem Mode unterschieden und per cast die entsprechende Routine genutzt.
193  *
194  *
195  * TYPECAST UNNÖTIG
196  *
197  *
198  * Für die KFZ-modes MIV, Taxi und Passagier werden die entspr. Funktionen getDistance aufgerufen. Da
199  * distanceBeeline immer einen minimalen Wert hat, kann es in diesen Funktionen nicht zu Komplikationen kommen.
200  *
201  * Den getTravelTime-Routinen werden ebenfalls die actCodes der beteiligten Episoden (comingFrom & goingTo) und
202  * die Job-ID der Person übergeben. Derzeit werden diese nur für die Modes PubTrans und Train zur Berechnung der
203  * Schulbusreisezeiten genutzt.
204  */
205 
206  TPS_Mode primaryMode = mode.primary;
207 
208  if (primaryMode.isType(ModeType.PT)) {
209  tt = primaryMode.getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(), SimulationType.SCENARIO,
210  pComingFrom.getStay().getActCode(), pGoingTo.getStay().getActCode(), plan.getPerson(), null);
211  if (pLocComingFrom.getTrafficAnalysisZone().equals(pLocGoingTo.getTrafficAnalysisZone())) {
212  if (TPS_Mode.noConnection(tt)) {
213  if (this.plan.getParameters().isFalse(ParamFlag.FLAG_USE_BLOCK_LEVEL)) {
214  tt = TPS_Mode.get(ModeType.WALK).getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(),
215  SimulationType.SCENARIO, pComingFrom.getStay().getActCode(),
216  pGoingTo.getStay().getActCode(), plan.getPerson(), null);
217  } else if (this.plan.getParameters().isTrue(ParamFlag.FLAG_USE_BLOCK_LEVEL)) {
218  if (pLocComingFrom.getBlock() == null || pLocGoingTo.getBlock() == null) {
219  tt = TPS_Mode.get(ModeType.WALK).getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(),
220  SimulationType.SCENARIO, pComingFrom.getStay().getActCode(),
221  pGoingTo.getStay().getActCode(), plan.getPerson(), null);
222  } else if (pLocComingFrom.getBlock().equals(pLocGoingTo.getBlock())) {
223  tt = TPS_Mode.get(ModeType.WALK).getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(),
224  SimulationType.SCENARIO, pComingFrom.getStay().getActCode(),
225  pGoingTo.getStay().getActCode(), plan.getPerson(), null);
226 
227  }
228  }
229  }
230  }
231  } else if (primaryMode.isType(ModeType.TRAIN) && this.plan.getParameters().isDefined(
232  ParamFlag.FLAG_USE_CARSHARING) && this.plan.getParameters().isTrue(
233  ParamFlag.FLAG_USE_CARSHARING)) { // carsharing faker!
234  tt = TPS_Mode.get(ModeType.MIT).getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(),
235  SimulationType.SCENARIO, pComingFrom.getStay().getActCode(), pGoingTo.getStay().getActCode(),
236  plan.getPerson(), null);
237  } else {
238  tt = primaryMode.getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(), SimulationType.SCENARIO,
239  pComingFrom.getStay().getActCode(), pGoingTo.getStay().getActCode(), plan.getPerson(), carForTrip);
240  }
241 
242  if (TPS_Mode.noConnection(tt)) { //you can always walk!
243  //TODO: really walk?!?
244  primaryMode = TPS_Mode.get(ModeType.WALK);
245  this.mode.primary = primaryMode;
246  tt = primaryMode.getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(), SimulationType.SCENARIO,
247  pComingFrom.getStay().getActCode(), pGoingTo.getStay().getActCode(), plan.getPerson(), null);
248  }
249 
250  double dist = primaryMode.getDistance(pLocComingFrom, pLocGoingTo, SimulationType.SCENARIO, carForTrip);
251 
252  if (primaryMode.isType(ModeType.WALK) && dist / (Math.max(60, tt)) > 10) {
253  //runners!
254  tt = primaryMode.getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(), SimulationType.SCENARIO,
255  pComingFrom.getStay().getActCode(), pGoingTo.getStay().getActCode(), plan.getPerson(), null);
256  dist = primaryMode.getDistance(pLocComingFrom, pLocGoingTo, SimulationType.SCENARIO, carForTrip);
257 
258  }
259 
260  this.setDistance(primaryMode.getDistance(pLocComingFrom, pLocGoingTo, SimulationType.SCENARIO, carForTrip));
261  if (Double.isNaN(tt)) {
262  System.err.println("Arg!");
263  tt = primaryMode.getTravelTime(pLocComingFrom, pLocGoingTo, this.getStart(), SimulationType.SCENARIO,
264  pComingFrom.getStay().getActCode(), pGoingTo.getStay().getActCode(), plan.getPerson(), null);
265  }
266  this.setDuration((int) tt);
267 
268  /*
269  * Abschließend werden für jeden Trip in Abhängigkeit vom Modus die Kosten für den Trip berechnet und in der
270  * Variablen myFinancialCosts abgelegt; dabei erfolgt die Rückrechnung der Kosten in Euro
271  */
272 
273  double costVal = 0, costStay = 0;
274  if (!primaryMode.isType(ModeType.MIT)) {
275  costVal = primaryMode.getCost_per_km(SimulationType.SCENARIO);
276  } else {
277  // Bestimmen, ob durch Wegfall der Pendlerpauschale betroffen: erwerbstätig und HHEinkommen über 2600 Euro
278  double pendlerkosten = 0.0;
279 
280  // determine whether working and income over average such that commuting tax benefit is appropriate
281  if (this.plan.getPerson().isWorking() &&
282  plan.getPerson().getHousehold().getIncome() >= this.plan.getParameters().getIntValue(
283  ParamValue.MIT_INCOME_CLASS_COMMUTE)) {
284  pendlerkosten += this.plan.getParameters().getDoubleValue(ParamValue.MIT_FUEL_COST_PER_KM_COMMUTE);
285  }
286 
287  costVal = pendlerkosten;
288  if (carForTrip == null) {
289  // default distance related costs
290  costVal += this.plan.getParameters().getDoubleValue(ParamValue.MIT_GASOLINE_COST_PER_KM) +
291  this.plan.getParameters().getDoubleValue(ParamValue.MIT_VARIABLE_COST_PER_KM);
292  } else {
293  // car-depending distance related costs
294  costVal += carForTrip.getCostPerKilometer(SimulationType.SCENARIO) +
295  carForTrip.getVariableCostPerKilometer(SimulationType.SCENARIO);
296 
297  }
298 
299  // location related costs
300  // determine whether parking fee or toll is charged in the destination zone; toll only relevant when coming
301  // from an toll free zone
302  costStay = 0.0;
303 
304  // toll fee
305  if (goingToTVZ.hasToll(SimulationType.SCENARIO) && !comingFromTVZ.hasToll(SimulationType.SCENARIO)) {
306  // Scenario: toll has to be payed on entrance into a toll zone (cordon toll)
307  costStay += goingToTVZ.getTollFee(SimulationType.SCENARIO);
308  }
309 
310  // toll fees leaving a toll zone
311  if (this.plan.getParameters().isTrue(ParamFlag.FLAG_USE_EXIT_MAUT)) {
312  if (!goingToTVZ.hasToll(SimulationType.SCENARIO) && comingFromTVZ.hasToll(SimulationType.SCENARIO)) {
313  // Scenario: toll has to be payed leaving a toll zone (cordon toll)
314  costStay += comingFromTVZ.getTollFee(SimulationType.SCENARIO);
315  if (TPS_Logger.isLogging(SeverenceLogLevel.FINE)) {
316  TPS_Logger.log(SeverenceLogLevel.FINE,
317  "MIV - trip has maut: " + comingFromTVZ.getTollFee(SimulationType.SCENARIO) +
318  " costSum: " + costStay);
319  }
320  }
321  }
322 
323  // parking fee
324  if (goingToTVZ.hasParkingFee(SimulationType.SCENARIO)) {
325  double costSzen = goingToTVZ.getParkingFee(SimulationType.SCENARIO);
326  if (costSzen > 0) {
327  double stayingHours = this.getDuration() * 2.7777777777e-4;// converting into seconds
328  costStay += (costSzen * stayingHours);
329  }
330  if (TPS_Logger.isLogging(SeverenceLogLevel.FINE)) {
331  TPS_Logger.log(SeverenceLogLevel.FINE,
332  "GoingTo has parkingfee of" + goingToTVZ.getParkingFee(SimulationType.SCENARIO) +
333  "; duration: " + this.getDuration() + "; costSum: " + costSzen);
334  }
335  }
336  }
337 
338  costs = (this.getDistance() * 0.001 * costVal) + costStay;
339  if (TPS_Logger.isLogging(SeverenceLogLevel.FINE)) {
340  TPS_Logger.log(SeverenceLogLevel.FINE, this.mode.toString() + " - trip-costs: " + costs);
341  TPS_Logger.log(SeverenceLogLevel.FINE, "Set travel time " + this.toString());
342  }
343 
344  return (true);
345  }
