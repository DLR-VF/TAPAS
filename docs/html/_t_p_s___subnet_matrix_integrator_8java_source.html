121  public Matrix updateValues(){
122  Matrix result = this.outernet.clone();
123 
124  /*
125  * Define cordon points:
126  * For a given connection between A in the subnet and B in the outer net do:
127  * Search over all points in the sub net the cell C with a minimum value of: (AC+CB)
128  * Update the Value AB with the value of (AC+CB) between the original net and the subnet
129  * Update all connections in the original matrix, which lie within the subnet
130  */
131  int a,b,c, minTime, ab, ac,acOrig, cb, cIndex, minTimeOrig;
132  for( Map.Entry<Integer,Integer> inner : this.mappingSubnet.entrySet()) {
133  a = inner.getValue();
134  for (Integer outer : this.outerTAZ) {
135  b = outer; // just to keep the names straight
136  minTime = Integer.MAX_VALUE;
137  minTimeOrig = ab = (int) this.outernet.getValue(this.mappingIndices.get(a), b);
138  cIndex = -1;
139  for (Map.Entry<Integer, Integer> innerC : this.mappingSubnet.entrySet()) {
140  c = innerC.getValue();
141  if (a != c) { //only update values, which need a third point
142  ac = (int) this.subnet.getValue(a, c);
143  acOrig = (int) this.outernet.getValue(this.mappingIndices.get(a), this.mappingIndices.get(c));
144  cb = (int) this.outernet.getValue(this.mappingIndices.get(c), b);
145  if ((ac + cb) < minTime) {
146  cIndex = c;
147  minTime = (ac + cb);
148  minTimeOrig = acOrig +cb; // save the deviation because of the detour to compensate the detourerror
149  }
150  }
151  }
152  //set the difference
153  if (cIndex >= 0) {
154  result.setValue(a, b, minTime -(minTimeOrig-ab)); //compensate the detourerror
155  }
156  }
157  //now set the subnetcells itself
158  for (Map.Entry<Integer, Integer> innerB : this.mappingSubnet.entrySet()) {
159  b = innerB.getValue();
160  if (a != b) { //never touch the main diagonal!
161  result.setValue(a, b, this.subnet.getValue(a, b));
162  }
163  }
164  }
165  return result;
166  }
