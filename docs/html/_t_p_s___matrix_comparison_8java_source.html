59  public void analyze(int radix) {
60  //size check
61  if (matrix1.length != matrix2.length) return;
62  if (matrix1[0].length != matrix2[0].length) return;
63  Integer count, bin;
64  this.result.clear();
65  int maxVal, minVal, maxCount, minCount;
66  Map<Integer, Integer> oMax = new TreeMap<>();
67  Map<Integer, Integer> oMin = new TreeMap<>();
68  for (int i = 0; i < matrix1.length; ++i) {
69  maxVal = 0;
70  minVal = 0;
71  maxCount = 0;
72  minCount = 0;
73  for (int j = 0; j < matrix1[i].length; ++j) {
74  if (matrix1[i][j] < 0 || matrix2[i][j] < 0) continue;
75  if (this.weightMatrix == null) count = 1;
76  else count = this.weightMatrix[i][j];
77  //calc val
78  bin = matrix1[i][j] - matrix2[i][j];
79  //count values with an bigger/smaller val
80  if (bin > 0) {
81  maxVal += bin;
82  maxCount++;
83  }
84  if (bin < 0) {
85  minVal += bin;
86  minCount++;
87  }
88 
89  //calc radix
90  bin = (bin / radix) * radix;
91  //fetch old value if existand
92  if (this.result.containsKey(bin)) count += this.result.get(bin);
93  //store value
94  this.result.put(bin, count);
95  }
96  if (maxCount > 0) {
97  oMax.put(-maxVal / maxCount, i);//minus to get biggest values first!
98  } else {
99  oMax.put(0, i);
100  }
101  if (minCount > 0) {
102  oMin.put(minVal / minCount, i);
103  } else {
104  oMin.put(0, i);
105  }
106  }
107  System.out.println("Max: ");
108  int num = 0;
109  for (Entry<Integer, Integer> entry : oMax.entrySet()) {
110  System.out.println(entry.getKey() + "\t" + entry.getValue());
111  num++;
112  if (num > 10) break;
113  }
114  num = 0;
115  System.out.println("Min: ");
116  for (Entry<Integer, Integer> entry : oMin.entrySet()) {
117  System.out.println(entry.getKey() + "\t" + entry.getValue());
118  num++;
119  if (num > 10) break;
120  }
121  }
