220  private boolean consumeParameter(File currentPath, String key, String value, EnumSet<ParamFlag> flagSet, EnumSet<ParamString> stringSet, EnumSet<ParamValue> valueSet) {
221  boolean consumed = false;
222  if (!consumed) {
223  try {
224  ParamFlag pf = ParamFlag.valueOf(key);
225  if (flagSet.contains(pf)) {
226  if (value.equalsIgnoreCase("true")) {
227  this.setFlag(pf, true);
228  flagSet.remove(pf);
229  }
230  if (value.equalsIgnoreCase("false")) {
231  this.setFlag(pf, false);
232  flagSet.remove(pf);
233  } else {
234  TPS_Logger.log(SeverenceLogLevel.ERROR,
235  "Illegal " + "Parameter value: " + key + " -> " + value + " not in {true, false}");
236  }
237  } else {
238  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {
239  TPS_Logger.log(SeverenceLogLevel.WARN,
240  "Parameter " + "already defined: " + key + " -> " + this.paramFlagClass.isTrue(pf));
241  }
242  consumed = true;
243  }
244  consumed = true;
245  } catch (IllegalArgumentException e) {
246  /*
247  * ParamFlag pf = ParamFlag.valueOf(key.toString()); this line
248  * throws the exception, i.e. that there exists no enum constant
249  * with this name
250  */
251  }
252  }
253  if (!consumed) {
254  try {
255  ParamString ps = ParamString.valueOf(key);
256  if (stringSet.contains(ps) || ParamString.FILE_PARENT_PROPERTIES.equals(ps)) {
257  this.setString(ps, value);
258  stringSet.remove(ps);
259  consumed = true;
260 
261  switch (ps) {
262  case FILE_DATABASE_PROPERTIES:
263  case FILE_LOGGING_PROPERTIES:
264  case FILE_PARAMETER_PROPERTIES:
265  case FILE_PARENT_PROPERTIES:
266  if (currentPath != null) {
267  String propertiesFileName = this.getString(ps);
268  File thisPath = new File(currentPath.getAbsolutePath());
269  while (propertiesFileName.startsWith("./")) {
270  thisPath = thisPath.getParentFile();
271  propertiesFileName = propertiesFileName.substring(2);
272  }
273  this.parameterFiles.push(new File(thisPath, propertiesFileName));
274  }
275  break;
276  default:
277  // everything ok
278  }
279  } else {
280  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {
281  TPS_Logger.log(SeverenceLogLevel.WARN,
282  "Parameter " + "already defined: " + key + " -> " + this.getString(ps));
283  }
284  consumed = true;
285  }
286  } catch (IllegalArgumentException e) {
287  /*
288  * ParamString ps = ParamString.valueOf(key.toString()); this
289  * line throws the exception, i.e. that there exists no enum
290  * constant with this name
291  */
292  }
293  }
294  if (!consumed) {
295  try {
296  ParamValue pv = ParamValue.valueOf(key);
297  if (valueSet.contains(pv)) {
298  this.paramValueClass.setValue(pv, Double.parseDouble(value));
299  valueSet.remove(pv);
300  consumed = true;
301  } else {
302  if (TPS_Logger.isLogging(SeverenceLogLevel.WARN)) {
303  TPS_Logger.log(SeverenceLogLevel.WARN, "Parameter " + "already defined: " + key + " -> " +
304  this.paramValueClass.getDoubleValue(pv));
305  }
306  consumed = true;
307  }
308  } catch (NumberFormatException e) {
309  TPS_Logger.log(SeverenceLogLevel.ERROR, "Illegal parameter " + "value: " + key, e);
310  consumed = true;
311  } catch (IllegalArgumentException e) {
312  /*
313  * ParamValue pv = ParamValue.valueOf(key.toString()); this line
314  * throws the exception, i.e. that there exists no enum constant
315  * with this name
316  */
317  }
318  }
319  if (!consumed) {
320  if (TPS_Logger.isLogging(SeverenceLogLevel.DEBUG)) {
321  TPS_Logger.log(SeverenceLogLevel.DEBUG, "Unknown parameter defined in file: " + key);
322  }
323  }
324  return consumed;
325  }

332  PreparedStatement prepareStatement = null;
333  try {
334  prepareStatement = con.prepareStatement(
335  "DELETE FROM " + "simulation_parameters WHERE sim_key=? AND param_key=?");
336  prepareStatement.setString(1, sim_key);

504  private void load(File currentPath, File propertiesFile, EnumSet<ParamFlag> flagSet, EnumSet<ParamString> stringSet, EnumSet<ParamValue> valueSet) throws FileNotFoundException, IOException {
505  boolean consumed;
506  String key;
507  String value;
508  int counter = 0;
509  CsvReader reader = new CsvReader(new FileReader(propertiesFile));
510  reader.readHeaders();
511  while (reader.readRecord()) {
512  key = reader.get(0);
513  value = reader.get(1);
514  consumed = consumeParameter(currentPath, key, value, flagSet, stringSet, valueSet);
515  if (consumed) {
516  counter++;
517  }
518  }
519  reader.close();
520  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
521  TPS_Logger.log(SeverenceLogLevel.INFO,
522  "Load file: " + propertiesFile.getPath() + " with " + counter + " key-value pairs");
523  }
524  }

537  public void loadRuntimeParameters(File runPropertiesFile) throws IOException {
538  EnumSet<ParamFlag> flagSet = EnumSet.allOf(ParamFlag.class);
539  EnumSet<ParamString> stringSet = EnumSet.allOf(ParamString.class);
540  EnumSet<ParamValue> valueSet = EnumSet.allOf(ParamValue.class);
541 
542  this.setString(ParamString.PATH_ABS_PROPERTIES, runPropertiesFile.getParent());
543  stringSet.remove(ParamString.PATH_ABS_PROPERTIES);
544 
545  File absPath = runPropertiesFile;
546  while (!SIM_DIR.startsWith(absPath.getName())) {
547  absPath = absPath.getParentFile();
548  }
549  absPath = absPath.getParentFile();
550  File absInput = new File(absPath, INPUT_DIR);
551  this.setString(ParamString.PATH_ABS_INPUT, absInput.getPath());
552  this.setString(ParamString.PATH_ABS_OUTPUT, this.getString(ParamString.PATH_ABS_PROPERTIES));
553  stringSet.remove(ParamString.PATH_ABS_INPUT);
554  stringSet.remove(ParamString.PATH_ABS_OUTPUT);
555 
556  this.parameterFiles.push(runPropertiesFile);
557 
558  File actFile = null;
559  while (!this.parameterFiles.empty()) {
560  actFile = this.parameterFiles.pop();
561  load(actFile.getParentFile(), actFile, flagSet, stringSet, valueSet);
562  }
563 
564  generateTemporaryParameters();
565 
566  if (this.isDefined(ParamString.LOG_CLASS)) TPS_Logger.setLoggingClass(this.getString(ParamString.LOG_CLASS),
567  this);
568  }
