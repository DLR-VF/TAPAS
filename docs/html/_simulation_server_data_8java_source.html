156  public void setServerProcessInfo(ResultSet serverinfo) throws SQLException {
157 
158  if (this.serverprocessinfo == null) this.serverprocessinfo = new HashMap<>();
159 
160  this.serverprocessinfo.put(ServerInfo.IDENTIFIER, serverinfo.getString("identifier"));
161  this.serverprocessinfo.put(ServerInfo.RUNSINCE, serverinfo.getString("start_time"));
162  this.serverprocessinfo.put(ServerInfo.PID, serverinfo.getString("p_id"));
163  this.serverprocessinfo.put(ServerInfo.WORKON, serverinfo.getString("sim_key"));
164 
165  int hashstatus = serverinfo.getInt("sha512_status");
166 
167  switch (hashstatus) {
168  case -1:
169  this.hashstatus = HashStatus.INVALID;
170  this.hashstring = "";
171  break;
172  case 0:
173  this.hashstatus = HashStatus.NON_DOMINANT;
174  this.hashstring = serverinfo.getString("sha512");
175  break;
176  case 1:
177  this.hashstatus = HashStatus.DOMINANT;
178  this.hashstring = serverinfo.getString("sha512");
179  break;
180  case 2:
181  this.hashstatus = HashStatus.DEBUG;
182  this.hashstring = serverinfo.getString("sha512");
183  break;
184  default:
185  this.hashstatus = HashStatus.UNKNOWN;
186  this.hashstring = "";
187  }
188  }

259  public void update(ResultSet rs) throws SQLException {
260  name = rs.getString("server_name");
261  cpu = rs.getDouble("server_usage");
262  online = rs.getBoolean("server_online");
263  cores = rs.getInt("server_cores");
264  timestamp = rs.getTimestamp("server_timestamp");
265  remote_control = rs.getBoolean("remote_boot");
266  if (this.serverprocessinfo != null) serverprocessinfo.put(ServerInfo.SHUTDOWN,
267  Boolean.toString(remote_control));
268 
269  if (!online) {
