128  public double getIndirectWayFactor(String modus, String terrain, double distance) {
129  double iwf;
130  iwf = factors[0] / Math.log10((factors[1] * Math.pow(distance / 1000.0, factors[2])) + factors[3]);
131  iwf += factors[4] + this.correctionFactorModus.get(modus) + this.correctionFactorTerrain.get(terrain);
132 
133  return iwf;
134  }

136  public double[][] getTravelTime(String modus) {
137 
138  double Vm = this.calcData.getTravelTimeParameter()[Modus.valueOf(modus).getValue()].getSpeed() * 1000.0 / 60.0;
139 
140  // Tm = L *Um / Vm + Zm
141  // tm: traveltime by modus
142  // L: bee-line
143  // Um: indirect way factor by modus
144  // Vm speed by modus
145  // Zm: additional travel time by modus
146 
147  for (int i = 0; i < beeLine.length; i++) {
148  for (int k = 0; k < beeLine[0].length; k++) {
149  travelTime[i][k] = beeLine[i][k];
150  travelTime[i][k] /= Vm;
151  travelTime[i][k] += this.calcData.getTravelTimeParameter()[Modus.valueOf(modus).getValue()]
152  .getTimeDeparture();
153  travelTime[i][k] *= 60.0;
154  }
155  }
156 
157 
158  return travelTime;
