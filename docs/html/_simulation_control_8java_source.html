510  private int loadParameters(File name, HashMap<String, String> parameters) throws IOException {
511  String key;
512  String value;
513  String parent;
514  int counter = 0;
515  CsvReader reader = new CsvReader(new FileReader(name.getAbsolutePath()));
516  reader.readHeaders();
517  while (reader.readRecord()) {
518  key = reader.get(0);
519  value = reader.get(1);
520 
521  if (key.equals("FILE_PARENT_PROPERTIES") || key.equals("FILE_FILE_PROPERTIES") || key.equals(
522  "FILE_LOGGING_PROPERTIES") || key.equals("FILE_PARAMETER_PROPERTIES") || key.equals(
523  "FILE_DATABASE_PROPERTIES")) {
524  parent = name.getParent();
525  while (value.startsWith("./")) {
526  value = value.substring(2);
527  parent = new File(parent).getParent();
528  }
529  this.parameterFiles.push(new File(parent, value));
530  } else {
531  if (!parameters.containsKey(key)) {
532  counter++;
533  parameters.put(key, value); // this does not overwrites old
534  // values!
535  }
536  }
537  }
538  reader.close();
539 
540  return counter;
541  }
