60  public Double getLogSum(TPS_Plan plan, TPS_PlanningContext pc, TPS_ModeChoiceContext prevMCC, TPS_ModeChoiceContext nextMCC, TPS_TrafficAnalysisZone taz, TPS_ParameterClass parameterClass, double mu) {
61 
62  Double logSum = (double) 0;
63  double arrModeLogSum, depModeLogSum;
64 
65 
66  // The WALK-mode is used to get distances on the net.
67  double distanceNetTo = Math.max(parameterClass.getDoubleValue(ParamValue.MIN_DIST), TPS_Mode.get(ModeType.WALK)
68  .getDistance(
69  prevMCC.fromStayLocation,
70  prevMCC.toStayLocation,
71  SimulationType.SCENARIO,
72  null));
73  double distanceNetFrom = Math.max(parameterClass.getDoubleValue(ParamValue.MIN_DIST),
74  TPS_Mode.get(ModeType.WALK)
75  .getDistance(nextMCC.fromStayLocation, nextMCC.toStayLocation, SimulationType.SCENARIO, null));
76 
77  arrModeLogSum = this.getModeLogSum(plan, pc, taz, distanceNetFrom, distanceNetFrom + distanceNetTo, prevMCC,
78  parameterClass, mu);
79  depModeLogSum = this.getModeLogSum(plan, pc, taz, distanceNetTo, distanceNetFrom + distanceNetTo, nextMCC,
80  parameterClass, mu);
81 
82  if (arrModeLogSum > 0 && depModeLogSum > 0) {
83  logSum = Math.log(arrModeLogSum + depModeLogSum) / mu;
84 
85  return logSum;
86  } else {
87  return Double.NaN; //not reachable!
88  }
89  }

91  public double getModeLogSum(TPS_Plan plan, TPS_PlanningContext pc, TPS_TrafficAnalysisZone taz, double distanceNet, double distanceNetInclReturn, TPS_ModeChoiceContext mcc, TPS_ParameterClass parameterClass, double mu) {
92  // getting the distribution of modes from the mode choice tree according to the attributes of the plan
93  plan.setAttributeValue(TPS_Attribute.CURRENT_DISTANCE_CLASS_CODE_MCT,
94  TPS_Distance.getCode(TPS_DistanceCodeType.MCT, distanceNet));
95  //these attributes need to be set here again, because it is not guarantied, that they are set elsewhere
96  TPS_ActivityConstant currentActCode = mcc.toStay.getActCode();
97  plan.setAttributeValue(TPS_Attribute.CURRENT_EPISODE_ACTIVITY_CODE_MCT,
98  currentActCode.getCode(TPS_ActivityCodeType.MCT));
99  plan.setAttributeValue(TPS_Attribute.CURRENT_EPISODE_ACTIVITY_CODE_VOT,
100  currentActCode.getCode(TPS_ActivityCodeType.VOT));
101  plan.setAttributeValue(TPS_Attribute.CURRENT_EPISODE_ACTIVITY_CODE_TAPAS,
102  currentActCode.getCode(TPS_ActivityCodeType.TAPAS));
103 
104  // init
105 
106  TPS_DiscreteDistribution<TPS_Mode> dist = TPS_ModeDistribution.getDistribution(null);
107 
108  double[] utilities = new double[dist.size()];
109  double sumOfUtilities = 0;
110  // calculate utilities
111  for (int i = 0; i < utilities.length; ++i) {
112  utilities[i] = 0; //old habbit: init the value...
113  // get the parameter set
114  TPS_Mode mode = TPS_Mode.get(TPS_Mode.MODE_TYPE_ARRAY[i]);
115  if (!mcc.isBikeAvailable && mode.isType(ModeType.BIKE) || //no bike
116  mcc.carForThisPlan == null && mode.isType(ModeType.MIT) || //no car
117  (mode.isType(ModeType.TAXI) && mode.getParameters().isFalse(ParamFlag.FLAG_USE_TAXI)) //disable TAXI
118  ) {
119  utilities[i] = TPS_UtilityFunction.minModeProbability;
120  continue;
121  }
122  //travel time
123  double travelTime = mode.getTravelTime(mcc.fromStayLocation, mcc.toStayLocation, mcc.startTime,
124  SimulationType.SCENARIO, TPS_ActivityConstant.DUMMY, TPS_ActivityConstant.DUMMY, plan.getPerson(),
125  mcc.carForThisPlan);
126  if (TPS_Mode.noConnection(travelTime)) { //no connection
127  utilities[i] = TPS_UtilityFunction.minModeProbability;
128  continue;
129  }
130  //calc the value
131  utilities[i] = ((TPS_UtilityMNL) (TPS_Mode.getUtilityFunction(parameterClass))).getCostOfMode(mode, plan,
132  travelTime, mcc, SimulationType.SCENARIO);
133  if (Double.isNaN(utilities[i])) {
134  utilities[i] = TPS_UtilityFunction.minModeProbability;
135  continue;
136  }
137  // restricted cars will not enter restricted areas
138  if (pc.carForThisPlan != null && pc.carForThisPlan.isRestricted() && taz.isRestricted() && mode.isType(
139  ModeType.MIT)) {
140  continue;
141  }
142 
143  //too long trips get capped for walk
144  if (mode.isType(ModeType.BIKE) && distanceNetInclReturn > parameterClass.getIntValue(
145  ParamValue.MAX_WALK_DIST)) {
146  continue;
147  }
148 
149 
150  utilities[i] = TPS_FastMath.exp(mu * utilities[i]);
151 
152  sumOfUtilities += utilities[i];
153  }
154 
155  if (sumOfUtilities > 0) {
156  return sumOfUtilities;
157  }
158  return -1;
159  }

162  public TPS_Location selectLocationFromChoiceSet(TPS_RegionResultSet choiceSet, TPS_Plan plan, TPS_PlanningContext pc, TPS_LocatedStay locatedStay) {
163  if (this.PM == null) {
164  TPS_Logger.log(SeverenceLogLevel.FATAL,
165  "TPS_LocationSelectModel not properly initialized! Persistance manager is null?!?! Driving home!");
166  return plan.getPerson().getHousehold().getLocation();
167  }
168  if (this.region == null) {
169  TPS_Logger.log(SeverenceLogLevel.FATAL,
170  "TPS_LocationSelectModel not properly initialized! Region is null?!?! Driving home!");
171  return plan.getPerson().getHousehold().getLocation();
172  }
173 
174  if (!(TPS_Mode.getUtilityFunction(
175  this.PM.getParameters()) instanceof TPS_UtilityMNL)) { // no LogSum for non MNL-Models!
176  super.selectLocationFromChoiceSet(choiceSet, plan, pc, locatedStay);
177  }
178 
179 
180  TPS_Stay stay = locatedStay.getStay();
181  TPS_ActivityConstant actCode = stay.getActCode();
182  TPS_TourPart tourpart = (TPS_TourPart) locatedStay.getStay().getSchemePart();
183  TPS_Stay comingFrom = tourpart.getStayHierarchy(stay).getPrevStay();
184  TPS_Stay goingTo = tourpart.getStayHierarchy(stay).getNextStay();
185  TPS_Location locComingFrom = plan.getLocatedStay(comingFrom).getLocation();
186  TPS_Location locGoingTo = plan.getLocatedStay(goingTo).getLocation();
187  TPS_SettlementSystem regionType = locComingFrom.getTrafficAnalysisZone().getBbrType();
188  double muScale = this.PM.getParameters().getDoubleValue(
189  ParamValue.LOGSUM_WEIGHT_MU); // should be 1 if the logsums are in a normal range and not completely out of range
190  double muInteract = this.PM.getParameters().getDoubleValue(
191  ParamValue.LOGSUM_INTERACT_MU); // this should be estimated in the logit model
192 
193  double calibFactor = 1;
194  switch (actCode.getCode(TPS_ActivityCodeType.TAPAS)) {
195  case 0: //free time home
196  calibFactor = this.PM.getParameters().getDoubleValue(ParamValue.LOGSUM_CALIB_FREETIME_HOME);
197  break;
198  case 1: //work
199  calibFactor = this.PM.getParameters().getDoubleValue(ParamValue.LOGSUM_CALIB_WORK);
200  break;
201  case 2: //education
202  calibFactor = this.PM.getParameters().getDoubleValue(ParamValue.LOGSUM_CALIB_EDUCATION);
203  break;
204  case 3: //shopping
205  calibFactor = this.PM.getParameters().getDoubleValue(ParamValue.LOGSUM_CALIB_SHOP);
206  break;
207  case 4: //private matters
208  calibFactor = this.PM.getParameters().getDoubleValue(ParamValue.LOGSUM_CALIB_ERRANT);
209  break;
210  case 5: //free time
211  calibFactor = this.PM.getParameters().getDoubleValue(ParamValue.LOGSUM_CALIB_FREETIME);
212  break;
213  case 6: //misc
214  calibFactor = this.PM.getParameters().getDoubleValue(ParamValue.LOGSUM_CALIB_MISC);
215  break;
216  case 7: //university
217  calibFactor = this.PM.getParameters().getDoubleValue(ParamValue.LOGSUM_CALIB_STUDENT);
218  break;
219  default:
220  calibFactor = 1;
221  break;
222  }
223  muInteract *= calibFactor;
224 
225  // different cnf4-params according to the type of trip
226  double cfn4 = region.getCfn().getCFN4Value(regionType, actCode);
227  double cnfX = region.getCfn().getCFNXValue(regionType);
228  double param = 1;
229  TPS_CFN pot = region.getPotential();
230  if(pot!=null){
231  double val = pot.getParamValue(regionType, actCode);
232  if (val>=0)
233  param = val;
234  };
235 
236  SortedSet<WeightedResult> weightedChoiceSet = new TreeSet<>();
237  TPS_Location locRepresentant = null;
238  double sumWeight = 0;
239  for (Result result : choiceSet.getResultIterable()) {
240  if (result.sumWeight <= 0) continue;
241  // Draw a location at random. This location represents the zone for this round.
242  TPS_TrafficAnalysisZone taz = result.taz;
243  locRepresentant = result.loc;
244  //here we can switch between different travel time models
245  TPS_ModeChoiceContext prevMCC = new TPS_ModeChoiceContext();
246  prevMCC.isBikeAvailable = pc.isBikeAvailable;
247  prevMCC.carForThisPlan = pc.carForThisPlan;
248  prevMCC.duration = stay.getOriginalDuration();
249  prevMCC.startTime = stay.getOriginalStart();
250  prevMCC.fromStayLocation = locComingFrom;
251  prevMCC.fromStay = comingFrom;
252  prevMCC.toStayLocation = locRepresentant;
253  prevMCC.toStay = stay;
254  TPS_ModeChoiceContext nextMCC = new TPS_ModeChoiceContext();
255  nextMCC.isBikeAvailable = pc.isBikeAvailable;
256  nextMCC.carForThisPlan = pc.carForThisPlan;
257  nextMCC.duration = stay.getOriginalDuration();
258  nextMCC.startTime = stay.getOriginalStart();
259  nextMCC.fromStayLocation = locRepresentant;
260  nextMCC.fromStay = stay;
261  nextMCC.toStayLocation = locGoingTo;
262  nextMCC.toStay = goingTo;
263 
264 // //we need the beeline for calibration
265 // Double beeLineDist = this.PM.getParameters().paramMatrixClass.getValue(ParamMatrix.DISTANCES_BL, locComingFrom.getTAZId(), taz.getTAZId())+
266 // this.PM.getParameters().paramMatrixClass.getValue(ParamMatrix.DISTANCES_BL, taz.getTAZId(), locGoingTo.getTAZId()) ;
267 
268  Double weightedTT = this.getLogSum(plan, pc, prevMCC, nextMCC, taz, this.PM.getParameters(), muInteract);
269  if (!weightedTT.isNaN()) {
270  weightedTT = TPS_FastMath.exp(muScale * weightedTT); //use the activity based here..
271  //here we can switch between different Opportunity-weighting
272  WeightedResult weightedResult = this.createLocationOption(result, weightedTT, param);
273  //now the calib the weight with the distance
274  //weightedResult.result.sumWeight *= calibFactor/beeLineDist;
275 
276  sumWeight += weightedResult.getAdaptedWeight();
277  weightedChoiceSet.add(weightedResult);
278  }
279  }
280 
281 
282  if (weightedChoiceSet.size() > 0) {
283  double rand = Randomizer.random(); //uniform distribution from 0 to 1
284  double posMicro = rand;
285 
286  if (this.PM.getParameters().isDefined(ParamValue.GAMMA_LOCATION_WEIGHT)) {
287  posMicro = Math.pow(posMicro, this.PM.getParameters().getDoubleValue(ParamValue.GAMMA_LOCATION_WEIGHT));
288  }
289 
290  posMicro *= sumWeight; // norm to max weight
291  posMicro *= cnfX; //apply region-specific restiction factor
292  posMicro *= cfn4; // apply activity based restiction factor
293 
294  double weightPos = 0;
295  for (WeightedResult entry : weightedChoiceSet) {
296  weightPos += entry.getAdaptedWeight();
297  if (weightPos >= posMicro) {
298  locRepresentant = entry.result.loc;
299  break;
300  }
301  }
302  } else {
303  TPS_Logger.log(SeverenceLogLevel.WARN,
304  "No specific location found for activity " + actCode.getCode(TPS_ActivityCodeType.ZBE));
305  return region.selectDefaultLocation(plan, pc, locatedStay);
306  }
307  return locRepresentant;
308  }
