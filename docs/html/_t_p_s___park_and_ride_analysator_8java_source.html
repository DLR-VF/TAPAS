46  public void analyzeTrips(String name) {
47  //store into DB
48  String query = "";
49 
50  try {
51 
52  query = "SELECT taz_id_start, taz_id_end, start_time_min, activity_start_min, activity_duration_min FROM " +
53  name + " where mode = 2";
54 
55  ResultSet rs = this.dbCon.executeQuery(query, this);
56  int from, to, interchange = 0, num;
57  double interchangeD;
58  while (rs.next()) {
59  from = rs.getInt("taz_id_start");
60  if (!interchangeCounter.containsKey(from)) {
61  interchangeCounter.put(from, 0);
62  }
63  from -= 1;
64  to = rs.getInt("taz_id_end") - 1;
65  interchangeD = (int) interchanges.getValue(from, to);
66  if (interchangeD >= 0.) {
67  interchange = (int) (interchangeD + 1.5); //round plus index increment
68  num = 1;
69  if (interchangeCounter.containsKey(interchange)) {
70  num += interchangeCounter.get(interchange);
71  }
72  interchangeCounter.put(interchange, num);
73  }
74  if (this.originalMIVTimes != null && this.pnrMIVTimes != null && interchangeD >= 0.) {
75 
76  double diff = this.pnrMIVTimes.getValue(from, to) - this.originalMIVTimes.getValue(from, to);
77  int bin = (int) (diff / timeBinWitdh);
78  int numCount = 1;
79  if (connTimeHisto.containsKey(bin)) {
80  numCount += connTimeHisto.get(bin);
81  }
82  connTimeHisto.put(bin, numCount);
83  }
84  if (this.isPnR.get(from) != this.isPnR.get(to) && interchange > 0) { //in or out the region
85  if (this.isPnR.get(to)) {
86 
87  if (!this.pnrOccupancy.containsKey(interchange)) {
88  this.pnrOccupancy.put(interchange, new PnRSpace());
89  }
90  PnRSpace tmp = this.pnrOccupancy.get(interchange);
91  int start = rs.getInt("start_time_min");
92  int duration = rs.getInt("activity_duration_min") + 2 * (rs.getInt("activity_start_min") -

94  tmp.bookParkingLot(start, duration);
95  }
96  }
97  }
98  rs.close();
99  for (int i = 0; i < this.pnrMIVTimes.getNumberOfRows(); ++i) {
100  for (int j = 0; j < this.pnrMIVTimes.getNumberOfColums(); ++j) {

113  }

202  class PnRSpace {
203  int index;
204  int[] parkingTimes = new int[60 * 24];
205 
206  public void bookParkingLot(int start, int duration) {
207  if (start < 0) start = 0;
208  start = Math.min(start, parkingTimes.length);
209  int end = Math.min(start + duration, parkingTimes.length);
210  for (int i = start; i < end; ++i) {
211  parkingTimes[i] += 1;
212  }
213  }
214 
215  public int maxParkingOccupancy() {
216  int max = 0;
217  for (int parkingTime : parkingTimes) {
218  max = Math.max(max, parkingTime);
219  }
220  return max;
221  }
222 
223  public int peakTimeParking() {
224  int max = 0, indexOfMax = -1;
225  for (int i = 0; i < parkingTimes.length; ++i) {
226  if (max < parkingTimes[i]) {
227  max = parkingTimes[i];
228  indexOfMax = i;
229  }
230  }
231  return indexOfMax;
232  }
