143  public ArrayList<TASplitData> getSplit(Enum... c) {

147 
148  // split category:
149  int idxSplit = analyzers.length - 1;
150  Categories catSplit = analyzers[idxSplit].getCategories();
151  ArrayList<TripStats> stats = new ArrayList<>();
152  for (int i = 0; i < catSplit.getEnumeration().getEnumConstants().length; ++i) {
153  stats.add(new TripStats(0, 0, 0));
154  }
155 
156  TripStats total = new TripStats(0, 0, 0);
157 
158  for (CategoryCombination cc : allCat) {
159  boolean skip = false;
160  for (int i = 0; i < c.length; ++i) {
161  if (cc.getCategories()[i] != c[i]) {
162  skip = true;
163  break;
164  }
165  }
166  if (skip) {
167  continue;
168  }
169  int curIdx = cc.getCategory(catSplit).ordinal();
170  TripStats ts = getTripStats(cc.getCategories());
171 
172  stats.get(curIdx).add(ts);
173  total.add(ts);
174  }
175 
176  ArrayList<TASplitData> result = new ArrayList<>();
177 
178  for (TripStats ts : stats) {
179  if (ts.getCntTrips() == 0) {
180  result.add(new TASplitData(0, 0, 0));
181  } else {
182  result.add(new TASplitData((double) ts.getCntTrips() / total.getCntTrips(), ts.getCntTrips(),
183  ts.getDist() / ts.getCntTrips()));
184  empty = false;
185  }
186  }
187 
188  if (empty) return null;
189  return result;
190  }
