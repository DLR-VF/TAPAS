533  " WHERE \"matrix_name\" = '" + entry + "'");
534  ResultSet rs = statement.executeQuery(query.toString());
535  if (rs.next()) {
536  Object array = rs.getArray(2).getArray();
537  if (array instanceof Integer[]) {
538  //parse data to memory model
539  Integer[] matrixVal = (Integer[]) array;
540  double[] values = new double[matrixVal.length];
541  for (int j = 0; j < matrixVal.length; ++j) values[j] = (double) matrixVal[j];

545  rb.getString("EDIT_BUTTON") + " " + entry);
546  int answer = JOptionPane.showConfirmDialog(this.controlInstance,
547  rb.getString("SAVE_TO_DATABASE_DIALOG"), rb.getString("SAVE_CHANGES"),
548  JOptionPane.YES_NO_OPTION);

550  //save data back
551  query = new StringBuilder(
552  "UPDATE core." + this.matrixDBNameComboBox.getSelectedItem().toString() +
553  " SET \"matrix_values\" = '{");
554  StringBuilder buffer;
555  for (int j = 0; j < matrix.getMatrix().getActualYSize(); ++j) {
556  buffer = new StringBuilder();
557  for (int k = 0; k < matrix.getMatrix().getActualXSize(); ++k) {
558  if (k == matrix.getMatrix().getActualXSize() - 1 &&
559  j == matrix.getMatrix().getActualYSize() - 1) buffer.append(
560  Math.round(matrix.getMatrix().getValue(j, k)));
561  else buffer.append(Math.round(matrix.getMatrix().getValue(j, k))).append(",");
562  }
563  query.append(buffer);
564  }
565  query.append("}' WHERE \"matrix_name\" = '").append(entry).append("'");
566  statement.execute(query.toString());
567  }
