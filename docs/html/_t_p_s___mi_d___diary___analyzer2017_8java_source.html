111  public void calcDiaryStatistics() {
112  Map<Integer, Integer> groupStatistics;
113  Map<Integer, Integer> groupSumStatistics;
114  Map<Integer, Integer> groupDistribution;
115  int countGroup;
116  for (Diary e : this.diaryMap.values()) {
117 
118  if (!diaryStatistics.containsKey(e.group)) { // make new group
119  groupStatistics = new HashMap<>();
120  for (Integer i : this.activities) {//fill with init-values for all activities
121  groupStatistics.put(i, 0);
122  }
123  diaryStatistics.put(e.group, groupStatistics);
124  } else {
125  groupStatistics = diaryStatistics.get(e.group);
126  }
127 
128  if (!diarySumStatistics.containsKey(e.group)) { // make new group
129  groupSumStatistics = new HashMap<>();
130  for (Integer i : this.activities) {//fill with init-values for all activities
131  groupSumStatistics.put(i, 0);
132  }
133  diarySumStatistics.put(e.group, groupSumStatistics);
134  } else {
135  groupSumStatistics = diarySumStatistics.get(e.group);
136  }
137 
138  if (!personGroupDistribution.containsKey(e.pGroup)) { //make new person group
139  groupDistribution = new HashMap<>();
140  for (Integer i : this.diaryGroups) {//fill with init-values for all groups
141  groupDistribution.put(i, 0);
142  }
143  personGroupDistribution.put(e.pGroup, groupDistribution);
144  } else {
145  groupDistribution = personGroupDistribution.get(e.pGroup);
146  }
147  //now fill the statistics
148  for (DiaryElement d : e.activities) {
149  if (d.stay && !d.home) {
150  groupStatistics.put(d.purpose, groupStatistics.get(d.purpose) + 1);
151  groupSumStatistics.put(d.purpose, groupSumStatistics.get(d.purpose) + d.getDuration());
152  }
153  }
154  //and count one diary for this persongroup
155  countGroup = groupDistribution.get(e.group);
156  groupDistribution.put(e.group, countGroup + 1);
157  }
158  }
