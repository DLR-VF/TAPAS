241  private MatrixMap loadSumoValues(String matrixMap, double weight) {
242  MatrixMap oldM = this.manager.readMatrixMap(matrixMap, 0, this);
243  MatrixMap newM = oldM.clone();
244 
245  Map<Integer, Integer> tazMap = new HashMap<>();
246  ResultSet rs;
247  String query = "";
248  String table = this.parameterClass.getString(ParamString.DB_TABLE_TAZ);
249  String entryTable;
250  int taz_num_start, taz_num_end, tazStart, tazEnd;
251  double tt;
252  double timeslot;
253  double weightedTT;
254  double weightOld = 1.0 - weight;
255  try {
256  query = "SELECT taz_id, taz_num_id FROM " + table;
257  rs = this.manager.executeQuery(query, this);
258  while (rs.next()) {
259  tazMap.put(rs.getInt("taz_num_id"), rs.getInt("taz_id") - 1); //-1 for offset!
260  }
261  rs.close();
262  //load the values from the DB:
263  table = this.parameterClass.getString(ParamString.DB_SCHEMA_TEMP) + "sumo_od_" + this.actualKey;
264  entryTable = this.parameterClass.getString(ParamString.DB_SCHEMA_TEMP) + "sumo_od_entry_" + this.actualKey;
265  query = "SELECT t.taz_id_start, t.taz_id_end, t.interval_end, e.travel_time_sec[3] as tt FROM " + table +
266  " as t join " + entryTable + " as e on t.entry_id = e.entry_id";
267  rs = this.manager.executeQuery(query, this);
268  while (rs.next()) {
269  taz_num_start = rs.getInt("taz_id_start");
270  taz_num_end = rs.getInt("taz_id_end");
271  tazStart = tazMap.get(taz_num_start);
272  tazEnd = tazMap.get(taz_num_end);
273  tt = rs.getDouble("tt");
274  timeslot = rs.getDouble("interval_end");
275  Matrix tmp = newM.getMatrix((int) timeslot);
276  weightedTT = tmp.getValue(tazStart, tazEnd) * weightOld + tt * weight; //weight the new value
277  tmp.setValue(tazStart, tazEnd, weightedTT);
278  }
279 
280  rs.close();
281 
282  } catch (SQLException e) {
283  System.err.println("Error in sql-statement: " + query);
284  e.printStackTrace();
285  }
286 
287 
288  return newM;
289  }

391  public void run() {
392  Runtime.getRuntime().addShutdownHook(new RunWhenShuttingDownSumo());
393 
394  while (keepOnRunning) {
395  this.searchSimsToProcess();
396  for (Entry<String, Integer> entries : this.simKeysTriggerIteration.entrySet()) {
397  this.actualKey = entries.getKey();
398  this.actualIter = entries.getValue();
399  if (this.actualKey.length() > 0) {
400  System.out.println(
401  "Processing simulation: " + this.actualKey + " in iteration: " + this.actualIter);
402 
403  this.readParameter();
404 
405  String mapName = this.parameterClass.getString(ParamString.DB_NAME_MATRIX_TT_MIT);
406  String newName = this.updateStringIter(mapName);
407 
408  MatrixMap mivTT = this.loadSumoValues(mapName, 0.33);
409  List<String> matrixNames = this.createNewMatrixMap(mapName, newName);
410  this.saveMatrices(mivTT, matrixNames);
411  this.manager.updateSingleParameter(this.actualKey, "DB_NAME_MATRIX_TT_MIT", newName);
412  this.manager.updateSingleParameter(this.actualKey, "ITERATION",
413  Integer.toString((this.actualIter + 1)));
414  this.resetSimulation(this.actualKey);
415  System.out.println("Finished simulation: " + this.actualKey + " in iteration: " + this.actualIter);
416  }
417  }
418  simKeysTriggerIteration.clear(); // everything done (for now)!
419  //wait a bit for next polling
420  try {
421  Thread.sleep(SumoPollingInterval * 1000);
422  } catch (InterruptedException e) {
423  e.printStackTrace();
424  }
425  }
426  }

489  private String updateStringIter(String val) {
490  String appendix = "_" + this.actualKey + "_IT_" + this.actualIter;
