377  public void start(String[] args) {
378  CruisingSpeedCalculate calculation;
379  StringWriter sqlVektor;
380 
381  // command line arguments present, if there are no properties -> show help
382  if (args != null && (args.length > 1 || properties == null)) {
383  System.out.println("reading parameter from command line.");
384  CommandLine lvCmd = cliParser(args);
385 
386  if (lvCmd == null) return;
387 
388  if (lvCmd.hasOption("newProperty")) {
389  writeNewPropertyFile();
390  System.exit(0);
391  }
392 
393  if (!lvCmd.hasOption("property")) { // write command line parameter into temporary property-file
394  writePropertyFile(lvCmd);
395  properties = new Properties();
396  writePropertyFile(lvCmd, true);
397  }
398 
399  }
400 
401  calcData = new CruisingSpeedCalcData();
402  valData = new CruisingSpeedValData();
403 
404  readProperties();
405  calculation = new CruisingSpeedCalculate(getCalcData());
406 
407  /* try{System.in.read();}
408  catch(IOException ex) {}*/
409 
410  for (int i = 0; i < 4; i++) {
411  if (getCalcData().getCalculation()[i]) {
412  double[][] dist = calculation.calcDistance(Modus.values()[i].toString());
413 
414  if (dist != null) {
415  if (calcData.isTop3()) TPS_Geometrics.calcTop3(dist);
416 
417  sqlVektor = TPS_BasicConnectionClass.matrixToStringWriterSQL(dist, 0);
418  writeMatrix(sqlVektor, Modus.values()[i].getName(), DISTANCE);
419 
420  if (getCalcData().getCalculation()[i + 4]) {
421  double[][] travelTime = calculation.getTravelTime(Modus.values()[i].toString());
422  if (travelTime != null) {
423  sqlVektor = TPS_BasicConnectionClass.matrixToStringWriterSQL(travelTime, 0);
424  writeMatrix(sqlVektor, Modus.values()[i].getName(), TRAVELTIME);
425  } else {
426  throw new RuntimeException("error in travel time calculation");
427  }
428  }
429  } else {
430  throw new RuntimeException("error in distance calculation");
431  }
432  }
433 
434 
435  }
436 
437  if (validation) {
438  CruisingSpeedValidate cv = new CruisingSpeedValidate(this.valData, this.calcData);
439  cv.start();
440  }
441 
442  java.awt.EventQueue.invokeLater(() -> {
443  });
444 
445  // Exit, OK.
446  System.out.println("Exit(0)");
447  }
