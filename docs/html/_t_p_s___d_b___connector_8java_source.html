350  public Connection getConnection(Object key) throws SQLException {
351  Connection c = connections.get(key);
352  if (c != null && c.isClosed()) {
353  c = null;
354  }
355  SQLException ex = null;
356  for (int i = 0; i < 120 && c == null; ++i) {
357  try {
358  if (i > 0) {
359  Thread.sleep(1000);
360  }
361  c = this.openConnection();
362  this.connections.put(key, c);
363  } catch (SQLException e) {
364  TPS_Logger.log(SeverenceLogLevel.ERROR,
365  "Error creating db-connection. Try: " + i + "\n" + e.getMessage(), e);
366  ex = e;
367  } catch (InterruptedException e) {
368  TPS_Logger.log(SeverenceLogLevel.ERROR, e.getMessage(), e);
369  }
370  }
371  if (c == null) {
372  if (ex == null) throw new RuntimeException("Unknown error while creating connection");
373  throw ex;
374  }
375  return c;
376  }

409  private Connection openConnection() throws SQLException {
410  String url = "jdbc:" + this.parameterClass.getString(ParamString.DB_TYPE) + "://" +
411  this.parameterClass.getString(ParamString.DB_HOST) + ":" + this.parameterClass.getIntValue(
412  ParamValue.DB_PORT) + "/" + this.parameterClass.getString(ParamString.DB_DBNAME);
413 
414 
415  if (this.myUserName == null) {
416  if (this.parameterClass.isDefined(ParamString.DB_USER)) {
417  this.myUserName = this.parameterClass.getString(ParamString.DB_USER);
418  } else {
419  JLabel label = new JLabel("Please enter database user:");
420  JTextField jtf = new JTextField();
421  JOptionPane.showConfirmDialog(null, new Object[]{label, jtf}, "Password:",
422  JOptionPane.OK_CANCEL_OPTION);
423  this.myUserName = jtf.getText();
424  }
425  }
426 
427  if (this.myUserPwd == null) {
428  if (this.parameterClass.isDefined(ParamString.DB_PASSWORD)) {
429  this.myUserPwd = this.parameterClass.getString(ParamString.DB_PASSWORD);
430  } else {
431  JLabel label = new JLabel("Please enter password for database user '" + this.myUserName + "':");

434  JOptionPane.OK_CANCEL_OPTION);

437  }

454  public MatrixMap readMatrixMap(String matrixName, int sIndex, Object caller) {
455  MatrixMap matrix = null;
456  String query = "SELECT \"matrixMap_num\", \"matrixMap_matrixNames\", \"matrixMap_distribution\" FROM " +
457  this.parameterClass.getString(ParamString.DB_TABLE_MATRIXMAPS) + " WHERE \"matrixMap_name\"='" +
458  matrixName + "'";
459  TPS_Logger.log(SeverenceLogLevel.INFO, "Loading matrix map distribution " + matrixName + " from DB: ");
460  try {
461  ResultSet rs = this.executeQuery(query, caller);
462  if (rs.next()) {
463  //get number of matrices to load
464  int numOfMatrices = rs.getInt("matrixMap_num");
465  // get matrix names
466  String[] matrix_names = TPS_DB_IO.extractStringArray(rs, "matrixMap_matrixNames");
467  // get distribution
468  double[] distribution = TPS_DB_IO.extractDoubleArray(rs, "matrixMap_distribution");
469  rs.close();
470  //check sizes
471  if (numOfMatrices != matrix_names.length || numOfMatrices != distribution.length) {
472  throw new SQLException("Couldn't load matrixmap " + matrixName +
473  " from database. Different array sizes (num, matrices, distribution): " + numOfMatrices +
474  " " + matrix_names.length + " " + distribution.length);
475  }
476 
477  //init matrix map
478  Matrix[] matrices = new Matrix[numOfMatrices];
479  //load matrix map
480  for (int i = 0; i < numOfMatrices; ++i) {
481  matrices[i] = this.readMatrix(matrix_names[i], sIndex, caller);
482  if(matrices[i] != null){
483  TPS_Logger.log(SeverenceLogLevel.INFO,
484  "Loaded matrix from DB: " + matrix_names[i] + " End time: " + distribution[i] +
485  " Average value: " + matrices[i].getAverageValue(false, true));
486  } else {
487  throw new SQLException(
488  "Couldn't load matrix " + matrix_names[i] + " form matrix map" + matrixName +
489  ": No such matrix.");
490  }
491  }
492  matrix = new MatrixMap(distribution, matrices);
493  } else {
494  TPS_Logger.log(SeverenceLogLevel.ERROR, "No results from DB!");
495  }
496  } catch (SQLException e) {
497  TPS_Logger.log(SeverenceLogLevel.ERROR, "Error in sql-query: " + query);
498  TPS_Logger.log(SeverenceLogLevel.ERROR, e);
499  }
500  return matrix;
501  }

510  public Matrix readMatrix(String matrixName, int sIndex, Object caller) {
511  Matrix returnVal = null;
512  String query ="";
513  try {
514  query = "SELECT matrix_values FROM " + this.parameterClass.getString(
515  ParamString.DB_TABLE_MATRICES) + " WHERE matrix_name='" + matrixName + "'";
516  ResultSet rs = this.executeQuery(query, caller);
517  if (rs.next()) {
518  int[] iArray = TPS_DB_IO.extractIntArray(rs, "matrix_values");
519  int len = (int) Math.sqrt(iArray.length);
520  returnVal = new Matrix(len, len, sIndex);
521  for (int index = 0; index < iArray.length; index++) {
522  returnVal.setRawValue(index, iArray[index]);
523  }
524  rs.close();
525  TPS_Logger.log(SeverenceLogLevel.INFO,
526  "Loaded matrix from DB: " + matrixName);
527  } else {
528  TPS_Logger.log(SeverenceLogLevel.ERROR, "No results from DB for"+ matrixName+
529  ": No such matrix.");
530  }
531  } catch (SQLException e) {
532  TPS_Logger.log(SeverenceLogLevel.ERROR, "Error in sql-query: " + query);
533  TPS_Logger.log(SeverenceLogLevel.ERROR, e);
