71  public TPS_Episode(int id, TPS_ActivityConstant actCode, int start, int duration, double startEarlier, double startLater, double shorterDuration, double longerDuration, TPS_ParameterClass parameterClass) {
72  this.id = id;
73  this.actCode = actCode;
74  this.originalStart = start;
75  this.originalDuration = duration;
76  this.startEarlier = startEarlier;
77  this.startLater = startLater;
78  this.alphaEarlier = 1.0;
79  this.alphaLater = 1.0;
80  this.shorterDuration = shorterDuration;
81  this.longerDuration = longerDuration;
82  this.betaShorter = 1.0;
83  this.betaLonger = (1.0);
84  this.parameterClass = parameterClass;
85 
86  // Calculation of the alpha and beta values according to
87 
88  double eps = 1; // one second
89  // alphaEarlier and alphaLater
90  double delta = start - startEarlier;
91  if (delta < eps) {
92  delta = eps;
93  }
94  delta *= this.parameterClass.getDoubleValue(ParamValue.SCALE_SHIFT);
95  alphaEarlier = 1.0 / (delta * delta);
96 
97  delta = startLater - start;
98  if (delta < eps) {
99  delta = eps;
100  }
101  delta *= this.parameterClass.getDoubleValue(ParamValue.SCALE_SHIFT);
102  alphaLater = 1.0 / (delta * delta);
103 
104  // betaMinus and betaPlus
105  delta = duration - shorterDuration;
106  if (delta < eps) {
107  delta = eps;
108  }
109  delta *= this.parameterClass.getDoubleValue(ParamValue.SCALE_STRETCH);
110  betaShorter = 1.0 / (delta * delta);
111 
112  delta = longerDuration - duration;
113  if (delta < eps) {
114  delta = eps;
115  }
116  delta *= this.parameterClass.getDoubleValue(ParamValue.SCALE_STRETCH);
117  betaLonger = 1.0 / (delta * delta);
118  }
