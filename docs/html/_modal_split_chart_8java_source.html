251  public ModalSplitRenderer(ModalSplitChartDataset dataset) {
252  super();
253  this.dataset = dataset;
254 
255  setDefaultToolTipGenerator(new ModalSplitToolTipGenerator());
256 
257  setUseOutlinePaint(true);
258  for (int i = 0; i < dataset.getSeriesCount(); ++i) {
259  if (dataset.isDataSeries(i)) {
260  setSeriesLinesVisible(i, false);
261  setSeriesShapesVisible(i, true);
262  } else {
263  setSeriesLinesVisible(i, true);
264  setSeriesShapesVisible(i, false);
265  setSeriesPaint(i, Color.black);
266  }
267  }
268 
269  }

271  @Override
272  public Paint getItemOutlinePaint(int row, int column) {
273  if (dataset.isDataSeries(row) && selected.contains(column)) {
274  return Color.BLACK;
275  } else {
276  return getItemPaint(row, column);

280  @Override
281  public Paint getItemPaint(int row, int column) {
282  if (!dataset.isDataSeries(row)) return Color.BLACK;
283  else return dataset.getQuality(column).getColor();
284  }
285 
286  @Override
287  public Shape getItemShape(int row, int column) {
288  if (!dataset.isDataSeries(row)) return null;
289 
290  Mode m = dataset.getMode(column);

293  return shapeMap.get(m);
