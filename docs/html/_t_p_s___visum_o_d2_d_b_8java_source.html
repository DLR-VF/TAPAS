88  public void generateIntraCellInfos(double weight) {
89 
90  int i, j;
91 
92  final int length = 3;
93 
94  double[] mins = new double[length];
95  double intra;
96 
97 
98  for (i = 0; i < this.matrix.length; ++i) {
99  //init the mins
100  for (j = 0; j < length; ++j) {
101  mins[j] = 1e100;
102  }
103  //find the mins
104  for (j = 0; j < this.matrix[0].length; ++j) {
105  if (i != j) { //diagonal is zero!
106  checkMultiMin(this.matrix[i][j], mins);
107  }
108  }
109  // init the avg
110  intra = 0;
111  //build average
112  for (j = 0; j < length; ++j) {
113  intra += mins[j];
114  }
115  intra *= weight / length; //intraTT is half the average to the n closest cells
116  this.matrix[i][i] = intra;
117 
118  }

155  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {

157  }
158  //read tvzs

160  TPS_Logger.log(SeverenceLogLevel.INFO, "Analyzing TVZ-IDs: ");
161  }
162 
163  while ((line = input.readLine()) != null) {
164  if (line.startsWith("$") || line.startsWith("*") || !line.startsWith(" ")) { // comment or cell name
165  continue;
166  }
167  String[] tok = line.replaceAll(" ", " ").trim().split(delimiter);
168  //check format
169  if (tok.length != 3) continue;
170 
171  //get from
172  fromTVZ = Integer.parseInt(tok[0].trim());
173 
174  if (idToIndex.get(fromTVZ) == null) {//new tvz
175  idToIndex.put(fromTVZ, tvzCounter);
176  indexToId.put(tvzCounter, fromTVZ);
177  tvzCounter++;
178  }
179 
180  //get to
181  toTVZ = Integer.parseInt(tok[1].trim());
182  if (idToIndex.get(toTVZ) == null) {//new tvz
183  idToIndex.put(toTVZ, tvzCounter);
184  indexToId.put(tvzCounter, toTVZ);
185  tvzCounter++;
186  }
187  }
188  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
189  TPS_Logger.log(SeverenceLogLevel.INFO, "Found " + tvzCounter + " TVZ-IDs");
190  }
191  input.close();
192  in.close();
193  }
194 
195

197  this.matrix = new double[tvzCounter][tvzCounter];

201  //open input

203  input = new BufferedReader(in);

205  TPS_Logger.log(SeverenceLogLevel.INFO, "File opened: " + fileName);
