131  public void calcBeelineTAZDist() {
132  for(Entry<Integer, List<Location>> e : this.locations.entrySet()) {
133  int start, stop;
134  double beelineAccumulated=0, beeline;
135  int sumWeight=0,weight;
136  Location l1, l2;
137  List<Location> locs= e.getValue();
138  for(start= 0; start< locs.size(); start++) {
139  l1= locs.get(start);
140  for(stop= start+1; stop <locs.size(); stop++) {
141  l2= locs.get(stop);
142  weight = Math.min(l1.cappa, l2.cappa);
143  beeline = l1.point.getEuclidianDistance(l2.point);
144  sumWeight+= weight;
145  beelineAccumulated += beeline*weight;
146  }
147  }
148  if(sumWeight>0)
149  this.intraTazDist.put(e.getKey(), (beelineAccumulated/sumWeight));
150  else
151  this.intraTazDist.put(e.getKey(), -1d);
152  }
153  }

201  public void calcBeeLineFactors() {
202  Location target;
203  double dist,tmp;
204  for(Location ref: this.tazes.values()) {
205  dist = Double.MAX_VALUE;
206  target = null;
207  for(Location t: this.tazes.values()) {
208  if(t.taz!=ref.taz){
209  tmp= this.dist.getValue(ref.taz-1, t.taz-1);
210  if(tmp<dist) {
211  target = t;
212  dist= tmp;
213  }
214  }
215  }
216  if(target!=null) {
217  double bl = ref.point.getEuclidianDistance(target.point);
218  this.tazBeeLineFactor.put(ref.taz, dist/bl);
219  }
220  else {
221  this.tazBeeLineFactor.put(ref.taz, 1.4);
222  }
223  }
