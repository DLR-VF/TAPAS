307  public static void main(String[] args) {
308  Options options = new Options();
309  options.addOption("n", "dbname", true, "Database Name");
310  options.addOption("u", "dbuser", true, "Database User with sufficient rights");
311  options.addOption("s", "dbserver", true, "Database Server Address");
312  options.addOption("p", "dbpassword", true, "Database Password of DBUSER");
313  options.addOption(Option.builder("h").longOpt("help").desc("Show help").build());
314 
315  //Create a parser
316  CommandLineParser parser = new DefaultParser();
317 
318  //parse the options passed as command line arguments
319  try {
320  CommandLine cmd = parser.parse(options, args);
321  if (cmd.hasOption('h')) {
322  help(options);
323  }
324  DBSERVER = promptCmdArgument(cmd, options, "dbserver", false);
325  DBNAME = promptCmdArgument(cmd, options, "dbname", false);
326  DBUSER = promptCmdArgument(cmd, options, "dbuser", false);
327  DBPASSWORD = promptCmdArgument(cmd, options, "dbpassword", true);
328  } catch (ParseException e) {
329  exitAndCleanUp(e, false);
330  }
331  connectToDB("postgres");
332  createDB(DBNAME);
333  // reconnect to new database
334  closeConnection();
335  connectToDB(DBNAME);
336 
337  //unzip necessary data
338  unzip(source, destination);
339 
340  executeCompleteQuery(destination);
341  //delete directory
342  if (!DESTINATION_ALREADY_EXISTS) deleteDirectory(new File(destination));
343  }

356  //first basic sql statements which are not a function

399  private static void existsDestinationDirectory(String destination) {
400  File f = new File(destination);
401  // we use !DESTINATION_ALREADY_EXISTS because we do not want to run into this condition again
402  // unzip will be called recursively and there existsDestinationDirectory too
403  // There are four possibilities here
404  // 1. destination already exists (and is a directory) at the beginning of the Installer script
405  // a) first call of existsDestinationDirectory -> it goes into the if
406  // b) later call of existsDestinationDirectory -> it goes into the if
407  // 2. destination does not exist already at the beginning of the Installer script
408  // a) first call of existsDestinationDirectory -> it goes into the else
409  // b) later call of existsDestinationDirectory (dir is created through the attempt of zip extraction) ->
410  // -> it goes into the else
411  if (f.exists() && f.isDirectory() && DESTINATION_ALREADY_EXISTS) {
412  System.out.println(destination + " already exists. It will not be deleted afterwards.");
413  } else {
414  DESTINATION_ALREADY_EXISTS = false;
415  System.out.println(destination + " does not exist already. It will be created and deleted afterwards. ");
416  }
417  }
