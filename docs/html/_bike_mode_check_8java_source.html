35  public void readTrips(String key) {
36  String query = "";
37  try {
38  boolean recordsFound;
39  int step = 0;
40  int chunk = 1000000;
41  int pID = -1, hhID = -1, homeID = -1;
42  boolean bikeForbidden = false;
43  do {
44  recordsFound = false;
45  query = "SELECT p_id, hh_id, start_time_min, mode, loc_id_start FROM " + key +
46  " ORDER BY hh_id, p_id, start_time_min LIMIT " + chunk + " OFFSET " + (step * chunk);
47  ResultSet rs = dbCon.executeQuery(query, this);
48  step++;
49  while (rs.next()) {
50  recordsFound = true;
51  int actPID = rs.getInt("p_id");
52  int actHHID = rs.getInt("hh_id");
53  int actMode = rs.getInt("mode");
54  int modeBike = 1;
55  if (pID == actPID && hhID == actHHID) {
56  if (rs.getInt("loc_id_start") == homeID) { //trip starts at home
57  bikeForbidden = actMode != modeBike; //is this a Bike?
58  this.tripCounter++;
59  } else if (actMode == modeBike && bikeForbidden) { //illegal use of bike?
60  this.defectTrips.add(new Trip(actPID, actHHID, rs.getInt("start_time_min")));
61  }
62  } else { //new person
63  this.tripCounter++;
64  pID = actPID;
65  hhID = actHHID;
66  bikeForbidden = actMode != modeBike; //is this a Bike?
67  homeID = rs.getInt("loc_id_start"); // all trips start at home
68  }
69  }
70  rs.close();
71  } while (recordsFound);
72 
73  } catch (SQLException e) {
74  System.err.println(this.getClass().getCanonicalName() + " readTrips: SQL-Error during statement: " + query);
75  e.printStackTrace();
76  }
77  }
