111  @Override

117  // System.out.println("Got Geometries");
118  for (EWZPojo ewz : geometries) {
119  ArrayList<Integer> addresses = getAddresses(ewz);
120 
121  if (addresses.size() == 0 && ewz.inhabitants > 0) {
122  System.err.println("No addresses in " + ewz.key + ". " + ewz.inhabitants +
123  " inhabitants could not be distributed.");
124  continue;
125  }
126 
127  List<AddressPojo> inhs = distributeInhabitants(addresses, ewz.inhabitants);
128 
129  queue.addAll(inhs);

131 
132  } catch (SQLException e) {

134  } finally {
