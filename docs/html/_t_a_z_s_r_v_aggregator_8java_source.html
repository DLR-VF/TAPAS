96  private double[][] calculateAggegation(double[][] source, boolean useMedian) {
97 
98  final int sizeStatArea = this.statisticalAreas.size();
99 
100  double[][] destination = new double[sizeStatArea][sizeStatArea];
101 
102 
103  List<Double> values;
104  double average;
105 
106  //wow! four nested loops!
107 
108  //over all areas
109  for (Integer fromS : this.statisticalAreas.keySet()) {
110  for (Integer toS : this.statisticalAreas.keySet()) {
111  values = new ArrayList<>();
112  //collect data
113  for (Integer fromT : this.statisticalAreas.get(fromS)) {
114  for (Integer toT : this.statisticalAreas.get(toS)) {
115  if (!fromT.equals(toT) || source[fromT - 1][toT - 1] > 0) {
116  values.add(source[fromT - 1][toT - 1]);
117  }
118  }
119  }
120  if (values.size() > 0) {
121  //build average
122  average = 0;
123  if (useMedian) { //mean of median center
124  Collections.sort(values);
125  //enough data: forget closest and farest cell
126  if (values.size() >= 3) {
127  for (int i = 1; i < values.size() - 1; ++i) {
128  average += values.get(i);
129  }
130  average /= values.size() - 2;
131  } else { //use all
132  for (Double value : values) {
133  average += value;
134  }
135  average /= values.size();
136  }
137  } else { //mean
138  for (Double value : values) {
139  average += value;
140  }
141  average /= values.size();
142  }
143  destination[fromS - 1][toS - 1] = average;
144  }
145  }
146  }
147  return destination;
148  }

155  private double[][] calculateBeeline() {
156 
157  int sizeTAZ = this.tazCentroids.size();
158  double[][] destination = new double[sizeTAZ][sizeTAZ];
159 
160 
161  //collect data
162  for (int fromT = 0; fromT < sizeTAZ; ++fromT) {
163  for (int toT = 0; toT < sizeTAZ; ++toT) {
164  if (fromT != toT || destination[fromT][toT] > 0) {
165  destination[fromT][toT] = this.tazCentroids.get(fromT + 1).distanceTo(
166  this.tazCentroids.get(toT + 1));
167  } else {
168  destination[fromT][toT] = 0;
169  }
170  }
171  }
172  return destination;
173  }

201  }
202  i = 0;

205  destination[j][i] = Integer.parseInt(tok);
