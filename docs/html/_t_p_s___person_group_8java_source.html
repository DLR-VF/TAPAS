50  public TPS_PersonGroup(int id, String[] attributes) {
51  if ((attributes.length - 4) % 3 != 0) {
52  throw new RuntimeException("Person need n*3 + 4 attributes n*(name, code, type) + (code_ageclass, " +
53  "code_sex, code_cars, persType): " + attributes.length);
54  }
55 
56  this.map = new EnumMap<>(TPS_PersonGroupType.class);
57  this.id = id;
58  TPS_InternalConstant<TPS_PersonGroupType> tic;
59  for (int i = 0; i < attributes.length - 4; i += 3) {
60  tic = new TPS_InternalConstant<>(attributes[i], Integer.parseInt(attributes[i + 1]),
61  TPS_PersonGroupType.valueOf(attributes[i + 2]));
62  this.map.put(tic.getType(), tic);
63  }
64 
65  for (TPS_PersonGroupType type : TPS_PersonGroupType.values()) {
66  if (!this.map.containsKey(type)) {
67  throw new RuntimeException(
68  "Person group code for " + this.getId() + " for type " + type.name() + " not " + "defined");
69  }
70  }
71  this.setAgeClasses(TPS_AgeClass
72  .getConstants(TPS_AgeCodeType.PersGroup, Integer.parseInt(attributes[attributes.length - 4])));
73  this.setSex(TPS_Sex.getEnum(Integer.parseInt(attributes[attributes.length - 3])));
74  this.setCarCode(TPS_CarCode.getEnum(Integer.parseInt(attributes[attributes.length - 2])));
75  this.setPersType(TPS_PersonType.valueOf(attributes[attributes.length - 1]));
76  }
