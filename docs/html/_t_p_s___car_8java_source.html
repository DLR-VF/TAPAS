303  public static TPS_Car selectCar(TPS_Plan plan, TPS_TourPart tourpart) {
304 
305  //check if a car is already attached to this tourpart
306  if (tourpart.isCarUsed()) {
307  return tourpart.getCar();
308  }
309 
310  //check for available cars
311  TPS_Car[] availableCars = plan.getPerson().getHousehold().getAvailableCars(
312  tourpart.getOriginalSchemePartStart(), tourpart.getOriginalSchemePartEnd());
313 
314  //select a car if available
315  if (availableCars != null) {
316  //TODO: make a choice here!
317  for (TPS_Car availableCar : availableCars) {
318  if (!availableCar.isRestricted()) // first take unrestricted cars
319  return availableCar;
320  }
321  return availableCars[0];
322  } else {
323  return null;
324  }

461  case 101: //ecomove Small
