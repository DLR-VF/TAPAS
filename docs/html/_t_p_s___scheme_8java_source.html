75  public boolean addEpisode(TPS_Episode episode) {
76  TPS_SchemePart sp;
77  if (this.schemeParts.isEmpty()) {
78  sp = new TPS_HomePart();
79  sp.setScheme(this);
80  this.schemeParts.add(sp);
81  } else {
82  sp = this.schemeParts.get(this.schemeParts.size() - 1);
83  }
84 
85  if (episode.isHomePart && sp.isTourPart()) {
86  if (sp.getLastEpisode().isStay()) {
87  sp.getLastEpisode().setOriginalDuration(
88  Math.min(this.parameterClass.getIntValue(ParamValue.SEC_TIME_SLOT),
89  sp.getLastEpisode().getOriginalDuration() - this.parameterClass
90  .getIntValue(ParamValue.SEC_TIME_SLOT))); //make the last stay 5 min shorter
91  TPS_ActivityConstant activity = TPS_ActivityConstant.getActivityCodeByTypeAndCode(
92  TPS_ActivityCodeType.ZBE, 80); // activity: moving
93  TPS_Trip takeMeHome = new TPS_Trip(sp.getId(), activity, (int) sp.getLastEpisode().getOriginalEnd(), 5,
94  this.parameterClass); // insert a "take me home in 5 minutes"-trip
95  ((TPS_TourPart) sp).addTrip(takeMeHome);
96  }
97  sp = new TPS_HomePart();
98  sp.setScheme(this);
99  this.schemeParts.add(sp);
100  } else if (episode.tourNumber > 0 && sp.isHomePart()) {
101  sp = new TPS_TourPart(episode.tourNumber);
102  sp.setScheme(this);
103  this.schemeParts.add(sp);
104  }
105 
106  try {
107  if (episode.isStay()) {
108  sp.addStay((TPS_Stay) episode, parameterClass);
109  } else {
110  ((TPS_TourPart) sp).addTrip((TPS_Trip) episode);
111  }
112  } catch (ClassCastException e) {
113  System.err.println("Scheme: " + sp.getScheme().getId());
114  System.err.println(episode.toString());
115  e.printStackTrace();
116  return false;
117  }
118  return true;
119  }
