81  private void readData() throws SQLException {
82  double avg;
83  double stdDev;
84  double tmp;
85  double[] values;
86  int count;
87  Vector<Double> vals = new Vector<>();
88  String statement = "select scheme_id,act_code_zbe from core.global_episodes order by scheme_id, start";
89  ResultSet rs = this.st.executeQuery(statement);
90  int lastId = -1, lastActCode = -1, id, actCode;
91  while (rs.next()) {
92  id = rs.getInt("scheme_id");
93  actCode = rs.getInt("act_code_zbe");
94  if (id == lastId) {
95  if ((actCode == 80 || actCode == 880) && (lastActCode == 80 || lastActCode == 880)) System.out.println(
96  "bad scheme: " + id);
97  }
98  lastActCode = actCode;
99  lastId = id;
100  }
101 
102 
103  int i, j;
104  for (i = this.minSchemeClass; i <= this.maxSchemeClass; ++i) {
105  statement =
106  "select sum(duration) from core.global_episodes epis where act_code_zbe in (80, 880) and scheme_id IN (select scheme_id from core.global_schemes where scheme_class_id = " +
107  i + ") group by scheme_id";
108  rs = this.st.executeQuery(statement);
109  this.con.commit();
110 
111  count = 0;
112  while (rs.next()) {
113  vals.add(rs.getInt(1) * slotFactor);
114  count += 1;
115  }
116  if (count > 0) {
117  avg = 0;
118  for (j = 0; j < vals.size(); ++j) {
119  avg += vals.elementAt(j);
120  }
121  avg = avg / (double) count;
122  stdDev = 0;
123  for (j = 0; j < vals.size(); ++j) {
124  tmp = (vals.elementAt(j) - avg);
125  stdDev += tmp * tmp;
126  }
127 
128  stdDev /= (count - 1);
129  stdDev = Math.sqrt(stdDev);
130  } else { // no entries for this class!
131  avg = 1;
132  stdDev = 1;
133  }
134 
135  values = new double[2];
136  values[0] = avg;
137  values[1] = stdDev / avg;
138  this.SchemeClassEntries.put(i, values);
139  vals.clear();
140  }
141  for (Integer ids : this.SchemeClassEntries.keySet()) {
142  values = this.SchemeClassEntries.get(ids);
143  System.out.println("Klasse: " + ids + " avg: " + values[0] + " sdtDev: " + values[1]);
144  }
