76  private double calcUtility(ChoiceParams param, TPS_Person person) {
77  TPS_Household myHH = person.getHousehold();
78  double utility = param.constant; //choice constant
79  if (person.getSex() == TPS_Sex.FEMALE) {
80  utility += param.female;
81  }
82  if (person.getAge() >= 65) {
83  utility += param.age65;
84  } else if (person.getAge() >= 45) {
85  utility += param.age45;
86  } else if (person.getAge() >= 25) {
87  utility += param.age25;
88  } else if (person.getAge() >= 18) {
89  utility += param.age18;
90  }
91 
92 
93  if (person.getEducationLevel() >= 3 && person.getEducationLevel() <= 4) {
94  utility += param.abi;
95  } //FIXME TPS_PersonGroup is no TPS_PersonType
96  if (person.getPersGroup().equals(TPS_PersonGroup.TPS_PersonType.RETIREE)) {
97  utility += param.retired;
98  }
99  if (person.isWorking()) {
100  utility += param.working;
101  }
102  //count adults, children, cars and bikes
103  double children = 0;
104  double adults = 0;
105  double cars = myHH.getCarNumber();
106  double bikes = 0;
107  for (TPS_Person p : myHH.getMembers(TPS_Household.Sorting.NONE)) {
108  if (p.getAge() >= 18) {
109  adults += 1.0;
110  } else {
111  children += 1.0;
112  }
113  if (p.hasBike()) {
114  bikes += 1.0;
115  }
116  }
117  utility += adults * param.numAdults;
118  utility += children > 0 ? param.hasChildren : 0;
119  utility += cars > 0 ? param.hasCars : 0;
120  utility += bikes * param.numBikes;
121 
122  if (myHH.getIncome() > 4000.0) {
123  utility += param.income4000;
124  } else if (myHH.getIncome() > 3200.0) {
125  utility += param.income3200;
126  } else if (myHH.getIncome() > 2000.0) {
127  utility += param.income2000;
128  } else if (myHH.getIncome() > 1100.0) {
129  utility += param.income1100;
130  } else if (myHH.getIncome() > 500.0) {
131  utility += param.income1100;
132  } else {
133  utility += param.income500;
134  }
135  return utility;
136  }
