68  public TPS_ActivityConstant(int id, String[] attributes, boolean isTrip, boolean isFix, String attr) {
69  if ((attributes.length) % 3 != 0) {
70  throw new RuntimeException(
71  "ActivityConstant need n*3 attributes n*(name, code, type): " + attributes.length);
72  }
73  this.id = id;
74  this.map = new EnumMap<>(TPS_ActivityCodeType.class);
75  TPS_InternalConstant<TPS_ActivityCodeType> iac;
76  for (int i = 0; i < attributes.length; i += 3) {
77  iac = new TPS_InternalConstant<>(attributes[i], Integer.parseInt(attributes[i + 1]),
78  TPS_ActivityCodeType.valueOf(attributes[i + 2]));
79  this.map.put(iac.getType(), iac);
80  }

82  for (TPS_ActivityCodeType type : TPS_ActivityCodeType.values()) {
