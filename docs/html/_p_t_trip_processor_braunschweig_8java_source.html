236  public void processTrip(String entry) {
237  entry = formatCSVLine(entry);
238 
239  StringTokenizer tok = new StringTokenizer(entry, ";");
240  int i = 0;
241  String token;
242  if (tok.countTokens() != NUM_TRIP_COLUMNS) {
243  System.out.println(
244  "Error: Found " + tok.countTokens() + " tokens. Expected: " + NUM_TRIP_COLUMNS + " tokens.");
245  while (tok.hasMoreTokens()) {
246  System.out.println(tok.nextToken());
247  }
248  return;
249  }
250 
251  StringBuilder start = new StringBuilder();
252  StringBuilder stop = new StringBuilder();
253  boolean arr = false, arrStop = false, depStop = false;
254  int tokensFound = 0;
255 
256  while (tok.hasMoreTokens()) {
257  token = tok.nextToken();
258  token = token.trim();
259  if (!token.equals("\"\"") && !token.equals("")) {
260  switch (i) {
261  case 3:
262  arr = token.equals("arr");
263  tokensFound++;
264  break;
265  case 6:
266  case 11:
267  if ((arr && i == 6) || (!arr && i == 11)) arrStop = token.equals("stop");
268  else depStop = token.equals("stop");
269  tokensFound++;
270  break;
271  case 9:
272  case 14:
273  if ((arr && i == 9) || (!arr && i == 14)) start = new StringBuilder(token + ", ");
274  else stop = new StringBuilder(token + ", ");
275  tokensFound++;
276  break;
277  case 10:
278  case 15:
279  if ((arr && i == 10) || (!arr && i == 15)) start.append(token);
280  else stop.append(token);
281  tokensFound++;
282  break;
283  default:
284  break;
285  }
286  }
287  ++i;
288  }
289  if (tokensFound == 7 && arrStop && depStop) {
290  this.totalTips++;
291  String key = start + "-" + stop;
292  if (this.valueMap.containsKey(key)) {
293  Trip myTrip = this.valueMap.get(key);
294  myTrip.incNum();
295  } else {
296  this.valueMap.put(key, new Trip(start.toString(), stop.toString()));
297  if (this.stopMap.containsKey(start.toString())) {
298  if (!this.knownPTStops.contains(start.toString())) this.knownPTStops.add(start.toString());
299  } else {
300  if (!this.unknownPTStops.contains(start.toString())) this.unknownPTStops.add(start.toString());
301  }
302  if (this.stopMap.containsKey(stop.toString())) {
303  if (!this.knownPTStops.contains(stop.toString())) this.knownPTStops.add(stop.toString());
304  } else {
305  if (!this.unknownPTStops.contains(stop.toString())) this.unknownPTStops.add(stop.toString());
306  }
307  }
308  if (this.knownPTStops.contains(stop.toString()) && this.knownPTStops.contains(start.toString()))
309  this.validTips++;
310  }
