148  public void aggregateAndWriteCSVFile(String fileName) {
149  try {

152  //the headder

548  public void readPTMatrix(String filename) {
549  FileReader in;
550 
551  final int PT_TOK_FROM = 0;
552  final int PT_TOK_TO = 1;
553  final int PT_TOK_START = 2;
554  final int PT_TOK_END = 3;
555  final int PT_TOK_INTER = 4;
556  final int PT_TOK_MW = 5;
557  final int PT_TOK_SW = 6;
558  final int PT_TOK_ACC = 7;
559  final int PT_TOK_EGR = 8;
560 
561  try {
562  HashMap<Integer, Integer> tazConverter = new HashMap<>();
563  if (this.ptTaz.length == 879) {
564  int j = 0;
565  for (int i = 1; i <= 881; ++i) {
566  if (i != 186 && i != 187) {//staken
567  tazConverter.put(i, j);
568  j++;
569  }
570  }
571  } else {
572  int j = 1;
573  for (int i = 0; i < this.ptTaz.length; ++i) {
574  tazConverter.put(j, i);
575  j++;
576  }
577  }
578  in = new FileReader(filename);
579  String line, cleanedLine = "";
580  BufferedReader input = new BufferedReader(in);
581  int lineNr = 1;
582  input.readLine();
583  lineNr++;//header
584  input.readLine();
585  lineNr++;//header
586  int from, to;
587  while (cleanedLine != null) {
588  cleanedLine = input.readLine();
589  lineNr++;
590  if (cleanedLine != null) {
591 
592  do {
593  line = cleanedLine;
594  cleanedLine = line.replace(" ", " "); // delete double spaces
595  } while (cleanedLine.length() != line.length());
596  String[] tokens = line.trim().split(" ");
597  if (tokens.length == 9) {
598 
599  from = Integer.parseInt(tokens[PT_TOK_FROM]);
600  to = Integer.parseInt(tokens[PT_TOK_TO]);
601  if (!tazConverter.containsKey(from) || !tazConverter.containsKey(to)) {
602  continue;
603  }
604  from = tazConverter.get(from);
605  to = tazConverter.get(to);
606  int tt = (Integer.parseInt(tokens[PT_TOK_END]) - Integer.parseInt(tokens[PT_TOK_START]));
607  int inter = Integer.parseInt(tokens[PT_TOK_INTER]);
608  int maxW = Integer.parseInt(tokens[PT_TOK_MW]);
609  int sumW = Integer.parseInt(tokens[PT_TOK_SW]);
610  int acc = Integer.parseInt(tokens[PT_TOK_ACC]);
611  int egr = Integer.parseInt(tokens[PT_TOK_EGR]);
612  this.ptTaz[to][from].tt = this.ptTaz[from][to].tt = tt * 60;
613  this.ptTaz[to][from].interchanges = this.ptTaz[from][to].interchanges = inter;
614  this.ptTaz[to][from].maxWaiting = this.ptTaz[from][to].maxWaiting = maxW * 60;
615  this.ptTaz[to][from].sumWaiting = this.ptTaz[from][to].sumWaiting = sumW * 60;
616  this.ptTaz[to][from].access = this.ptTaz[from][to].access = acc * 60;
617  this.ptTaz[to][from].egress = this.ptTaz[from][to].egress = egr * 60;
618 
619  } else {
620  System.out.printf("Wrong number of tokens: %d in line %d: %s\n", tokens.length, lineNr, line);
621  }
622  }
