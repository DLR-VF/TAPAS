42  public double getTravelTime(TPS_Plan plan, TPS_PlanningContext pc, TPS_ModeChoiceContext prevMCC, TPS_ModeChoiceContext nextMCC, TPS_TrafficAnalysisZone taz, TPS_ParameterClass parameterClass) {
43  double weightedTT = 0; // this value stores the weighted traveltime
44  double ttC0, ttC1, modeProb, arrModeProb, depModeProb;
45  boolean connectionFound = false;
46 
47  //now model the accessibility preference
48  ShoppingPreferenceAccessibility currentAccessibilityPreference = pc.pe
49  .getPerson().currentAccessibilityPreference;
50  if (currentAccessibilityPreference.equals(ShoppingPreferenceAccessibility.Naehe)) {
51  weightedTT = Math.max(parameterClass.getDoubleValue(ParamValue.MIN_DIST), TPS_Mode.get(ModeType.WALK)
52  .getDistance(
53  pc.pe.getPerson()
54  .getHousehold()
55  .getLocation(),
56  taz,
57  SimulationType.SCENARIO,
58  null));
59  connectionFound = true;
60  } else if (currentAccessibilityPreference.equals(ShoppingPreferenceAccessibility.Erreichbarkeit)) {
61 
62  double ttCarFrom = TPS_Mode.get(ModeType.MIT).getTravelTime(prevMCC.fromStayLocation,
63  prevMCC.toStayLocation, (int) prevMCC.fromStay.getOriginalEnd(), SimulationType.SCENARIO,
64  prevMCC.fromStay.getActCode(), prevMCC.toStay.getActCode(), plan.getPerson(), null);
65  double ttCarTo = TPS_Mode.get(ModeType.MIT).getTravelTime(nextMCC.fromStayLocation, nextMCC.toStayLocation,
66  (int) nextMCC.fromStay.getOriginalEnd(), SimulationType.SCENARIO, nextMCC.fromStay.getActCode(),
67  nextMCC.toStay.getActCode(), plan.getPerson(), null);
68  double ttPTFrom = TPS_Mode.get(ModeType.PT).getTravelTime(prevMCC.fromStayLocation, prevMCC.toStayLocation,
69  (int) prevMCC.fromStay.getOriginalEnd(), SimulationType.SCENARIO, prevMCC.fromStay.getActCode(),
70  prevMCC.toStay.getActCode(), plan.getPerson(), null);
71  double ttPTTo = TPS_Mode.get(ModeType.PT).getTravelTime(nextMCC.fromStayLocation, nextMCC.toStayLocation,
72  (int) nextMCC.fromStay.getOriginalEnd(), SimulationType.SCENARIO, nextMCC.fromStay.getActCode(),
73  nextMCC.toStay.getActCode(), plan.getPerson(), null);
74  boolean carConnectionAvailable = ttCarFrom >= 0 && ttCarTo >= 0;
75  boolean ptConnectionAvailable = ttPTFrom >= 0 && ttPTTo >= 0;
76 
77  if (!carConnectionAvailable && !ptConnectionAvailable) {
78  return -1; //no connection!
79  }
80 
81  // The WALK-mode is used to get distances on the net.
82  TPS_Mode walkMode = TPS_Mode.get(ModeType.WALK);
83  double distanceNetTo = Math.max(parameterClass.getDoubleValue(ParamValue.MIN_DIST),
84  walkMode.getDistance(prevMCC.fromStayLocation, prevMCC.toStayLocation, SimulationType.SCENARIO,
85  null));
86  double distanceNetFrom = Math.max(parameterClass.getDoubleValue(ParamValue.MIN_DIST),
87  walkMode.getDistance(nextMCC.fromStayLocation, nextMCC.toStayLocation, SimulationType.SCENARIO,
88  null));
89 
90 
91  TPS_DiscreteDistribution<TPS_Mode> arrDis = PM.getModeSet().getModeDistribution(plan, distanceNetFrom,
92  prevMCC);
93  TPS_DiscreteDistribution<TPS_Mode> dstDis = PM.getModeSet().getModeDistribution(plan, distanceNetTo,
94  nextMCC);
95 
96  double probCarFrom = arrDis.getValueByKey(TPS_Mode.get(ModeType.MIT));
97  double probCarTo = dstDis.getValueByKey(TPS_Mode.get(ModeType.MIT));
98  double probPTFrom = arrDis.getValueByKey(TPS_Mode.get(ModeType.PT));
99  double probPTTo = dstDis.getValueByKey(TPS_Mode.get(ModeType.PT));
100  double totalProb = 0;
101 
102  if (carConnectionAvailable && probCarFrom > 0 && probCarTo > 0) {
103  connectionFound = true;
104  weightedTT += (ttCarFrom + ttCarTo) * (probCarFrom + probCarTo);
105  totalProb += (probCarFrom + probCarTo);
106  }
107  if (ptConnectionAvailable && probPTFrom > 0 && probPTTo > 0) {
108  connectionFound = true;
109  weightedTT += (ttPTFrom + ttPTTo) * (probPTFrom + probPTTo);
110  totalProb += (probPTFrom + probPTTo);
111  }
112  if (connectionFound) return weightedTT / totalProb;
113  else return -1; //no connection!
114  } else { //default weight
115 
116  // The WALK-mode is used to get distances on the net.
117  TPS_Mode walkMode = TPS_Mode.get(ModeType.WALK);
118  double distanceNetTo = Math.max(parameterClass.getDoubleValue(ParamValue.MIN_DIST),
119  walkMode.getDistance(prevMCC.fromStayLocation, prevMCC.toStayLocation, SimulationType.SCENARIO,
120  null));
121  double distanceNetFrom = Math.max(parameterClass.getDoubleValue(ParamValue.MIN_DIST),
122  walkMode.getDistance(nextMCC.fromStayLocation, nextMCC.toStayLocation, SimulationType.SCENARIO,
123  null));
124  TPS_Mode mode = null;
125 
126  TPS_DiscreteDistribution<TPS_Mode> arrDis = PM.getModeSet().getModeDistribution(plan, distanceNetFrom,
127  prevMCC);
128  TPS_DiscreteDistribution<TPS_Mode> dstDis = PM.getModeSet().getModeDistribution(plan, distanceNetTo,
129  nextMCC);
130  for (ModeType mt : TPS_Mode.MODE_TYPE_ARRAY) {
131  // calculate the mode probabilities and omit forbidden modes
132  mode = TPS_Mode.get(mt);
133  arrModeProb = arrDis.getValueByKey(mode);
134  if (arrModeProb <= 0) {
135  continue;
136  }
137  depModeProb = dstDis.getValueByKey(mode);
138  if (depModeProb <= 0) {
139  continue;
140  }
141  modeProb = (arrModeProb + depModeProb) * 0.5;
142 
143  // restricted cars will not enter restricted areas
144  if (pc.carForThisPlan != null && pc.carForThisPlan.isRestricted() && taz.isRestricted() && mode.isType(
145  ModeType.MIT)) {
146  continue;
147  }
148 
149  if (mt.equals(ModeType.WALK) && (distanceNetTo + distanceNetFrom) > parameterClass.getIntValue(
150  ParamValue.MAX_WALK_DIST)) {
151  continue;
152  }
153  // get the traveltime
154  plan.setAttributeValue(TPS_Attribute.CURRENT_MODE_CODE_VOT, mode.getCode(TPS_ModeCodeType.VOT));
155 
156  ttC0 = mode.getTravelTime(prevMCC.fromStayLocation, prevMCC.toStayLocation,
157  (int) prevMCC.fromStay.getOriginalEnd(), SimulationType.SCENARIO, prevMCC.fromStay.getActCode(),
158  prevMCC.toStay.getActCode(), plan.getPerson(), null);
159  //prevMCC.carForThisPlan);
160  ttC1 = mode.getTravelTime(nextMCC.fromStayLocation, nextMCC.toStayLocation,
161  (int) nextMCC.fromStay.getOriginalEnd(), SimulationType.SCENARIO, nextMCC.fromStay.getActCode(),
162  nextMCC.toStay.getActCode(), plan.getPerson(), null);
163  //nextMCC.carForThisPlan);
164 
165  if (TPS_Mode.hasConnection(ttC0) && TPS_Mode.hasConnection(ttC1)) { // only valid travel times
166  connectionFound = true;
167  // add the traveltime
168  weightedTT += (ttC0 + ttC1) * modeProb;
169  }// no connection check
170  }// mode loop
171  }
172 
173  if (connectionFound) {
174  return weightedTT;
175  } else {
176  return -1; //not reachable!
177  }
178  }
