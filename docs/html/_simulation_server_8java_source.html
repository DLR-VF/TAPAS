374  " WHERE host = '" + hostname + "' AND end_time IS NULL", this);
375 
376  if (rset.next()) SimulationServer.shuttingDown = rset.getBoolean("shutdown");
377  rset.close();
378 
379  if (!SimulationServer.shuttingDown && !SimulationServer.shutdown) {
380  String query = "SELECT server_name FROM " + this.sTable + " WHERE server_name = '" +
381  SimulationServer.hostname + "'";
382  ResultSet rs = SimulationServer.this.dbConnector.executeQuery(query, this);
383  if (rs != null) { //sometimes the connection times out. It will be restablished on the next try...
384  if (rs.next()) {
385  query = "UPDATE " + this.sTable + " SET server_online = TRUE, server_usage = " +
386  usage.getCPUUsage() + ", server_ip = " + this.iAddr + " WHERE server_name = '" +
387  SimulationServer.hostname + "'";
388  } else {
389  query = "INSERT INTO " + this.sTable +
390  " (server_ip, server_online, server_cores, server_name) VALUES(" + this.iAddr +
391  ", TRUE, " + this.procs + ", '" + SimulationServer.hostname + "')";
392  }
393  SimulationServer.this.dbConnector.execute(query, this);
394  rs.close();
395  }
396  }
397  if (SimulationServer.shuttingDown) {
398  this.cancel();
399  Thread shutdown = new RunWhenShuttingDown();
400  shutdown.start();
401  }
402  } catch (SQLException ex) {
403  ex.printStackTrace();
404  if (ex.getNextException() != null) {
405  ex.getNextException().printStackTrace();
406  }
407  } catch (IOException | InterruptedException e) {
408  e.printStackTrace();
409  }

416  public void run() {

431 
432  for (int i = 0; i <

434  if (TPS_Main.waitForMe) {
435  System.out.println(sysoutPrefix + "Waiting for TAPAS to finish...");
436  Thread.sleep(1000);
437  } else {
438  tapas_finished = true;
439  break;
440  }
441  }
442  //Thread.sleep(2000); //this allows the timer to cancel before updating the server table.
443  //Otherwise it might happen that changes by DB-trigger functions get overwritten.
444 
445 
446  //update server process in the server_processes table. This will trigger cleanup procedures inside the DB
447  query = "UPDATE " + SimulationServer.this.getParameters().getString(
448  ParamString.DB_TABLE_PROCESSES) + " SET end_time = '" + LocalDateTime.now().toString() +
449  "', tapas_exit_ok = " + tapas_finished + " WHERE host = '" + SimulationServer.hostname +
450  "' AND end_time IS NULL";
451

455  } catch (Exception e) {
456  e.printStackTrace();
457  }
458  SimulationServer.shutdown = true;
459  SimulationServer.shuttingDown = false;
460  }
