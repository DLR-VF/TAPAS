196  public Map<Mode, ModalSplitForDistanceCategory> getModalSplitForDistanceCategory(boolean[] modalSplitForDistanceCatFilter) {
197  Map<Mode, ModalSplitForDistanceCategory> result = new HashMap<>();
198 
199  for (Mode mode : Mode.values()) {
200  ModalSplitForDistanceCategory modalSplit = new ModalSplitForDistanceCategory();
201  modalSplit.mode = mode;
202  long otherAbs = 0;
203  long tripsAll = 0;
204  ModalSplitForDistanceCategoryElement element = null;
205  for (DistanceCategory cat : DistanceCategory.values()) {
206 
207  if (element == null) element = new ModalSplitForDistanceCategoryElement();
208  element.getDistanceCategories().add(cat);
209  if (!modalSplitForDistanceCatFilter[cat.getId()]) {
210  // Wenn diese Kategorie nicht selektiert wurde, addiere die Werte zu otherAbs dazu
211  otherAbs += getModalSplitForDistanceCategoryAbs(mode, cat);
212 
213  for (Mode m : Mode.values()) {
214  for (TripIntention ti : TripIntention.values()) {
215  tripsAll += distModeTapas[m.getId()][ti.getId()][cat.getId()];
216  }
217  }
218 
219  } else {
220  // Wenn diese Kategorie selektiert wurde, addiere die Werte zu otherAbs dazu und setze diese Werte + den relativen Anteil im
221  // element.
222  otherAbs += getModalSplitForDistanceCategoryAbs(mode, cat);
223  for (Mode m : Mode.values()) {
224  for (TripIntention ti : TripIntention.values()) {
225  tripsAll += distModeTapas[m.getId()][ti.getId()][cat.getId()];
226  }
227  }
228  element.modalSplit = saveDiv(otherAbs, tripsAll) * 100.0;
229  element.modalSplitAbs = otherAbs;
230  // Füge das Element dem Modalspli hinzu und setze wieder alle Zähler auf 0
231  modalSplit.getElements().add(element);
232  element = null;
233  otherAbs = 0;
234  tripsAll = 0;
235 
236  }
237  }
238  result.put(mode, modalSplit);
239  }
240  return result;

280

284  // Ein neues Element wird immer erzeugt, wenn die aktuelle Distanzkategorie zu einer neuen Gruppe zusammengefasster Elemente gehört
285  if (element == null) element = new ModalSplitForDistanceCategoryAndTripIntentionElement();

289

298  element.putModalSplitTripIntentionSeparatedAbs(ti, modalSplitAbsOld + modalSplitAbs);

311  }
312  }
313 
314  }

316

318  for (TripIntention ti : TripIntention.values()) {
319 
320  long modalSplitAbs = getModalSplitForDistanceCategoryTripIntentionAbs(mode, cat, ti);
321 
322  if (modalSplitForDistanceCatAndTripIntentionFilterTripInt[ti.getId()]) {
323  // Wenn der Wegezweck ausgewählt wurde, zähle die absoluten Werte für diesen Wegezweck hoch

333  distModeTapas[m.getId()][ti.getId()][cat.getId()]);

337  saveDiv(modalSplitAbsOld + modalSplitAbs * 100,
338  tripsAllTripIntentionSeparated.get(ti)));
339  } else {
340  // andernfalls Zähle die Werte für "sonstiges" hoch
341  // und berechne die relativen Anteile für "sonstiges"
342  element.modalSplitOtherAbs += modalSplitAbs;
343 
344  for (Mode m : Mode.values()) {
345  tripsAll += distModeTapas[m.getId()][ti.getId()][cat.getId()];
346  }

350  }

352  modalSplit.getElements().add(element);
353  element = null;

355  tripsAll = 0;

357  }

360  }
