45  public static TPS_DiscreteDistribution<TPS_Mode> calculateDistribution(TPS_DiscreteDistribution<TPS_Mode> srcDis, TPS_Plan plan, double distanceNet, TPS_ModeChoiceContext mcc) {
46 
47  TPS_DiscreteDistribution<TPS_Mode> dstDis = getDistribution(srcDis);
48  dstDis.setValues(srcDis.getValues());
49  if (!mcc.isBikeAvailable) {
50  dstDis.setValueByKey(TPS_Mode.get(ModeType.BIKE), TPS_UtilityFunction.minModeProbability);
51  }
52  if (mcc.carForThisPlan == null) {
53  dstDis.setValueByKey(TPS_Mode.get(ModeType.MIT), TPS_UtilityFunction.minModeProbability);
54  }
55 
56  if (!dstDis.normalize()) {
57  return null;
58  }
59 
60  // calculating differences in utility for the modes with changeable attribute values;
61  // assuming the financial and time based costs for using bike, foot and misc to be fix
62 
63  // calculation for all modes the new probabilities
64  // determining first the dividing term for Pi1 = Pi0 exp(ß Delta Ci) / sum of all modes Pj0 exp(ß Delta Cj)
65  // non-variable modes keep their probability shares from the base situation
66 
67  // adding modes with variable costs;
68  for (ModeType m : EnumSet.allOf(ModeType.class)) {
69  // differences in utility need only be calculated if a mode share exists
70  TPS_Mode mode = TPS_Mode.get(m);
71  double baseProbability = dstDis.getValueByKey(mode);
72  if (baseProbability > 0.0) {
73  double delta = mode.calculateDelta(plan, distanceNet, mcc);
74  double newProbability = baseProbability * TPS_FastMath.exp(delta);
75  if (Double.isNaN(newProbability) || Double.isInfinite(newProbability)) {
76  newProbability = 0;
77  delta = mode.calculateDelta(plan, distanceNet, mcc);
78  TPS_FastMath.exp(delta);
79  }
80  dstDis.setValueByKey(mode, newProbability);
81  }
82  }
83 
84  if (!dstDis.normalize()) {
85  for (ModeType m : EnumSet.allOf(ModeType.class)) {
86  TPS_Mode mode = TPS_Mode.get(m);
87  // differences in utility need only be calculated if a mode share exists
88  double baseProbability = dstDis.getValueByKey(mode);
89  if (baseProbability > 0.0) {
90  double delta = mode.calculateDelta(plan, distanceNet, mcc);
91  delta = baseProbability * TPS_FastMath.exp(delta);
92  dstDis.setValueByKey(mode, delta);
93  }
94  }
95  }
