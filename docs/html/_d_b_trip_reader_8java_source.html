502  private TapasTrip getTripFromResult(ResultSet rs) {

521  int locIdEnd = rs.getInt("t_loc_id_end");

524 
525  // bbr type conversion

538  } catch (SQLException e) {

540  e.printStackTrace();
541  return null;
542  }

556  public void run() {
557 
558  try {
559  if (!createTripTable()) {
560 
561  isFinished = true;
562  pcs.firePropertyChange("triptable", null, false);
563 
564  Thread.currentThread().interrupt();
565  close();
566  } else {
567  pcs.firePropertyChange("triptable", null, true);
568  }
569  } catch (BadLocationException | SQLException e1) {
570  // TODO Auto-generated catch block
571  e1.printStackTrace();
572  }
573 
574  while (!isFinished && !isCancelled) {
575  try {
576  fetchNext.setInt(1, fetchStart);
577  fetchNext.setInt(2, fetchStart + FETCH_SIZE);
578 
579  ResultSet currentSet = fetchNext.executeQuery();
580 
581  isFinished = !currentSet.next();
582 
583  if (!isFinished) {
584  do {
585  TapasTrip tt = getTripFromResult(currentSet);
586  if (tt != null) {
587  try {
588  queue.put(tt);
589  } catch (InterruptedException e) {
590  close();
591  }
592  } else {
593  System.err.println("Bad trip found");
594  }
595  } while (currentSet.next() && !isCancelled);
596  }
597  } catch (SQLException e) {
598  e.printStackTrace();
599  isFinished = true;
600  }
601  fetchStart += FETCH_SIZE;
602  }
603  try {
604  queue.put(POISON_INSTANCE);
605  } catch (InterruptedException e) {
606  close();
607  }
608  close();
609  }
610  }
