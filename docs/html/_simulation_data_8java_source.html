172  private String format(long duration) {
173  String s = "";
174  int hour, minute, seconds;
175 
176  // convert from miliseconds to seconds
177  duration = duration / 1000;
178 
179 
180  // hours
181  hour = (int) (duration / 3600); //3600 seconds are one hour
182  s += hour < 10 ? "0" + hour : Integer.toString(hour);
183  s += ":";
184  duration = duration % 3600; // remainer
185 
186  //minutes
187  minute = (int) (duration / 60);
188  s += minute < 10 ? "0" + minute : Integer.toString(minute);
189  s += ":";
190  duration = duration % 60; // remainer
191 
192  // seconds
193  seconds = (int) duration;
194  s += seconds < 10 ? "0" + seconds : Integer.toString(seconds);
195 
196  return s;

436  private class ProgressCalculator {

438  private long estimatedTime;
439  private long processedHouseholds;
440  private long startTime;
441  private long lastTimeUpdate;
442  private final SimulationData simData;
443  private boolean lastStateStarted;
444  private final double[] slidingAverage;
445  private int indexSlidingAverage = 0;
446 
447  public ProgressCalculator(SimulationData sim) {
448  this.simData = sim;
449  this.lastStateStarted = this.simData.isStarted();
450  slidingAverage = new double[12];
451  Arrays.fill(this.slidingAverage, -1);
452  stateUpdate();
453  }
454 
455  public long getEstimatedTime() {
456  if (this.lastStateStarted) return Math.max(0,
457  this.estimatedTime - System.currentTimeMillis() + this.lastTimeUpdate);
458  else return 0;
459  }
460 
461  private void stateUpdate() {
462  this.estimatedTime = 0;
463  if (this.lastStateStarted) {
464  this.processedHouseholds = this.simData.sim_progress;
465  this.startTime = System.currentTimeMillis();
466  this.lastTimeUpdate = this.startTime;
467  } else {
468  this.processedHouseholds = -1;
469  this.startTime = -1;
470  this.lastTimeUpdate = 0;
471  }
472  }
473 
474  public void update() {
475  long currentTime = System.currentTimeMillis();
476 
477  if (this.simData.isStarted() != this.lastStateStarted) {
478  this.lastStateStarted = this.simData.isStarted();
479  stateUpdate();
480  } else {
481  if (this.lastStateStarted && this.simData.sim_progress != this.processedHouseholds) {
482  //calc delta
483  double number = this.simData.sim_progress - this.processedHouseholds;
484  double duration = currentTime - this.lastTimeUpdate;
485  double mSecPerHH = duration / number;
486  //store new values
487  this.lastTimeUpdate = currentTime;
488  this.processedHouseholds = this.simData.sim_progress;
489  //insert new delta
490  this.slidingAverage[this.indexSlidingAverage] = mSecPerHH;
491  //move cursor with wrap around
492  this.indexSlidingAverage = (this.indexSlidingAverage + 1) % this.slidingAverage.length;
493  //sum valid measurements
494  mSecPerHH = 0;
495  int elements = 0;
496  for (double v : this.slidingAverage) {
497  if (v > 0) {
498  mSecPerHH += v;
499  elements++;
500  }
501  }
502  mSecPerHH /= elements;
503  //calc estimate
504  this.estimatedTime = (long) ((double) (this.simData.sim_total - this.simData.sim_progress) *
505  mSecPerHH);
506  }
507  }
508  }
509  }
