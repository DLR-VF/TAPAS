74  protected List<File> chooseFile() {
75  JFileChooser fd = new JFileChooser(FileChooserHistoryManager.getLastDirectory(name));
76  fd.setFileSelectionMode(JFileChooser.FILES_ONLY);
77 
78  fd.setMultiSelectionEnabled(true);
79  fd.setVisible(true);
80 
81  fd.setDialogTitle("Dateiauswahl");
82  int value = fd.showOpenDialog(this);
83  if ((value == JFileChooser.APPROVE_OPTION) && fd.getSelectedFile() != null && fd.getSelectedFile().isFile() &&
84  fd.getSelectedFile().exists()) {
85  FileChooserHistoryManager.updateLastDirectory(name, fd.getSelectedFile().getPath().substring(0,
86  fd.getSelectedFile().getPath().lastIndexOf(File.separatorChar)));
87  return new ArrayList<>(Arrays.asList(fd.getSelectedFiles()));
88  } else if (value != JFileChooser.CANCEL_OPTION) {
89  JOptionPane.showMessageDialog(this, "Keine Datei ausgewählt: ", "Fehler", JOptionPane.ERROR_MESSAGE);
