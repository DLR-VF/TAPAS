267  public void generateIntraCellInfos(double weight) {
268  intraBeeLineFactor = new double[this.distance.length];
269  intraSpeed = new double[this.distance.length];
270 
271  int i, j, insert;
272 
273  final int length = 3;
274 
275  double[] tt = new double[length], speed = new double[length], at = new double[length], et = new double[length], dist = new double[length], bl = new double[length];
276  double intraBLF, intraV, intraAT, intraET, intraTT, intraDist;
277 
278 
279  for (i = 0; i < this.distance.length; ++i) {
280  //init the mins
281  for (j = 0; j < length; ++j) {
282  tt[j] = 1e100;
283  speed[j] = 1e100;
284  at[j] = 1e100;
285  et[j] = 1e100;
286  dist[j] = 1e100;
287  bl[j] = 1e100;
288  }
289  //find the mins
290  for (j = 0; j < this.distance[0].length; ++j) {
291  if (i != j) { //diagonal is zero!
292  if (isPT) {
293  checkMultiMin(this.accessTime[i][j], at); // calc three shortest access times
294  checkMultiMin(this.egressTime[i][j], et); // calc three shortest egress times
295  }
296  insert = checkMultiMin(this.distance[i][j], dist);
297  if (insert >= 0) {
298  insertMinAt(this.travelTime[i][j], tt, insert);
299  insertMinAt(this.beeline[i][j], bl, insert);
300  if (!isPT) insertMinAt(this.speed[i][j], speed, insert);
301  }
302  }
303  }
304  // init the avg
305  intraBLF = 0;
306  intraV = 0;
307  intraAT = 0;
308  intraET = 0;
309  intraTT = 0;
310  intraDist = 0;
311  //build average
312  for (j = 0; j < length; ++j) {
313  intraTT += tt[j];
314  intraDist += dist[j];
315  if (isPT) {
316  intraAT += at[j];
317  intraET += et[j];
318  } else {
319  intraBLF += dist[j] / bl[j];
320  intraV += speed[j];
321  }
322  }
323  intraTT *= weight / length; //intraTT is half the average to the n closest cells
324  intraDist *= weight / length; //intraDist is half the average to the n closest cells
325  this.travelTime[i][i] = intraTT;
326  this.distance[i][i] = intraDist;
327 
328  if (isPT) {
329  // access/egress times are not influenced by the cell-size: take the average of the three lowest times
330  this.accessTime[i][i] = intraAT / length;
331  this.egressTime[i][i] = intraET / length;
332  intraSpeed[i] = intraDist / intraTT; //average speed
333  } else {
334  intraBeeLineFactor[i] = intraBLF / length;
335  intraSpeed[i] = intraV / length;
336  }
337  }
338  }

472  public void mergeMatrixPair(double[][] oldVals, double[][] newVals) {
473  int i, j;
474  double oldVal, newVal;
475  double error, maxError = 0, minError = 1e100, avgError = 0;
476  final int size = this.travelTime.length;
477  final double normalizationFactor = 1.0 / (1.0 + this.weight);
478  for (i = 0; i < size; ++i) {
479  for (j = 0; j < size; ++j) {
480  if (i != j) { //do not update the diagonal! these values are calculated separately!
481  oldVal = oldVals[i][j];
482  newVal = newVals[i][j];
483  error = oldVal - newVal;
484  if (newVal > 0.0) {//ignore invalid cells (no trafic)
485  maxError = Math.max(maxError, error);
486  minError = Math.min(minError, error);
487  avgError += error;
488  newVals[i][j] = (oldVal + this.weight * newVal) * normalizationFactor;
489  }
490  }
491  }
492  }
493  avgError /= size * size;
494  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
495  TPS_Logger.log(SeverenceLogLevel.INFO,
496  "max error: " + maxError + " minError: " + minError + " avg error: " + avgError);
497  }
498  }

513  public void readCSVFile(String fileName) throws IOException {
514  FileReader in = null;
515  BufferedReader input = null;
516  String line = "";
517  try {
518  int fromTVZ, toTVZ, tvzCounter = 0;
519  tvzCounter = this.idToIndex.size();
520  String delimiter = ";";
521  boolean oFormat = false;
522  if (tvzCounter == 0) {
523  in = new FileReader(fileName);
524  input = new BufferedReader(in);
525  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
526  TPS_Logger.log(SeverenceLogLevel.INFO, "File opened: " + fileName);
527  }
528  //read tvzs
529  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
530  TPS_Logger.log(SeverenceLogLevel.INFO, "Analyzing TVZ-IDs: ");
531  }
532 
533  while ((line = input.readLine()) != null) {
534  if (line.startsWith("$") || line.startsWith("*")) { // comment or visum file-format
535  if (line.startsWith("$O;D")) {
536  delimiter = " "; //O-Format hast two spaces as delimiter
537  oFormat = true;
538  }
539  if (line.startsWith("$NAMES")) //end of o-File reached
540  break;
541  continue;
542  }
543  StringTokenizer tok = new StringTokenizer(line.trim(), delimiter);
544  //check format
545  if (!((tok.countTokens() == 7 && isPT && !oFormat) ||
546  (tok.countTokens() == 6 && !isPT && !oFormat) || (tok.countTokens() == 3 && oFormat)))
547  continue;
548 
549  //get from
550  fromTVZ = Integer.parseInt(tok.nextToken().trim());
551 
552  if (idToIndex.get(fromTVZ) == null) {//new tvz
553  idToIndex.put(fromTVZ, tvzCounter);
554  indexToId.put(tvzCounter, fromTVZ);
555  tvzCounter++;
556  }
557 
558  //get to
559  toTVZ = Integer.parseInt(tok.nextToken().trim());
560  if (idToIndex.get(toTVZ) == null) {//new tvz
561  idToIndex.put(toTVZ, tvzCounter);
562  indexToId.put(tvzCounter, toTVZ);
563  tvzCounter++;
564  }
565  }
566  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
567  TPS_Logger.log(SeverenceLogLevel.INFO, "Found " + tvzCounter + " TVZ-IDs");
568  }
569  input.close();
570  in.close();
571  }
572 
573 
574  //prepare arrays
575  travelTime = new double[tvzCounter][tvzCounter];
576  speed = new double[tvzCounter][tvzCounter];
577  accessTime = new double[tvzCounter][tvzCounter];
578  egressTime = new double[tvzCounter][tvzCounter];
579  distance = new double[tvzCounter][tvzCounter];
580  beeline = new double[tvzCounter][tvzCounter];
581 
582  //new local variables
583  double[] tt = new double[3], v = new double[3], at = new double[3], et = new double[3], dist = new double[3], bl = new double[3];
584  int indexFrom, indexTo;
585  //open input
586  in = new FileReader(fileName);
587  input = new BufferedReader(in);
588  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
589  TPS_Logger.log(SeverenceLogLevel.INFO, "File opened: " + fileName);
590  }
591  tt[1] = v[1] = at[1] = et[1] = dist[1] = bl[1] = 99999;
592  tt[2] = v[2] = at[1] = et[2] = dist[2] = bl[2] = 0;
593  while ((line = input.readLine()) != null) {
594  if (line.startsWith("$") || line.startsWith("*")) { // comment or visum file-format
595  if (line.startsWith("$O;D")) {
596  delimiter = " "; //O-Format hast two spaces as delimiter
597  oFormat = true;
598  speed = null;
599  accessTime = null;
600  egressTime = null;
601  distance = null;
602  beeline = null;
603 
604  }
605  if (line.startsWith("$NAMES")) //end of o-File reached
606  break;
607  continue;
608  }
609  StringTokenizer tok = new StringTokenizer(line.trim(), delimiter);
610  //check format
611  if (!((tok.countTokens() == 7 && isPT && !oFormat) || (tok.countTokens() == 6 && !isPT && !oFormat) ||
612  (tok.countTokens() == 3 && oFormat))) continue;
613 
614  //get from
615  fromTVZ = Integer.parseInt(tok.nextToken().trim());
616  indexFrom = idToIndex.get(fromTVZ);
617 
618  //get to
619  toTVZ = Integer.parseInt(tok.nextToken().trim());
620  indexTo = idToIndex.get(toTVZ);

622  //get acc t
623  at[0] = Double.parseDouble(tok.nextToken().trim());
624  // get bl
625  bl[0] = Double.parseDouble(tok.nextToken().trim());
626  //get egr t

628  //get travel dist

630  //get tt
631  tt[0] = Double.parseDouble(tok.nextToken().trim());
632 
633  //incl check
634  setValue(travelTime, indexFrom, indexTo, tt[0] != 999999.0 ? tt[0] * 60.0 : -1, true);
635  setValue(accessTime, indexFrom, indexTo, at[0] != 999999.0 ? at[0] * 60.0 : -1, true);
636  setValue(egressTime, indexFrom, indexTo, et[0] != 999999.0 ? et[0] * 60.0 : -1, true);
637  setValue(distance, indexFrom, indexTo, dist[0] != 999999.0 ? dist[0] * 1000.0 : 999999, true);
638  setValue(beeline, indexFrom, indexTo, bl[0] != 999999.0 ? bl[0] * 1000.0 : 999999, true);
639  if (indexFrom != indexTo) {
640  calcMinMaxValue(tt);
641  calcMinMaxValue(at);
642  calcMinMaxValue(et);
643  calcMinMaxValue(dist);
644  calcMinMaxValue(bl);
645  }
646  } else {
647  //get tt
648  tt[0] = Double.parseDouble(tok.nextToken().trim());
649  setValue(travelTime, indexFrom, indexTo, tt[0] != 999999.0 ? tt[0] * 60.0 : -1, true);

651  v[0] = Double.parseDouble(tok.nextToken().trim());
652  dist[0] = Double.parseDouble(tok.nextToken().trim());
653  bl[0] = Double.parseDouble(tok.nextToken().trim());

655  setValue(distance, indexFrom, indexTo, dist[0] != 999999.0 ? dist[0] * 1000.0 : 999999, true);
656  setValue(beeline, indexFrom, indexTo, bl[0] != 999999.0 ? bl[0] * 1000.0 : 999999, true);
657  }
658 
659  if (indexFrom != indexTo) {
660  calcMinMaxValue(tt);
661  if (!oFormat) {
662  calcMinMaxValue(v);
663  calcMinMaxValue(dist);
664  calcMinMaxValue(bl);
665  }
666  }
667  }
668  }
669  if (TPS_Logger.isLogging(SeverenceLogLevel.INFO)) {
670  TPS_Logger.log(SeverenceLogLevel.INFO, "travelTime min/max: " + tt[1] + " " + tt[2]);
671  if (isPT) {
672  TPS_Logger.log(SeverenceLogLevel.INFO, "accessTime min/max: " + at[1] + " " + at[2]);
673  TPS_Logger.log(SeverenceLogLevel.INFO, "egressTime min/max: " + et[1] + " " + et[2]);
674  } else {
675  TPS_Logger.log(SeverenceLogLevel.INFO, "speed min/max: " + v[1] + " " + v[2]);
676  }
677  TPS_Logger.log(SeverenceLogLevel.INFO, "distance min/max: " + dist[1] + " " + dist[2]);
678  TPS_Logger.log(SeverenceLogLevel.INFO, "Beeline min/max: " + bl[1] + " " + bl[2]);
679  }
680  } finally {
681  try {
682  if (input != null) input.close();
683  if (in != null) in.close();
684  }//try
685  catch (IOException ex) {
686  TPS_Logger.log(SeverenceLogLevel.ERROR, " Could not close : " + fileName);
687  throw new IOException(ex);
688  }//catch
689  }//finally
690  }

719  private void readMatrix(ParamString matrixName, ParamMatrixMap matrix, SimulationType simType, int sIndex) {

722  try {
723  query = "SELECT \"matrixMap_num\", \"matrixMap_matrixNames\", \"matrixMap_distribution\" FROM " +

742  " SQL query: " + query);
