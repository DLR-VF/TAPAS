54  public static void calcTopX(double[][] matrix, int numOfElem, double weight) {
55  int i, j;
56  if (matrix == null || matrix.length == 0 || matrix.length < (numOfElem + 1) ||
57  matrix.length != matrix[0].length) return;
58 
59  for (i = 0; i < matrix.length; ++i) {
60  List<Double> row = new ArrayList<>();
61  for (j = 0; j < matrix[i].length; ++j) {
62  if (i != j) row.add(matrix[i][j]);
63  }
64  Collections.sort(row);
65  matrix[i][i] = 0;
66  for (j = 0; j < numOfElem; ++j) {
67  matrix[i][i] += row.get(j);
68  }
69  matrix[i][i] *= weight / (double) numOfElem;
70  }
71  }

73  public static void convertToCartesian(TPS_Coordinate g, double[] p) {
74  double phi, rho, z;
75  rho = Math.toRadians(g.getValue(0));
76  phi = Math.toRadians(g.getValue(1));
77  z = Math.cos(phi); //verwende so wenig trigonometische Funktionen wie möglich!
78  p[0] = earthRadius * z * Math.cos(rho);
79  p[1] = earthRadius * z * Math.sin(rho);
80  p[2] = earthRadius * Math.sin(phi);
81  }

103  public static double getDistance(Locatable startLoc, Locatable endLoc, TPS_Mode mode, SimulationType simType) {
104  double dist = -1;

106  /*
107  * TODO RITA: check if this calculation method is valid and reasonable Determines the distance in meters between two

109  * calculated by the coordinates of the locations. Returns at least the in the configuration specified minimal

128  // Hier: Rückkonvertierung zu Luftliniendistanz Multiplizierung mit den "Luftlinienfaktor"

162  public static double getDistance(double lat1, double lon1, double lat2, double lon2) {
163  // changes of mark_ma
164  // MANTIS ENTRY 0001024
165  // introduced beeline distance calculation for a ideal sphere with the same volume as the earth -> median radius
166  // @see http://de.wikipedia.org/wiki/Entfernungsberechnung
167 
168  double fs = Math.sin(Math.toRadians(lon1)) * Math.sin(Math.toRadians(lon2));
169  double fc = Math.cos(Math.toRadians(lon1)) * Math.cos(Math.toRadians(lon2));
170  double x = fs + fc * Math.cos(Math.toRadians(lat1 - lat2));
171  x = Math.max(-1.0,
172  Math.min(1.0, x)); //if c0 == c1 values slightly bigger than 1 occur. FIX: chop to valid acos-range
173  return TPS_FastMath.acos(x) * earthRadius;
174  }

185  public static double getDistance(TPS_Coordinate g0, TPS_Coordinate g1, TPS_Coordinate point, double min_distance) {
186  if (isSameLocation(g0, g1)) { // give point dist
187  return getDistance(g0, point, min_distance);
188  } else {
189 
190  double phi, rho, z;
191  double[] r = new double[3], p = new double[3], t = new double[3], s = new double[3], l = new double[3];
192  double actMaxDist;
193 
194  //konvertierung geograph. kugel->karthesisch (sin und cos für phi vertauscht)
195  // the point
196  convertToCartesian(point, p);
197 
198  //the base of the line
199  convertToCartesian(g0, t);
200 
201  //the 2nd point on the line
202  convertToCartesian(g1, s);
203 
204  //the direction of the line
205  for (int i = 0; i < 3; ++i) r[i] = s[i] - t[i];
206 
207  // die kürzeste Distanz zwischen einem Punkt und einer Linie ist der Abstand des Lotfußpunktes l auf der Linie zum Punkt.
208  // zusätzlich muss in diesem Fall der Lotfußpunkt l zwischen g0=t und g1=s liegen!
209  // die Gerade g ist deffiniert durch g = t + z*(s-t)
210  // Der Lotfußpunkt l liegt zwischen t uns s, wenn z zwischen 0 und 1 liegt
211  // Des weiteren ist das Skalarprodukt aus (s-t)(l-p)=0 (Senkrechtes Lot)
212  // aus der Gleichung (s-t)*(t+z*(s-t)-p)=0 ergibt sich für den Parameter z:
213  z = ((p[0] - t[0]) * r[0] + (p[1] - t[1]) * r[1] + (p[2] - t[2]) * r[2]) /
214  (r[0] * r[0] + r[1] * r[1] + r[2] * r[2]);
215 
216  if (z <= 0 || z >= 1) { //lot nciht zwischen g0 und g1
217  return Math.min(getDistance(g0, point, min_distance), getDistance(g1, point, min_distance));
218  } else { //lot zwischen g0 und g1
219 
220  // der Lotfußpunkt ist definiert durch:
221  for (int i = 0; i < 3; ++i) l[i] = t[i] + z * r[i];
222 
223  //konvertierung karthesisch->geograph. kugel, radius wird vergessen und auf die erdoberfläche gesetzt!
224  actMaxDist = Math.sqrt(l[0] * l[0] + l[1] * l[1] + l[2] * l[2]);
225  phi = Math.toDegrees(Math.asin(l[2] / actMaxDist));
226  rho = Math.toDegrees(Math.atan2(l[1], l[0]));
227 
228  //dieser punkt ist der Lotfußpunkt der direkten Linie von g0 nach g1 und wird dann auf dem Spheroiden Erde projiziert, aber das mach bei unseren Koodinaten nichts ;)
229  TPS_Coordinate lot = new TPS_Coordinate(rho, phi);
230  return getDistance(lot, point, min_distance);
231  }
232  }
