138  public static void log(Class<?> callerClass, HierarchyLogLevel hLog, SeverenceLogLevel sLog, String text, Throwable throwable) {
139  if (!shallBeLogged(sLog)) {
140  return;
141  }
142  for (int i = 0; i < throwable.getStackTrace().length; ++i) {
143  text = text.concat("\n" + throwable.getStackTrace()[i].toString());
144  }
145  if (LOGGING == null) {
146  LOG_ITEMS.add(new LogItem(callerClass, hLog, sLog, text, throwable));
147 // System.out.println("Stored log: "+text);
148 // throwable.printStackTrace();
149  } else if (shallBeLogged(hLog) && SEVERENCE_LOG_LEVEL_MASK.includes(sLog)) {
150  LOGGING.log(callerClass, sLog, text, throwable);
151  }
152  }

262  public static void setLoggingClass(String loggingClass, TPS_ParameterClass parameterClass) {
263  if (LOGGING == null) {
264  LOGGING = new TPS_Log4jLogger(parameterClass);
265  HIERARCHY_LOG_LEVEL_MASK = HierarchyLogLevel.valueOf(
266  parameterClass.getString(ParamString.HIERARCHY_LOG_LEVEL_MASK));
267  SEVERENCE_LOG_LEVEL_MASK = SeverenceLogLevel.valueOf(
268  parameterClass.getString(ParamString.SEVERENCE_LOG_LEVEL_MASK));
269 
270  for (LogItem logItem : LOG_ITEMS) {
271  if (logItem.throwable == null) {
272  log(logItem.callerClass, logItem.hLog, logItem.sLog, logItem.text);
273  } else {
274  log(logItem.callerClass, logItem.hLog, logItem.sLog, logItem.text, logItem.throwable);
275  }
276  }
277  LOG_ITEMS.clear();
278  LOG_ITEMS = null;
279  } else {
280  TPS_Logger.log(HierarchyLogLevel.CLIENT, SeverenceLogLevel.WARN, "Try to set the log level a second time");
281  }
282  }
