128  int[][] readMatrix(String name) {

132  String query = "";
133  try {
134  query = "SELECT matrix_values FROM core." + region + "_matrices WHERE matrix_name = '" + name + "'";
135  ResultSet rs = this.dbCon.executeQuery(query, this);
136  if (rs.next()) {
137  int[] numbers = TPS_DB_IO.extractIntArray(rs, "matrix_values");
138  //parse array
139  if (numbers.length == size * size) {
140  int k = 0;
141  for (int i = 0; i < size; ++i) {
142  for (int j = 0; j < size; ++j, ++k) {
143  mat[i][j] = numbers[k];
144  }
145  }
146  } else {
147  return null;
148  }
149  }
150  } catch (SQLException e) {
151  System.out.println("SQL error! Query: " + query);
152  e.printStackTrace();
