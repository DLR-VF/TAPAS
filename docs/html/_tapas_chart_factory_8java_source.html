147  public static MultiKeyMap createModalSplitDistanceCategorySeparated(StatisticPOJO statistics, boolean[] modalSplitForDistanceCatFilter, boolean asPie, ChartTheme theme) {
148  MultiKeyMap result = new MultiKeyMap();
149  int maxDistanceCategoryGroup = modalSplitForDistanceCatFilter.length - countFalses(
150  modalSplitForDistanceCatFilter);
151  for (RegionPOJO region : statistics.getAnalyses()) {
152  Map<Mode, ModalSplitForDistanceCategory> modalSplits = region.getModalSplitForDistanceCategory(
153  modalSplitForDistanceCatFilter);
154 
155  for (int currentDistanceCategoryGroup = 0;
156  currentDistanceCategoryGroup < maxDistanceCategoryGroup; currentDistanceCategoryGroup++) {
157  AbstractDataset dataSet = asPie ? new DefaultPieDataset() : new DefaultCategoryDataset();
158 
159  String distanceGroupName = null;
160  List<DistanceCategory> categories = null;
161  for (Mode mode : Mode.values()) {
162 
163  ModalSplitForDistanceCategory modalSplit = modalSplits.get(mode);
164  ModalSplitForDistanceCategoryElement element = modalSplit.getElements().get(
165  currentDistanceCategoryGroup);
166  if (element.getModalSplit() > 0) {
167  if (distanceGroupName == null)
168  distanceGroupName = DistanceCategory.createDistanceCategoriesName(
169  element.getDistanceCategories());
170  if (categories == null) categories = element.getDistanceCategories();
171 
172  if (asPie) {
173  ((DefaultPieDataset) dataSet).setValue(mode.getDescription(), element.getModalSplit());
174  } else {
175  ((DefaultCategoryDataset) dataSet).addValue(element.getModalSplit(), mode.getDescription(),
176  region.getRegionCode().getDescription());
177  }
178  }
179  }
180 
181  JFreeChart chart;
182  if (distanceGroupName != null && categories != null) {
183  if (asPie) chart = createChartPie(
184  "Modalsplit - " + region.getRegionCode().getDescription() + " - " + distanceGroupName,
185  (DefaultPieDataset) dataSet, theme);
186  else chart = createModalSplitChartBar(
187  "Modalsplit - " + region.getRegionCode().getDescription() + " - " + distanceGroupName,
188  (DefaultCategoryDataset) dataSet, theme);
189  result.put(region.getRegionCode(), categories, chart);
190  }
191  }
192  }
193  return result;
194  }
