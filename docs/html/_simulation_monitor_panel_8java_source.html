209  private void updateData() {
210  synchronized (table) {
211  HashMap<String, Integer> simulations = getSimulations();
212  HashSet<String> keySet = new HashSet<>(simulations.keySet());
213 
214  String simkey;
215  try {
216  ResultSet rs = connection.executeQuery(
217  "SELECT * FROM " + this.connection.getParameters().getString(ParamString.DB_TABLE_SIMULATIONS) +
218  " ORDER BY timestamp_insert", this);
219 
220  while (rs.next()) {
221  simkey = rs.getString("sim_key");
222  if (keySet.remove(simkey)) { // key existed, update
223  Object[] row = getRowFromSimulation(new SimulationData(rs));
224  int simIdx = simulations.get(simkey);
225  for (int i = 0; i < table.getColumnCount(); ++i) {
226  table.setValueAt(row[i], simIdx, i);
227  }
228 
229  } else {// new data
230  tableModel.addRow(getRowFromSimulation(new SimulationData(rs)));
231  }
232  }
233  rs.close();
234  } catch (SQLException e) {
235  System.err.println("Error while updating table data.");
236  e.printStackTrace();
237  }
238 
239  for (String s : keySet) {// remove old rows
240  removeSimulation(s);
241  }
242  }
243  }
