37  private double calculateAcceptanceProbBudgetConstraints(TPS_Plan plan) {
38  double budgetAcceptanceProbability = 0.0;
39  // children under 25 years living at a household with their parents
40  // don't have an own budget due to data restrictions;
41  // default value: 99999 each for variable costs pt and car
42 
43  double myFinancialBudgetVariable =
44  plan.getPerson().getBudget() / 30.5; //30.5 days per month in average: 365d/12m
45 
46  // Default value of 99999 each for variable budget pt and car for children resulting in following value for
47  // automated acceptance : 2* 99999 / 30 when calculated per day
48  // this value is store
49  if (myFinancialBudgetVariable == 2 * 99999 / 30) { // TODO integer compared to float?
50  budgetAcceptanceProbability = 1;
51  } else {
52  // Anschmiegeparameter der EVA1-Funktion
53  double EBottom = plan.getParameters().getDoubleValue(ParamValue.FINANCE_BUDGET_E);
54  // EVA1-Parameter, der das Absinken der Akzeptanz bei geringer Überschreitung beeinflußt
55  double FTop = plan.getParameters().getDoubleValue(ParamValue.FINANCE_BUDGET_F);
56  // Wendepunkt der Funktion; ein Wendepunkt von 0.5 besagt, dass es bei einer Überschreitung der Budget-
57  // vorgaben um 50 % zu einem starken Abfall der Akzeptanz des Plans kommt
58  double TurningPoint = plan.getParameters().getDoubleValue(ParamValue.FINANCE_BUDGET_WP);
59  // Verhältnis des Ausgaben zum Budget; nur bei Überschreitung (>1) wird auf Akzeptanz getestet, sonst
60  // immer 1
61  double ratio = plan.getTravelCosts() / myFinancialBudgetVariable;
62 
63  if (ratio <= 1) {
64  budgetAcceptanceProbability = 1;// Tagepläne bei denen die finanziellen Grenzen eingehalten werden
65  // werden
66  // akzeptiert
67  } else {
68  ratio = ratio - 1; // Basis ist immer 0, Abweichung darauf basierend
69  budgetAcceptanceProbability = TPS_GX.calculateEVA1Acceptance(ratio, EBottom, FTop, TurningPoint);
70  }
71  if (TPS_Logger.isLogging(SeverenceLogLevel.DEBUG)) {
72  TPS_Logger.log(SeverenceLogLevel.DEBUG,
73  "Plan with financial costs=" + plan.getTravelCosts() + " and person's budget=" +
74  myFinancialBudgetVariable + " leads to [ratio=" + ratio + ", acceptance=" +
75  budgetAcceptanceProbability + "]");
76  }
77  }
78 
79  plan.setBudgetAcceptanceProbability(budgetAcceptanceProbability);
80 
81  return budgetAcceptanceProbability;
82  }

93  private double calculateAcceptanceProbTimeConstraints(TPS_Plan plan) {
94  double timeAcceptanceProbability = 0.0;
95  // Anschmiegeparameter der EVA1-Funktion
96  double EBottom = plan.getParameters().getDoubleValue(ParamValue.TIME_BUDGET_E);
97  // EVA1-Parameter, der das Absinken der Akzeptanz bei geringer Überschreitung beeinflusst
98  double FTop = plan.getParameters().getDoubleValue(ParamValue.TIME_BUDGET_F);
99  // stattdessen Infos aus Tagebuchklasse!
100  // double TurningPoint = myTurningPointTime;
101 
102  double myTimeBudget = 1.0 / plan.getScheme().getTimeUsageAVG();
103  double TurningPoint = plan.getScheme().getTimeUsageSTD() * myTimeBudget;
104 
105  // prozentuale Abweichungen müssen normiert werden, um die EVA1-Funktion verwenden zu können (berücksichtigt nur
106  // positive Aufwände)
107  // heißt: zeitliche Über- als Unterschreitungen werden gleich behandelt
108  double ratio = plan.getTravelDuration() * myTimeBudget;
109 
110  if (ratio == 0 || ratio == 1 || ratio <= 1.0) {
111  timeAcceptanceProbability = 1.0; // Tagespläne ohne Trips haben keine zeitlichen Kosten; wenn die zeitlichen
112  // Kosten stimmen
113  // direkt akzeptieren
114  } else {
115  ratio = Math.abs(ratio - 1.0);
116  timeAcceptanceProbability = TPS_GX.calculateEVA1Acceptance(ratio, EBottom, FTop, TurningPoint);
117  }
118  if (TPS_Logger.isLogging(SeverenceLogLevel.DEBUG)) {
119  TPS_Logger.log(SeverenceLogLevel.DEBUG,
120  "Plan with time costs=" + plan.getTravelDuration() + " and person's budget=" + 1.0 / myTimeBudget +
121  " leads to [ratio=" + ratio + ", acceptance=" + timeAcceptanceProbability + "]");
122  }
123 
124  plan.setTimeAcceptanceProbability(timeAcceptanceProbability);
125 
126  return timeAcceptanceProbability;
127  }

171  private boolean isPlanAcceptedByOverallTravelTime(TPS_Plan plan) {
172  boolean accepted = false;
173  double acceptance = 1.0;
174  double originalDuration = plan.getScheme().getOriginalTravelDuration();
175  double realDuration = plan.getTravelDuration();
176  double ratio = 0;
177  if (Double.compare(0.0, originalDuration) == 0 || Double.compare(0.0, realDuration) == 0) {
178  // day plans without trips do not have travel times associated --> accept directly
179  accepted = true;
180  plan.setAcceptanceProbability(1);
181 
182  } else {
183  // Anschmiegeparameter der EVA1-Funktion
184  double EBottom = plan.getParameters().getDoubleValue(ParamValue.OVERALL_TIME_E);
185  // EVA1-Parameter, der das Absinken der Akzeptanz bei geringer Überschreitung beeinflußt
186  double FTop = plan.getParameters().getDoubleValue(ParamValue.OVERALL_TIME_F);
187  double TurningPoint = plan.getParameters().getDoubleValue(ParamValue.MAX_TIME_DIFFERENCE);
188 
189  ratio = realDuration / originalDuration;
190  ratio = Math.abs(ratio -
191  1.0); //even shorter plans should be rejected due to the magic "84min travel time"-postulation
192 // if(realDuration <originalDuration){
193 // ratio /=2; //shorter durations are only "halve bad"
194 // }
195 
196  acceptance = TPS_GX.calculateEVA1Acceptance(ratio, EBottom, FTop, TurningPoint);
197  if (Double.isNaN(acceptance) || Double.isInfinite(acceptance)) {
198  TPS_Logger.log(SeverenceLogLevel.FATAL, "Plan has NaN acceptance!");
199  plan.setPlanFeasible(false);
200  accepted = false;
201  } else {
202  plan.setAcceptanceProbability(acceptance);
203  accepted = Randomizer.random() < acceptance;
204  }
205  }
206  if (TPS_Logger.isLogging(SeverenceLogLevel.DEBUG)) {
207  TPS_Logger.log(SeverenceLogLevel.DEBUG,
208  "Plan with travel times [orig=" + originalDuration + ", real=" + realDuration +
209  "] leads to [ratio=" + ratio + ", acceptance=" + acceptance + "] is accepted => " +
210  accepted);
211  }
212 
213  return accepted;
214  }
