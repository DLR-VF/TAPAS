91  public void makeModifiedFleet() {
92  String query = "";
93  ResultSet rs;
94  try {
95  //load cars
96  query = "SELECT car_id, kba_no, engine_type, is_company_car, car_key, emission_type, restriction, fix_costs FROM " +
97  this.tableName + " WHERE car_key = '" + this.keyVal + "'";
98  rs = dbCon.executeQuery(query, this);
99  while (rs.next()) {
100  Car car = this.extractCarData(rs);
101  this.cars.put(car.car_id, car);
102  }
103  rs.close();
104 
105  //modify cars
106  for (Car car : this.cars.values()) {
107  double random = Math.random();
108  ValueDistribution dist = this.kbaValueMapping.get(car.kba_no);
109  if (dist == null) {
110  System.out.println(car.kba_no);
111  } else {
112  if (dist.probability >= random) {
113  car.kba_no = this.kbaValueMapping.get(car.kba_no).newKBA;
114  }
115  }
116  car.car_key = this.newKeyVal;
117  }
118 
119  //delete old entries
120  query = "DELETE FROM " + this.tableName + " WHERE car_key = '" + this.newKeyVal + "'";
121  dbCon.execute(query, this);
122 
123  //save back with a prepared statement
124 
125  PreparedStatement pS;
126 
127 
128  query = "INSERT INTO " + this.tableName +
129  " ( car_id, kba_no, engine_type, is_company_car, car_key, emission_type, restriction, fix_costs ) VALUES (?,?,?,?,?,?,?,?)";
130  pS = this.dbCon.getConnection(this).prepareStatement(query);
131  int count = 0;
132  //now parse the values in the prepared statement
133  for (Car car : this.cars.values()) {
134 
135  pS.setInt(1, car.car_id);
136  pS.setInt(2, car.kba_no);
137  pS.setInt(3, car.engine_type);
138  pS.setBoolean(4, car.is_company_car);
139  pS.setString(5, car.car_key);
140  pS.setInt(6, car.emission_type);
141  pS.setBoolean(7, car.restriction);
142  pS.setDouble(8, car.fix_costs);
143  pS.addBatch();
144  count++;
145  if (count > 1000) {
146  count = 0;
147  pS.executeBatch();
148  pS.clearBatch();
149  }
150  }

156  } catch (SQLException e) {
