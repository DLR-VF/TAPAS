72  private void generateComparisonList() {
73 
74  if (is_correct_input) {
75  Task<ObservableList<SimParamComparatorObject>> tsk = new Task<ObservableList<SimParamComparatorObject>>() {
76  @Override
77  protected ObservableList<SimParamComparatorObject> call() throws Exception {
78  List<SimParamComparatorObject> lst = new ArrayList<>();
79  ResultSet rs;
80 
81  //
82  String query = "SELECT Z.param_key as key, A.param_value as sim1 , B.param_value as sim2\r\n" +
83  "FROM (SELECT DISTINCT param_key FROM simulation_parameters WHERE param_key IS NOT NULL AND param_key != '') AS Z " +
84  "FULL OUTER JOIN (SELECT * FROM simulation_parameters WHERE sim_key = '" + simkey1 +
85  "') AS A " + "ON Z.param_key = A.param_key " +
86  "FULL OUTER JOIN (SELECT * FROM simulation_parameters WHERE sim_key = '" + simkey2 +
87  "') AS B " + "ON Z.param_key = B.param_key " + "ORDER BY Z.param_key";
88  rs = conn.executeQuery(query, SimParamComparatorController.class);
89 
90  Function<Object, String> handle_nulls = str -> str == null ? "" : str.toString();
91  Map<String, String> tps_parameters = new HashMap<>();
92 
93  Arrays.stream(ParamString.values()).forEach(param -> tps_parameters.put(param.name(),
94  handle_nulls.apply((new TPS_ParameterClass()).paramStringClass.getPreset(param))));
95  Arrays.stream(ParamValue.values()).forEach(param -> tps_parameters.put(param.name(),
96  handle_nulls.apply((new TPS_ParameterClass()).paramValueClass.getPreset(param))));
97  Arrays.stream(ParamFlag.values()).forEach(param -> tps_parameters.put(param.name(),
98  handle_nulls.apply((new TPS_ParameterClass()).paramFlagClass.getPreset(param))));
99 
100  while (rs.next()) {
101  lst.add(new SimParamComparatorObject(rs.getString("key"),
102  tps_parameters.get(rs.getString("key")), handle_nulls.apply(rs.getString("sim1")),
103  handle_nulls.apply(rs.getString("sim2"))));
104  }
105  rs.close();
106 
107 
108  return FXCollections.observableList(lst);
109  }
110  };
111 
112  tsk.run();
113 
114  tsk.setOnSucceeded(event -> {
115  try {
116  tablecontent = new SimpleListProperty<>(tsk.get());
117  lst = new FilteredList<>(tablecontent);
118  tvSimComparator.setItems(lst);
119  cbFilter.valueProperty().addListener((obsv, oldv, newv) -> lst.setPredicate(newv.getFilter()));
120  cbFilter.getSelectionModel().selectFirst();
121  } catch (InterruptedException | ExecutionException e) {
122  e.printStackTrace();
123  }
124  });
125  } else throw new IllegalArgumentException("At least one of the input arguments is either null or empty String");
126  }
